\begin{recipe}{Kalkoen in spek}
\kicker[Hoofdgerecht,Vlees,Amerikaans,Kerst]{Hoofdgerecht · vlees · Amerikaans · kerst}
\index[register]{Kalkoen in spek}
\index[register]{Kalkoen}
\index[register]{Spek}
\index[register]{Kerst}

\meta{3 uur + 30 minuten rusten \dvd Voor 10--12 personen \dvd Oven 220\,°C \& 165\,°C}

\lettrine{T}{wee} keer gemaakt voor kerst en telkens goed bevallen. Het spek
houdt de kalkoen sappig zonder dat je hoeft te pekelen of bedruipen, en levert
tegelijk een krokante ``huid'' op. Wat werk aan de voorbereiding, maar de oven
doet het grootste deel.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1}{kalkoen van 5,5--7 kg, ontdooid}\index[register]{Kalkoen}
  \ing{ca.\ 680 g}{dikgesneden spek (ca.\ 22 plakken), voor de vlecht en poten}\index[register]{Spek}
  \ing{ca.\ 115 g}{dikgesneden spek (ca.\ 4 plakken), grof gehakt, voor de boter}
  \ing{115 g}{ongezouten roomboter}
  \ing{1}{ui, gepeld en in vieren}
  \ing{1 el}{gedroogde tijm}
  \ing{1 tl}{gedroogde salie}
  \ing{ca.\ 475 ml}{water, voor de braadslede}
  \ingb{keukentouw}
  \ingb{bakpapier}
\end{ingredients}

\blockrule{Bereiding}
\tip{Een vleesthermometer is hier echt handig. Reken op ca.\ 74\,°C kerntemperatuur op drie plekken.}
\begin{steps}
  \item Maak de spekboter: doe de boter, het grof gehakte spek, tijm en salie
        in een keukenmachine en pulseer tot een gladde massa. Kan tot 3 dagen
        van tevoren; laat op kamertemperatuur komen voor gebruik.
  \item Haal de nek en het slachtafval uit de holte van de kalkoen (bewaar voor
        bouillon of jus). Leg de kalkoen met de borst naar boven op een braadrek
        in een braadslede. Wrijf de binnen- en buitenkant royaal in met de
        spekboter. Stop de ui in de holte en bind eventueel de poten samen met
        keukentouw.
  \item Wikkel elke poot in 2 plakken spek en elke vleugel in 1 plak spek. Stop
        de uiteinden onder het lijf of in de vouw tussen poot en borst.
  \item Vlecht de spekdeken: leg 8 plakken spek strak naast elkaar op een vel
        bakpapier. Weef er van onder naar boven nog 8 plakken doorheen, telkens
        om en om, en druk ze zo dicht mogelijk aan.
  \item Til de spekvlecht op het bakpapier op en draai hem ondersteboven op de
        borst van de kalkoen. Trek het papier voorzichtig los en druk de vlecht
        rond de borst en in de vouw naar de poten. Knip overhangend spek weg met
        een schaar als het te lang is.
  \item Laat de kalkoen 2 uur op kamertemperatuur komen voordat hij de oven in
        gaat. Je kunt hem ook een dag van tevoren met spek omwikkelen en
        afgedekt in de koelkast bewaren.
  \item Verwarm de oven voor op 220\,°C, met een rek in de onderste derde.
        Giet ca.\ 475 ml water in de braadslede onder de kalkoen; die stoom
        voorkomt dat het spek te snel krimpt of eraf glijdt. Rooster 30 minuten
        om het spek te bruinen en krokant te maken.
  \item Verlaag de oven naar 165\,°C. Rooster nog 2 tot 2,5 uur, tot de
        kerntemperatuur 74\,°C is. Meet op drie plekken: de borst, de
        buitenkant van de dij, en tussen dij en borst.
  \item Laat de kalkoen minstens 30 minuten rusten voor het aansnijden. Verwijder
        eerst de poten en snijd daarna de borst. Snijd door het spek heen om de
        vlecht heel te houden. Serveer met de spekkruimels die eraf vallen.
\end{steps}
