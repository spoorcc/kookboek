\begin{recipe}{Riso freddo}
\kicker{Voorgerecht · Italiaans}
\index[register]{Riso freddo}
\index[register]{Rijst}
\index[register]{Artisjok}
\index[register]{Olijven}

\meta{30 minuten \dvd Italiaans}

\lettrine{D}{it} recept leerden we kennen op een camping in Italië, waar de ober het
aankondigde als \textit{risoooo freeeedo}. Het is een koude rijstsalade die je goed
een dag van tevoren kunt maken.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{300 g}{langkorrelige rijst}\index[register]{Rijst}
  \ing{1 pot}{artisjokken op zuur, uitgelekt}\index[register]{Artisjok}
  \ing{1 pot}{gegrilde paprika, uitgelekt}\index[register]{Paprika}
  \ing{150 g}{zachte worst (b.v.\ cocktailworstjes of mortadella), in stukjes}
  \ing{1}{paprika, in kleine blokjes}
  \ing{1}{wortel, in kleine blokjes}
  \ing{150 g}{witte kaas (b.v.\ feta of emmental), in blokjes}
  \ing{1 handvol}{zwarte olijven}\index[register]{Olijven}
  \ing{2 el}{kappertjes}
  \ingb{olijfolie}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Maak de salade een paar uur of de dag van tevoren. De smaken trekken dan goed in.}
\begin{steps}
  \item Kook de rijst volgens de verpakking. Giet af, spoel kort met koud water en
        laat volledig afkoelen.
  \item Snijd de paprika en wortel in kleine blokjes. Snijd de gegrilde paprika en
        artisjokken in stukken. Snijd de worst en kaas in blokjes.
  \item Meng alle ingrediënten door de afgekoelde rijst. Besprenkel met olijfolie en
        breng op smaak met peper en zout.
  \item Zet minstens 30 minuten in de koelkast voor het serveren.
\end{steps}
\end{recipe}
