\begin{recipe}{Pasta met prei, hamblokjes en zwarte olijven}
\kicker[Hoofdgerecht,Vlees,Italiaans]{Hoofdgerecht · vlees · Italiaans}
\index[register]{Pasta met prei, hamblokjes en zwarte olijven}
\index[register]{Prei}
\index[register]{Ham}
\index[register]{Olijven}

\meta{25 minuten}

\lettrine{E}{en} romige pastasaus van prei en hamblokjes, met de zilte bite
van zwarte olijven erdoorheen. Snel klaar en met ingrediënten die vaak toch
al in huis zijn.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{3}{preien, in ringen}\index[register]{Prei}
  \ing{250 g}{hamblokjes}\index[register]{Ham}
  \ing{235 g}{zwarte olijven zonder pit, in schijfjes}\index[register]{Olijven}
  \ing{350 g}{pasta}
  \ing{200 ml}{kookroom}
  \ing{1}{knoflookteentje}\index[register]{Knoflook}
  \ingb{olijfolie, om te bakken}
  \ingb{klontje boter, om de prei te smoren}
  \ingb{versgemalen peper en zout}
  \ingb{Parmezaanse kaas, geraspt}\index[register]{Parmezaanse kaas}
\end{ingredients}

\blockrule{Bereiding}
\tip{Was de preiringen goed voor je ze bakt, er zit vaak zand tussen de
laagjes.}
\begin{steps}
  \item Kook de pasta beetgaar volgens de aanwijzingen op de verpakking. Bewaar
        een kop kookvocht voor je afgiet.
  \item Snijd ondertussen de prei in ringen en de olijven in schijfjes.
  \item Bak de hamblokjes bruin in een scheutje olijfolie in een hapjespan. Haal
        ze uit de pan en zet apart.
  \item Smelt een klontje boter in dezelfde pan op laag vuur en laat de prei
        er zo'n 10 minuten zacht in smelten, met het deksel op de pan. Pers de
        knoflook er de laatste minuut bij. Gesmolten prei wordt zoet en romig,
        en dat is de saus; roerbakken op hoog vuur geeft dat niet.
  \item Doe de hamblokjes terug in de pan. Voeg de kookroom en de olijven toe
        en breng op smaak met peper en zout.
  \item Giet de pasta af en schep hem met een scheut kookvocht door de saus in
        de pan.
  \item Rasp er wat Parmezaanse kaas overheen en serveer direct.
\end{steps}
\end{recipe}
