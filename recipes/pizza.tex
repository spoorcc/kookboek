\begin{recipe}{Pizza uit de reguliere oven}
\kicker[Hoofdgerecht,Vlees,Italiaans]{Hoofdgerecht · vlees · Italiaans}
\index[register]{Pizza}
\index[register]{Mozzarella}
\index[register]{Prosciutto}
\index[register]{Bloem}

\meta{Avond tevoren + 1{,}5 uur \dvd 2 pizza's \dvd Oven hoogste stand}

\lettrine{P}{izzadeeg} dat een nacht in de koelkast heeft gestaan smaakt heel anders
dan haastwerk. De meeste tijd in dit recept is wachten; het actieve werk valt wel
mee.

\blockrule{Ingrediënten}

\noindent\textit{Deeg}
\begin{ingredients}
  \ing{500 g}{bloem (tipo '00')}
  \ing{350 g}{koud water (70\%)}
  \ing{2 g}{instant gist}
  \ing{3 g}{zout}
  \ingb{kneepje honing of 12\,g kristalsuiker}
  \ingb{beetje olijfolie}
  \ingb{semolina, voor het uitrekken}
\end{ingredients}

\medskip
\noindent\textit{Saus}
\begin{ingredients}
  \ing{1 blik (400\,g)}{gepelde tomaten (San Marzano)}
  \ing{1}{knoflookteentje}
  \ingb{extra vergine olijfolie}
  \ingb{snuf zout}
\end{ingredients}

\medskip
\noindent\textit{Topping (voor beide pizza's samen)}
\begin{ingredients}
  \ing{80 g}{prosciutto}
  \ing{200 g}{mozzarella}
  \ingb{Parmigiano Reggiano}
  \ingb{basilicum}
\end{ingredients}

\blockrule{Bereiding}
\tip{Gebruik geen deegroller. Rek het deeg met je handen op en druk de randjes
niet in. Zo blijft de korst luchtig.}

\medskip
\noindent\textit{Avond tevoren}
\begin{steps}
  \item Meng alle deegingrediënten behalve de olijfolie. Kneed minstens
        10 minuten. Dit is het belangrijkste deel van het recept: zo ontwikkel
        je het gluten dat straks voor een luchtige korst zorgt. Dit deeg bevat
        bewust wat minder zout dan een traditioneel pizzadeeg: dat is milder,
        ook voor kinderen, en geeft een iets vlakkere korst. Proef het gerust
        en verhoog het zout naar smaak.
  \item Smeer de deegbal aan het einde licht in met olijfolie. Dek af met een
        natte theedoek en zet in de koelkast.
\end{steps}

\medskip
\noindent\textit{Dag van het bakken, 2 uur van tevoren}
\begin{steps}
  \item Haal het deeg uit de koelkast en laat het op kamertemperatuur komen.
  \item Verdeel in twee gelijke helften. Rek elk stuk uit en vouw de uiteinden
        meerdere keren naar binnen. Bol de stukken op tot strakke ballen.
  \item Leg de ballen onder een omgekeerde schaal en laat 2 uur rijzen op
        kamertemperatuur.
\end{steps}

\medskip
\noindent\textit{1 uur van tevoren}
\begin{steps}
  \item Verwarm de oven voor op de hoogste stand, met pizzastand als die
        beschikbaar is.
  \item Blend de tomaten met een scheutje olijfolie, het knoflookteentje en
        een snuf zout tot saus. Giet door een fijn vergiet en laat het overtollige
        vocht ca. 10 minuten uitlekken, zodat de bodem straks niet nat wordt.
  \item Scheur de mozzarella in kleine stukjes en laat uitdruipen in een vergiet,
        ook dat houdt de bodem droog.
  \item Rasp de parmigiano alvast.
\end{steps}

\medskip
\noindent\textit{Bakken (herhaal voor de tweede pizza)}
\begin{steps}
  \item Strooi wat semolina over de deegbal en wentel hem er licht doorheen.
        Strooi ook wat semolina op bakpapier, leg de bal erop en rek het deeg
        rustig met je handen op tot pizzavorm. De randjes niet indrukken.
  \item Smeer een dunne laag tomatensaus over het deeg. Je moet de kleur van
        het deeg er nog doorheen zien.
  \item \textbf{Eerste bak:} schuif de pizza in de oven tot de korst de eerste
        bruine of zwarte plekjes krijgt en de saus bijna droog is,
        ongeveer 6 minuten afhankelijk van je oven.
  \item Haal de pizza uit de oven. Verdeel de mozzarella erover, voeg
        blaadjes basilicum toe en rasp parmigiano erbovenop.
  \item \textbf{Tweede bak:} terug in de oven tot de mozzarella net gesmolten
        is, maar nog niet bruin. Haal de pizza dan uit de oven en leg de
        prosciutto er meteen, rauw, op de hete pizza: meegebakken prosciutto
        wordt zout en taai, terwijl hij op de hitte gelegd zacht en glanzend
        blijft. Zo doen pizzeria's het met crudo.
\end{steps}
\end{recipe}
