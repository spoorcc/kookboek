\begin{recipe}{Roerbak met kip en cashewnoten}
\kicker[Hoofdgerecht,Vlees,Aziatisch,Doordeweeks]{Hoofdgerecht · vlees · Aziatisch · doordeweeks}
\index[register]{Roerbak met kip en cashewnoten}
\index[register]{Kip}
\index[register]{Cashewnoten}
\index[register]{Eiernoodles}

\meta{20 minuten \dvd Eén pan}

\lettrine{E}{en} simpele roerbak voor doordeweeks: eiernoodles met kip en
groente uit de zak, aangemaakt met kant-en-klare woksaus. Supermarkten
hebben woksaus in allerlei smaken, dus neem gerust de variant die je het
lekkerst vindt of die je toevallig in huis hebt. Een flesje is meestal
ongeveer 200 ml. Geroosterde cashewnoten en een paar plakjes komkommer
erbij maken het gerecht af. Heb je liever rijst dan noodles, of andersom,
vervang ze dan gewoon: de rest van het recept blijft hetzelfde.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1}{kipfilet (ca. 200 g), in reepjes}\index[register]{Kip}
  \ing{150 g}{ongezouten cashewnoten}\index[register]{Cashewnoten}
  \ing{300 g}{eiernoodles}\index[register]{Eiernoodles}
  \ing{400 g}{roerbakgroente}
  \ing{1 flesje}{woksaus, ongeveer 200 ml}
  \ing{1}{komkommer, in plakjes}\index[register]{Komkommer}
\end{ingredients}

\blockrule{Bereiding}
\tip{Rooster de cashewnoten zonder olie of boter, op laag vuur en blijf roeren.
Zo kleuren ze gelijkmatig en verbranden ze niet.}
\begin{steps}
  \item Kook de eiernoodles volgens de aanwijzingen op de verpakking en laat uitlekken.
  \item Rooster de cashewnoten op laag vuur in een kleine koekenpan tot ze goudbruin
        zijn. Zet apart.
  \item Verhit een wokpan op hoog vuur en bak de kip in twee porties gaar en bruin: een
        volle wok zakt in temperatuur en dan stooft de kip in plaats van dat
        hij bakt. Zet elke portie na het bakken apart.
  \item Doe de kip terug in de wok, voeg de roerbakgroente toe en bak 2 tot 3
        minuten mee tot de groente net beetgaar is.
  \item Schep de eiernoodles en de woksaus erdoor en meng goed.
  \item Serveer met de geroosterde cashewnoten en een paar plakjes komkommer erbij.
\end{steps}
\end{recipe}
