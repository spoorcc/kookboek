\begin{recipe}{Semifreddo met aardbeien}
\kicker{Dessert · bevroren}
\index[register]{Semifreddo met aardbeien}
\index[register]{Aardbeien}
\index[register]{Slagroom}

\heroplaceholder{handgetekende illustratie van een plak semifreddo met aardbeien}
\meta{30 min + 6--8 uur invriezen \dvd Voor 12 personen \dvd Italiaans}

\lettrine{S}{emifreddo} is een Italiaans bevroren dessert dat zachter is dan gewoon
ijs. Dit recept met aardbeiencoulis en wafelbodem is al jaren een vaste gast op het
kerstmenu.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{500 g}{aardbeien + extra voor de garnering}\index[register]{Aardbeien}
  \ing{80 g}{poedersuiker (+ eventueel extra naar smaak)}
  \ing{1 el}{citroensap}
  \ing{250 g}{galette wafels}
  \ing{50 g}{zachte roomboter}
  \ing{300 ml}{slagroom}\index[register]{Slagroom}
  \ing{125 ml}{yoghurt}
  \ingb{muntblaadjes, voor garnering}
  \ingb{poedersuiker, voor garnering}
\end{ingredients}

\blockrule{Bereiding}
\tip{Maak de semifreddo een dag van tevoren. Na een nacht in de vriezer heb je de
dag zelf niets meer te doen.}
\ingredientsketch{aardbeien}
\begin{steps}
  \item Was de aardbeien en verwijder de kroontjes. Zet de helft apart voor later.
        Snijd de andere helft in stukken en doe ze in een pan met de poedersuiker en
        het citroensap. Breng aan de kook, prak fijn en laat inkoken tot een vloeibare
        coulis. Doe in een kom en zet 20 minuten in de vriezer.
  \item Smelt de boter. Verkruimel de wafels fijn, het snelst in een keukenmachine.
        Meng met de boter. Bekleed een spring- of cakevorm met vershoudfolie en druk
        het kruimelmengsel aan op de bodem.
  \item Klop de slagroom stijf. Spatel de yoghurt erdoor. Haal de coulis uit de
        vriezer (moet volledig afgekoeld zijn) en spatel erdoor. Proef en voeg
        eventueel een beetje extra poedersuiker toe.
  \item Snijd de achtergehouden aardbeien doormidden en leg ze op de koekbodem.
        Giet het roommengsel erover. Zet 6--8 uur (of langer) in de vriezer.
  \item Haal de semifreddo 15 minuten voor het serveren uit de vriezer. Haal uit de
        vorm en verwijder het vershoudfolie. Garneer met aardbeien, muntblaadjes en
        een beetje poedersuiker. Snijd met een warm mes in plakken.
\end{steps}
\end{recipe}
