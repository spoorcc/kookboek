\begin{recipe}{Pasta alla Norma}
\kicker[Hoofdgerecht,Vegetarisch,Siciliaans]{Hoofdgerecht · vegetarisch · Siciliaans}
\index[register]{Pasta alla Norma}
\index[register]{Aubergine}
\index[register]{Ricotta}
\index[register]{Pasta}

\meta{40 minuten}

\lettrine{P}{asta} alla Norma komt uit Sicilië en draait om gebakken aubergine,
tomatensaus en ricotta. Neem de tijd voor de aubergine: die mag rustig goudbruin
worden.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{350 g}{pasta (penne of rigatoni)}\index[register]{Pasta}
  \ing{1 grote}{aubergine}\index[register]{Aubergine}
  \ing{1}{knoflookteentje, fijngehakt}
  \ing{150 g}{verse ricotta}\index[register]{Ricotta}
  \ing{1 blik (400\,g)}{tomatenpulp (liefst met basilicumsmaak)}
  \ingb{olijfolie}
  \ingb{grof zout}
  \ingb{handvol verse basilicum, grof gesnipperd}
  \ingb{versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Schil de aubergine als kinderen bezwaar hebben tegen de schil. Zelf
ricotta maken kan ook, zie het basisrecept op p.~\pageref{recipe:ricotta}.}
\begin{steps}
  \item Was de aubergine en snijd hem ongeschild in blokjes van ± 1 cm. Leg ze op een
        snijplank, strooi er grof zout over en laat 10 minuten staan: dat trekt het
        vocht en de bitterheid eruit. Dep ze dan droog met keukenpapier.
  \item Bak de aubergineblokjes in een koekenpan in olijfolie goudbruin. Doe niet meer blokjes in de pan
        dan er naast elkaar passen en werk in batches, anders stomen ze in plaats van
        dat ze bruinen. Laat uitlekken op keukenpapier.
  \item Fruit de knoflook op laag vuur in een hapjespan met een scheut olijfolie. Zorg dat hij niet bruin
        wordt. Voeg de tomatenpulp toe, breng op smaak met zout en peper en laat
        10 minuten rustig pruttelen.
  \item Kook de pasta al dente. Op een Nederlandse verpakking: kook 1 minuut korter
        dan aangegeven. Bewaar een kop kookvocht voor je afgiet.
  \item Giet de pasta af en schep hem met een scheut kookvocht door de tomatensaus
        in de pan: het zetmeel bindt de saus om de pasta heen. Roer de helft van de
        ricotta en de basilicum erdoor, en dan ook de aubergineblokjes.
  \item Schep op en garneer met een lepel ricotta, wat losse basilicumblaadjes en
        versgemalen peper.
\end{steps}
\end{recipe}
