\begin{recipe}{Kofta}
\kicker[Hoofdgerecht,Vlees,Grieks]{Hoofdgerecht · vlees · Grieks}
\index[register]{Kofta}
\index[register]{Gehakt}
\index[register]{Lamsgehakt}
\index[register]{Courgette}

\meta{40 minuten}

\lettrine{K}{ofta} zijn gekruide gehaktworstjes die je na het bruin bakken terug
in een tomatensaus met courgette laat sudderen. Lekker met rijst.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{lamsgehakt of half-om-half gehakt}\index[register]{Lamsgehakt}
  \ing{2 kleine}{uien, gesnipperd}
  \ing{1}{courgette, in plakken of blokjes}\index[register]{Courgette}
  \ing{2}{knoflookteentjes, fijngehakt}
  \ing{500 g}{gepelde tomaten}
  \ing{2}{tomaten (naar keuze), in stukken}
  \ingb{olijfolie}
  \ingb{langkorrelige rijst, volgens verpakking}
  \ingb{handvol verse peterselie, fijngehakt}
  \ingb{komijn, kaneel, chilipoeder, zout en versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Wees niet bang voor de kaneel: een klein beetje geeft de saus een warme, iets
diepe smaak die kenmerkend is voor Grieks stoofvlees.}
\begin{steps}
  \item Meng het gehakt met de helft van de ui, de knoflook, de helft van de peterselie,
        zout en peper. Kneed goed door en vorm tot kleine worstjes.
  \item Bak de worstjes rondom bruin in olijfolie. Haal ze uit de pan en zet apart.
  \item Bak de courgette en de rest van de ui in dezelfde pan.
  \item Voeg de gepelde tomaten toe en breng op smaak met komijn, kaneel, chilipoeder,
        zout en peper. Voeg eventueel de verse tomatenstukken toe.
  \item Leg de worstjes terug in de pan en laat minstens 10 minuten zachtjes sudderen.
  \item Kook de rijst volgens de verpakking. Schep op en bestrooi met de rest van de
        peterselie.
\end{steps}
\end{recipe}
