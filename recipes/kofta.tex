\begin{recipe}{Kofta}
\kicker[Hoofdgerecht,Vlees,Midden-Oosters]{Hoofdgerecht · vlees · Midden-Oosters}
\index[register]{Kofta}
\index[register]{Gehakt}
\index[register]{Lamsgehakt}
\index[register]{Courgette}

\meta{40 minuten}

\lettrine{K}{ofta} zijn gekruide gehaktworstjes uit het Midden-Oosten, in Griekenland
beter bekend als keftedes, die je na het bruin bakken terug in een tomatensaus met
courgette laat sudderen. Lekker met rijst.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{500 g}{lamsgehakt of half-om-half gehakt}\index[register]{Lamsgehakt}
  \ing{170 g}{langkorrelige rijst}
  \ing{1}{courgette}\index[register]{Courgette}
  \ing{2 kleine}{uien, gesnipperd}
  \ing{2}{knoflookteentjes, fijngehakt}
  \ing{500 g}{gepelde tomaten}
  \ing{2}{tomaten (naar keuze), gewassen en in stukken}
  \ingb{olijfolie}
  \ingb{pot verse peterselie, fijngehakt}
  \ingb{komijn, kaneel, chilipoeder, zout en versgemalen peper}
  \ingb{yoghurt en verse munt, om te serveren}
\end{ingredients}

\blockrule{Bereiding}
\tip{Wees niet bang voor de kaneel: een klein beetje geeft de saus een warme, iets
diepe smaak die kenmerkend is voor Grieks stoofvlees.}
\begin{steps}
  \item Meng het gehakt met de helft van de ui, de knoflook, een paar fijngehakte
        takjes peterselie, een snuf komijn en kaneel, zout en peper. Kneed goed door
        en vorm tot kleine worstjes: kruid het gehakt zelf, niet alleen de saus, dat is
        wat kofta onderscheidt van gewone gehaktballetjes.
  \item Laat de gevormde worstjes 20 tot 30 minuten opstijven in de koelkast:
        dan vallen ze bij het bakken niet uit elkaar en houd je mooiere
        plakken saus.
  \item Gebruik deze tijd om de courgette te wassen en in kwartjes van ongeveer een halve
        centimeter dik te snijden, en de rest van de ui te snipperen.
  \item Bak de worstjes op middelhoog tot hoog vuur in een hapjespan rondom goed bruin
        in olijfolie: dat bruine korstje is waar de smaak vandaan komt. Ze hoeven nog niet
        helemaal gaar te zijn, want ze garen straks verder in de saus. Haal ze uit de pan
        en zet apart.
  \item Bak de courgette en de rest van de ui in dezelfde hapjespan: het vet
        en de aanbaksels van de worstjes blijven zo achter en geven meteen
        smaak aan de saus.
  \item Voeg de gepelde tomaten toe en breng op smaak met de rest van de komijn en kaneel,
        het chilipoeder, het grootste deel van de fijngehakte peterselie, zout en peper.
        Voeg eventueel de verse tomatenstukken toe.
  \item Leg de worstjes terug in de pan en laat minstens 10 minuten zachtjes sudderen.
  \item Kook de rijst volgens de verpakking. Schep op, bestrooi met het laatste
        beetje peterselie en zet er een lepel yoghurt met wat verse munt bij: het
        frisse, koele tegenwicht dat dit soort stoofgerechten in het hele
        Midden-Oosten meekrijgt.
\end{steps}
\end{recipe}
