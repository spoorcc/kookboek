\begin{recipe}{Hutspot}
\kicker[Hoofdgerecht,Vlees,Nederlands,Doordeweeks]{Hoofdgerecht · vlees · Nederlands · doordeweeks}
\index[register]{Hutspot}
\index[register]{Wortel}
\index[register]{Rookworst}
\index[register]{Stamppot}

\meta{30 minuten}

\lettrine{H}{utspot} is een klassieke stamppot van aardappel, wortel en ui,
gaar gekookt in bouillon en dan gestampt met roomboter.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{500 g}{kruimige aardappelen, geschild en in stukken}\index[register]{Aardappelen}
  \ing{1 zak}{hutspot groente (wortel en ui gesneden, Boons)}\index[register]{Wortel}
  \ing{1}{bouillonblokje (rund, kip of groente)}
  \ing{500 ml}{water}
  \ing{25 g}{roomboter}
  \ing{1}{rookworst}\index[register]{Rookworst}
  \ingb{scheutje melk}
  \ingb{zout en versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Niet te lang stampen: een hutspot met nog wat grove stukken aardappel en wortel is lekkerder dan een gladde puree.}
\begin{steps}
  \item Doe de aardappelen, de hutspot groente en het bouillonblokje in een grote pan.
        Voeg het water toe en breng aan de kook.
  \item Zet het vuur lager en kook zachtjes ca.\ 20 minuten tot de groenten gaar
        zijn. Verwarm de rookworst ondertussen ca.\ 10 minuten in een aparte pan
        met wat water op laag vuur.
  \item Giet de groenten af.
  \item Stamp grof met de roomboter: er mogen nog wat grove stukken in zitten.
        Voeg een scheutje melk toe voor de gewenste dikte.
  \item Breng op smaak met zout en peper. Serveer met de rookworst erbij.
\end{steps}
\end{recipe}
