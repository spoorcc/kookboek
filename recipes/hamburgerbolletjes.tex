\begin{recipe}{Hamburgerbolletjes}
\label{recipe:hamburgerbolletjes}
\kicker[Brood,Basisrecept]{Brood \& basis · basisrecept}
\index[register]{Hamburgerbolletjes}
\index[register]{Bloem}
\index[register]{Sesamzaad}

\meta{Rijzen ca. 2,5–3 uur \dvd 5 stuks \dvd Oven 180\,°C}

\lettrine{L}{uxe}, zachte hamburgerbolletjes met een beetje zoetigheid van
brioche, zonder dat er iets ingewikkelds aan te pas komt. Het deeg mag zacht
en licht plakkerig blijven, dat hoort bij brioche. Zolang je het genoeg tijd
geeft om te rijzen gaat er weinig mis.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{300 g}{tarwebloem}\index[register]{Bloem}
  \ing{7 g}{droge gist (1 zakje)}
  \ing{15 g}{suiker}
  \ing{6 g}{zout}
  \ing{120 ml}{lauwwarme melk}
  \ing{1}{ei}
  \ing{40 g}{zachte boter (kamertemperatuur)}
  \ing{1}{ei, voor de afwerking}
  \ing{1 el}{melk, voor de afwerking}
  \ingb{sesamzaad}\index[register]{Sesamzaad}
\end{ingredients}

\blockrule{Bereiding}
\tip{Laat de bolletjes niet te donker worden: door de suiker en het ei
kleurt brioche sneller dan gewoon brood.}

\medskip
\noindent\textit{Deeg maken}
\begin{steps}
  \item Meng de melk, gist en suiker.
  \item Voeg de bloem en het ei toe en meng tot een plakkerig deeg.
  \item Voeg het zout toe en kneed 5–7 minuten, met de hand of met een
        machine.
  \item Voeg de boter toe en kneed nog 5–8 minuten door tot het deeg soepel
        en elastisch is.
\end{steps}

\medskip
\noindent\textit{Eerste rijs}
\begin{steps}
  \item Dek het deeg af en laat 1–1,5 uur rijzen, tot het verdubbeld is.
\end{steps}

\medskip
\noindent\textit{Bollen}
\begin{steps}
  \item Verdeel het deeg in 5 stukken van ongeveer 95–100 g.
  \item Bol elk stuk strak op, dat geeft mooie ronde bolletjes.
  \item Leg de bolletjes op een bakplaat met bakpapier.
\end{steps}

\medskip
\noindent\textit{Tweede rijs}
\begin{steps}
  \item Laat de bolletjes 45–60 minuten rijzen tot ze luchtig aanvoelen.
        Verwarm ondertussen de oven voor op 180\,°C (boven- en
        onderwarmte), zodat hij op temperatuur is als de bolletjes klaar
        zijn.
\end{steps}

\medskip
\noindent\textit{Afwerking en bakken}
\begin{steps}
  \item Klop het ei los met de melk en bestrijk de bolletjes hier
        voorzichtig mee, zonder ze in te drukken.
  \item Bestrooi met sesamzaad.
  \item Bak 15–18 minuten, tot de bolletjes goudbruin zijn.
\end{steps}

\end{recipe}
