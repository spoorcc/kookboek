\begin{recipe}{Bobotie}
\kicker[Hoofdgerecht,Vlees,Zuid-Afrikaans]{Hoofdgerecht · vlees · Zuid-Afrikaans}
\index[register]{Bobotie}
\index[register]{Gehakt}
\index[register]{Sperziebonen}
\index[register]{Abrikozenjam}

\meta{50 minuten \dvd Oven 180\,°C}

\lettrine{B}{obotie} is een Zuid-Afrikaans gerecht van gekruid gehakt met appel en
groenten, afgedekt met een laag ei en melk en dan de oven in. Het wordt geserveerd
met rijst gekookt met rozijnen.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{rundergehakt, of half-om-half met lam of varken (15--20\% vet)}\index[register]{Gehakt}
  \ing{170 g}{langkorrelige rijst}
  \ing{1}{ui, gesnipperd}
  \ing{1}{paprika, gewassen en gesnipperd}
  \ing{1}{knoflookteentje, geperst}
  \ing{1}{appel (b.v.\ Elstar), gewassen en in blokjes}
  \ing{300 g}{sperziebonen}\index[register]{Sperziebonen}
  \ing{1 flinke el}{abrikozenjam}\index[register]{Abrikozenjam}
  \ing{1 el}{citroensap}
  \ing{1}{ei}
  \ing{100 ml}{melk}
  \ing{1½ el}{milde kerrie}
  \ing{1 tl}{kurkuma}
  \ing{1 tl}{gemberpoeder}
  \ing{½ tl}{kaneelpoeder}
  \ingb{olijfolie}
  \ingb{2 laurierblaadjes}
  \ingb{handvol rozijnen}
  \ingb{eventueel ½ tl komijn}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Gebruik niet te mager gehakt. Te mager gehakt kan droog worden.}
\begin{steps}
  \item Verwarm de oven voor op 180\,°C (boven- en onderwarmte).
  \item Maak de sperziebonen schoon en kook ze 2 minuten voor: ze garen straks
        verder mee in de oven. Giet af.
  \item Snipper de ui en fruit hem op laag vuur 10--15 minuten in een scheutje
        olijfolie tot hij zacht is en licht karamelliseert: dat geeft een
        zoete, volle basis. Voeg de knoflook toe en bak een minuut mee.
  \item Zet het vuur hoger. Voeg de paprika en het gehakt toe en bak rul. Roer de kerrie, kurkuma,
        gember, kaneel, eventueel de komijn, abrikozenjam, citroensap, peper en
        zout erdoor en laat 1--2 minuten meebakken. Het zuur brengt de zoete
        smaken van appel en jam in balans.
  \item Voeg de appelblokjes en de sperziebonen toe en meng goed door.
  \item Schep het geheel in een ovenschaal. Klop het ei met de melk en giet dit
        over het mengsel. Steek de laurierblaadjes in de custard.
  \item Bak 30--40 minuten in de oven tot de custard net gestold is.
  \item Kook de laatste 10 minuten de rijst samen met de rozijnen gaar. Laat
        de bobotie 10 minuten rusten voor je hem aansnijdt, dan blijft hij
        beter in vorm. Serveer naast de rijst.
\end{steps}
