\begin{recipe}{Bobotie}
\kicker[Hoofdgerecht,Vlees,Zuid-Afrikaans]{Hoofdgerecht · vlees · Zuid-Afrikaans}
\index[register]{Bobotie}
\index[register]{Gehakt}
\index[register]{Sperziebonen}
\index[register]{Abrikozenjam}

\meta{50 minuten}

\lettrine{B}{obotie} is een Zuid-Afrikaans gerecht van gekruid gehakt met appel en
groenten, afgedekt met een laag ei en melk en dan de oven in. Het wordt geserveerd
met rijst gekookt met rozijnen.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{gehakt}\index[register]{Gehakt}
  \ing{1}{ui, gesnipperd}
  \ing{1}{paprika, gesnipperd}
  \ing{1}{knoflookteentje, geperst}
  \ing{1}{appel (b.v.\ Elstar), in blokjes}
  \ing{300 g}{sperziebonen}\index[register]{Sperziebonen}
  \ing{1 flinke el}{abrikozenjam}\index[register]{Abrikozenjam}
  \ing{1}{ei}
  \ing{100 ml}{melk}
  \ing{150 g}{langkorrelige rijst}
  \ingb{handvol rozijnen}
  \ing{1½ el}{milde kerrie}
  \ing{1 tl}{kurkuma}
  \ing{1 tl}{gemberpoeder}
  \ing{½ tl}{kaneelpoeder}
  \ingb{eventueel ½ tl komijn}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\begin{steps}
  \item Verwarm de oven voor op 180\,°C (boven- en onderwarmte).
  \item Maak de sperziebonen schoon en kook ze 2 minuten voor: ze garen straks
        verder mee in de oven. Giet af.
  \item Snipper de ui en bak aan in een scheutje olijfolie. Voeg de knoflook toe en bak
        een minuut mee.
  \item Voeg de paprika en het gehakt toe en bak rul. Roer de kerrie, kurkuma,
        gember, kaneel, eventueel de komijn, abrikozenjam, peper en zout erdoor.
  \item Voeg de appelblokjes en de sperziebonen toe en meng goed door.
  \item Schep het geheel in een ovenschaal. Klop het ei met de melk en giet dit
        over het mengsel.
  \item Bak 30--40 minuten in de oven.
  \item Kook de laatste 10 minuten de rijst samen met de rozijnen gaar.
        Serveer naast de bobotie.
\end{steps}
\end{recipe}
