\begin{recipe}{Mac 'n cheese}
\kicker[Hoofdgerecht,Vlees,Amerikaans]{Hoofdgerecht · vlees · Amerikaans}
\index[register]{Mac 'n cheese}
\index[register]{Macaroni}
\index[register]{Bechamelsaus}
\index[register]{Varkenslapjes}

\meta{45 minuten \dvd Oven 200\,°C}

\lettrine{A}{merikaans} comfort food dat makkelijk op te schalen is als je een
grotere groep hebt. Er gaat wat extra groente in om het wat minder zwaar te
houden. De zelfgemaakte bechamel geeft meer diepte dan kant-en-klare kaassaus.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{400 g}{varkenslapjes, in reepjes}\index[register]{Varkenslapjes}
  \ing{350 g}{macaroni}\index[register]{Macaroni}
  \ing{1}{paprika}\index[register]{Paprika}
  \ing{1}{courgette}\index[register]{Courgette}
  \ing{1}{ui}
  \ing{40 g}{ongezouten roomboter}
  \ing{40 g}{patentbloem}\index[register]{Bloem}
  \ing{500 ml}{melk}
  \ing{200 g}{geraspte kaas (minstens de helft oude of belegen)}\index[register]{Kaas}
  \ing{1 tl}{mosterd(poeder)}
  \ingb{mespunt cayennepeper}
  \ingb{paneermeel}
  \ingb{nootmuskaat}
  \ingb{rozemarijn}
  \ingb{tijm}
  \ingb{komijn}
  \ingb{zout en versgemalen peper}
  \ingb{extra boter, om te bakken}
\end{ingredients}

\blockrule{Bereiding}
\tip{Kook de macaroni echt kort: hij gaart nog na in de oven, dus liever te al dente dan te gaar.}
\begin{steps}
  \item Verwarm de oven voor op 200\,°C (boven- en onderwarmte).
  \item Snipper de ui. Was de paprika en courgette en snijd ze in blokjes.
  \item Kook de macaroni 2 minuten korter dan de verpakking aangeeft, dus
        echt onder al dente: hij gaart in de oven door en zuigt intussen saus
        op. Giet af.
  \item Maak de bechamelsaus: smelt een klont boter in een steelpan, roer er
        bloem door tot een roux en schenk dan geleidelijk melk erbij. Blijf
        roeren tot een gladde, gebonden saus. Met alleen milde kaas wordt de
        saus vlak, gebruik daarom minstens de helft oude of belegen kaas.
        Breng op smaak met zout, peper, nootmuskaat, de mosterd en een
        mespunt cayennepeper: die twee proef je niet apart terug, maar ze
        maken de kaassmaak dieper.
  \item Bak de varkenslapjes in reepjes op middelhoog tot hoog vuur in wat boter bruin
        in een grote hapjespan. Haal uit de pan en zet apart.
  \item Zet het vuur lager en fruit de ui in dezelfde pan. Voeg de paprika toe, dan de courgette.
        Breng op smaak met rozemarijn, tijm, komijn, zout en peper.
  \item Voeg de varkenslapjes terug. Roer de macaroni, bechamelsaus en een
        flinke hand geraspte kaas erdoor.
  \item Schep alles in een ovenschaal. Verdeel de rest van de geraspte kaas
        erover. Meng het paneermeel met een klontje gesmolten boter en
        bestrooi daarmee de bovenkant: zo wordt de korst goudbruin in plaats
        van stoffig.
  \item Zet 20 minuten in de oven tot de bovenkant goudbruin is.
\end{steps}
