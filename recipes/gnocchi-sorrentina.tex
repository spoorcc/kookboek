\begin{recipe}{Gnocchi alla sorrentina}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans]{Hoofdgerecht · vegetarisch · Italiaans}
\index[register]{Gnocchi alla sorrentina}
\index[register]{Gnocchi}
\index[register]{Mozzarella}
\index[register]{Tomaat}
\index[register]{Basilicum}

\meta{45 minuten \dvd Oven 200\,°C}

\lettrine{E}{en} klassieker uit Zuid-Italië, uit de omgeving van Sorrento.
Gnocchi met een eenvoudige tomatensaus, verse mozzarella en basilicum, kort
gegratineerd in de oven tot de kaas smelt. Uit de oven is de mozzarella nog
rekbaar en de bovenkant net krokant.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{2 pakken (à 400\,g)}{verse gnocchi}\index[register]{Gnocchi}
  \ing{250 g}{mozzarella, in stukjes gescheurd}\index[register]{Mozzarella}
  \ing{600 g}{passata}\index[register]{Tomaat}
  \ing{4 el}{extra vergine olijfolie}
  \ing{2 tenen}{knoflook, gepeld en geplet}\index[register]{Knoflook}
  \ing{8 blaadjes}{basilicum, grof gescheurd (+ extra voor de afwerking)}\index[register]{Basilicum}
  \ingb{pecorino, geraspt}\index[register]{Pecorino}
  \ingb{zeezout en versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Heb je wat meer tijd? Maak de gnocchi
(p.~\pageref{recipe:gnocchi}) en de mozzarella
(p.~\pageref{recipe:mozzarella}) dan zelf.}
\begin{steps}
  \item Verwarm de oven voor op 200\,°C. Scheur de mozzarella in stukjes en
        laat op keukenpapier uitlekken. Dep hem droog voor hij de schaal in
        gaat: natte mozzarella maakt van de bovenkant soep in plaats van
        gratin.
  \item Verhit de olijfolie met de knoflook in een middelgrote pan op middelhoog
        vuur. Voeg de passata en een scheutje water (ca. 60 ml) toe, doe het
        deksel erop en laat de saus 30 minuten zachtjes koken. Roer regelmatig.
        Breng op smaak met zout en peper en houd warm.
  \item Breng ondertussen een grote pan ruim gezouten water aan de kook. Kook de
        gnocchi tot ze bovendrijven, ongeveer 2--3 minuten. Schep ze met een
        schuimspaan uit het water en direct in de tomatensaus.
  \item Roer voorzichtig om en schep alles in een ovenschaal. Verdeel de
        uitgelekte mozzarella en de helft van de basilicum erover en bestrooi
        met geraspte pecorino. Houd de rest van de basilicum achter voor
        erna: basilicum die mee de oven in gaat wordt zwart. Dek af met
        aluminiumfolie.
  \item Bak 10 minuten in de oven tot de mozzarella gesmolten is. Haal de
        aluminiumfolie eraf en zet nog 2--3 minuten onder de grill tot de kaas goudbruin is.
        Garneer met de rest van de basilicumblaadjes en serveer direct.
\end{steps}
\end{recipe}
