\begin{recipe}{Barbecue}
\kicker[Hoofdgerecht,Vlees,Zomer]{Hoofdgerecht · vlees · zomer}
\index[register]{Barbecue}
\index[register]{Huzarensalade}
\index[register]{Stokbrood}

\meta{45 minuten}

\lettrine{B}{ij} een barbecue houden we het simpel: een paar soorten vlees,
afbakstokbrood, huzarensalade en roomkaas, en genoeg sauzen om uit te
kiezen. De bbq doet het meeste werk.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{3–4 soorten}{vlees naar keuze: bijv. spareribs, kipspiesjes,
      hamburgers, worstjes, \dots}
  \ing{2}{afbakstokbroden}\index[register]{Stokbrood}
  \ing{600 g}{huzarensalade}\index[register]{Huzarensalade}
  \ing{150 g}{roomkaas}
  \ingb{snackgroente, naar keuze}
  \ingb{bbq-saus}
  \ingb{whiskeycocktailsaus}
  \ingb{ketchup}
\end{ingredients}

\blockrule{Bereiding}
\tip{Begin met het vlees dat het langst moet garen, zoals spareribs, en bak
snelle stukken als worstjes of kipspiesjes pas op het laatst.}
\begin{steps}
  \item Steek de bbq op tijd aan, zodat hij op temperatuur is tegen de tijd
        dat je gaat bakken.
  \item Bak de stokbroden af volgens de verpakking en snijd ze in plakjes.
  \item Zet de roomkaas, huzarensalade, snackgroente en de sauzen vast op
        tafel.
  \item Bak het vlees op de bbq, in etappes als niet alles tegelijk past.
  \item Serveer alles samen aan tafel.
\end{steps}
\end{recipe}
