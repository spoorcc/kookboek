\begin{recipe}{Pannenkoeken}
\kicker{Hoofdgerecht · vegetarisch}
\index[register]{Pannenkoeken}
\index[register]{Bloem}
\index[register]{Eieren}
\index[register]{Melk}

\meta{45 minuten \dvd Voor 5 personen \dvd Nederlands}

\lettrine{P}{annenkoeken} zijn een vaste avondmaaltijd in de week, snel gemaakt en
voor iedereen leuk. Het beslag heeft even rust nodig, dus begin iets op tijd.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{bloem}\index[register]{Bloem}
  \ing{500 ml}{melk}
  \ing{2}{eieren}\index[register]{Eieren}
  \ingb{snufje zout}
  \ingb{vloeibare boter, voor de pan}
\end{ingredients}

\blockrule{Bereiding}
\tip{Laat het beslag 20 minuten rusten. Dat maakt de pannenkoeken soepeler en
minder snel scheuren bij het omdraaien.}
\begin{steps}
  \item Doe de bloem en het zout in een kom. Maak een kuiltje in het midden
        en breek de eieren erin.
  \item Roer vanuit het midden en voeg de melk beetje bij beetje toe tot een
        glad, dun beslag. Laat minstens 20 minuten rusten.
  \item Verhit een koekenpan op middelhoog vuur en smeer de bodem in met een
        beetje vloeibare boter.
  \item Schep een soeplepel beslag in de pan en kantel meteen zodat de bodem
        gelijkmatig bedekt is.
  \item Bak de pannenkoeek op middelhoog vuur tot de bovenkant droog ziet en
        de onderkant goudbruin is, draai dan om en bak nog een minuut.
  \item Herhaal voor de rest van het beslag.
\end{steps}
\end{recipe}
