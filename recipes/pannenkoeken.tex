\begin{recipe}{Pannenkoeken}
\kicker[Hoofdgerecht,Vegetarisch,Nederlands]{Hoofdgerecht · vegetarisch · Nederlands}
\index[register]{Pannenkoeken}
\index[register]{Bloem}
\index[register]{Eieren}
\index[register]{Melk}

\meta{45 minuten}

\lettrine{P}{annenkoeken} zijn een vaste avondmaaltijd in de week, snel gemaakt en
een favoriet bij de kinderen. Het beslag heeft even rust nodig, dus begin iets
op tijd.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{bloem}\index[register]{Bloem}
  \ing{500 ml}{melk}
  \ing{2}{eieren}\index[register]{Eieren}
  \ingb{vloeibare boter, voor de pan}
  \ingb{snufje zout}
  \ingb{plakken jong belegen kaas, voor de kaaspannenkoeken (optioneel)}
\end{ingredients}

\blockrule{Bereiding}
\tip{Laat het beslag 20 minuten rusten, dan worden de pannenkoeken soepeler en
scheuren ze minder snel bij het omdraaien.}
\begin{steps}
  \item Doe de bloem en het zout in een kom. Maak een kuiltje in het midden
        en breek de eieren erin.
  \item Roer vanuit het midden en voeg de melk beetje bij beetje toe tot een
        glad, dun beslag. Laat minstens 20 minuten rusten.
  \item Verhit een koekenpan op middelhoog vuur en smeer de bodem in met een
        beetje vloeibare boter.
  \item Schep een soeplepel beslag in de pan en kantel meteen zodat de bodem
        gelijkmatig bedekt is.
  \item Bak de pannenkoek op middelhoog vuur tot de bovenkant droog ziet en
        de onderkant goudbruin is, draai dan om en bak nog een minuut. De
        eerste pannenkoek is een proefpannenkoek: zo proef je meteen of het
        beslag op smaak is, en is de pan intussen goed heet en ingevet voor
        de rest.
  \item Herhaal voor de rest van het beslag. Van de laatste 5 pannenkoeken
        maken wij vaak kaaspannenkoeken: leg na het omdraaien een paar plakken
        jong belegen kaas op de helft van de pannenkoek en vouw hem dubbel
        zodra de kaas begint te smelten.
  \item Wij bakken meestal vooruit en houden de gebakken pannenkoeken warm
        onder een deksel tot iedereen aan tafel zit.
\end{steps}
\end{recipe}
