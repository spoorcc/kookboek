\begin{recipe}{Groene salade}
\label{recipe:groene-salade}
\kicker[Bijgerecht,Vegetarisch]{Bijgerecht · vegetarisch}
\index[register]{Groene salade}
\index[register]{IJsbergsla}
\index[register]{Balsamico}

\meta{5 minuten \dvd Voor 4 personen}

\lettrine{D}{e} kinderen ontdekten deze salade op vakantie in een Franse crêperie
en sindsdien is het een vast bijgerecht. IJsbergsla met balsamico, olijfolie en
een snufje knoflookpoeder, klaar voor je het weet.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1}{ijsbergsla}\index[register]{IJsbergsla}
  \ing{2 el}{balsamicoazijn}\index[register]{Balsamico}
  \ing{3 el}{olijfolie}
  \ingb{knoflookpoeder, naar smaak}
  \ingb{verse dille}
  \ingb{zout \& versgemalen peper, naar smaak}
\end{ingredients}

\blockrule{Bereiding}
\tip{IJsbergsla kan je vervangen door spitskool. Die heeft iets meer bite.}
\begin{steps}
  \item Snijd de ijsbergsla in stukken en doe in een ruime schaal.
  \item Schenk de balsamicoazijn en olijfolie erover en bestrooi met
        knoflookpoeder, zout en peper.
  \item Meng goed door. Voeg eventueel wat dille toe en serveer direct.
\end{steps}
\end{recipe}
