\begin{recipe}{Pretzels}
\kicker[Brood,Duits]{Brood \& basis · Duits}
\index[register]{Pretzels}
\index[register]{Brood}

\meta{2 uur \dvd 6--8 pretzels}

\lettrine{P}{retzels} maken vraagt meer tijd dan de meeste broden, maar het grootste
deel is rusttijd. Het bad in zuiveringszout en honing is wat de kenmerkende donkere
korst en smaak geeft.

\blockrule{Ingrediënten}

\noindent\textit{Deeg}
\begin{ingredients}
  \ing{500 g}{tarwebloem of broodmeel}
  \ing{5 g}{actieve droge gist}
  \ing{5 g}{kristalsuiker}
  \ing{35 g}{boter, gesmolten}
  \ing{10 g}{zout}
\end{ingredients}

\medskip
\noindent\textit{Bad}
\begin{ingredients}
  \ing{65 g}{zuiveringszout}
  \ing{45 g}{honing}
\end{ingredients}

\medskip
\noindent\textit{Afwerking}
\begin{ingredients}
  \ingb{grof zout of vlokzout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Leg de gevormde pretzels een nacht afgedekt in de koelkast. Ze zijn dan de
volgende dag direct klaar voor het bad en de oven, en de smaak en structuur zijn dan
beter.}
\begin{steps}
  \item Meng 275 ml lauwwarm water (± 46\,°C) met de gist en de suiker. Laat
        5 minuten staan tot het mengsel schuimig is.
  \item Meng de bloem en het zout in een grote kom. Voeg het gistmengsel toe en
        kneed 2--3 minuten. Voeg dan geleidelijk de boter toe en kneed tot een zacht,
        glad deeg, nog ongeveer 5 minuten.
  \item Vorm een bal, dek af met een vochtige doek en laat 20--25 minuten rusten.
  \item Verdeel het deeg in porties van ± 120 g. Rol elke portie losjes tot een
        korte streng. Dek af en laat nogmaals 20 minuten rusten.
  \item Rol elke streng langzaam uit tot ongeveer 65 cm. Vouw tot een klassieke
        pretzelvorm. Dek af met een vochtige doek en laat 45 minuten rusten.
        Verwarm de oven de laatste 15 minuten van de rusttijd voor op
        220\,°C (boven- en onderwarmte), zodat hij op temperatuur is
        wanneer de pretzels de oven in gaan.
  \item Breng 950 ml water met het zuiveringszout en de
        honing aan de kook. Dompel elke pretzel 20 seconden aan de eerste kant en
        15 seconden aan de tweede kant. Laat uitlekken op bakpapier.
  \item Bestrooi met zout en bak 15--20 minuten tot diep goudbruin.
  \item Laat afkoelen op een rooster.
\end{steps}
\end{recipe}
