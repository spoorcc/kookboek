\begin{recipe}{Salade caprese}
\kicker[Voorgerecht,Vegetarisch,Italiaans]{Voorgerecht · vegetarisch · Italiaans}
\index[register]{Salade caprese}
\index[register]{Tomaat}
\index[register]{Mozzarella}
\index[register]{Basilicum}

\meta{10 minuten}

\lettrine{H}{et} basisrecept voor een klassieke salade caprese bestaat uit maar een
paar ingrediënten, maar dan wel van goede kwaliteit: rijpe tomaten, romige
mozzarella en verse basilicum. Rood, wit en groen op een bord, de kleuren van
de Italiaanse vlag. Lekker met zelfgemaakte mozzarella of bij zelfgebakken
focaccia of ciabatta om de olijfolie en het tomatensap mee op te deppen.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{2 bollen}{mozzarella (bij voorkeur romige buffelmozzarella)}\index[register]{Mozzarella}
  \ing{5}{grote rijpe tomaten (bijvoorbeeld trostomaten of vleestomaten)}\index[register]{Tomaat}
  \ing{10}{verse blaadjes basilicum}\index[register]{Basilicum}
  \ing{2 el}{extra vergine olijfolie (van goede kwaliteit)}
  \ingb{grof zeezout en versgemalen zwarte peper, naar smaak}
  \ingb{een paar druppels goede balsamicoazijn, naar smaak}
\end{ingredients}

\blockrule{Bereiding}
\tip{Haal de mozzarella minimaal een half uur van tevoren uit de koelkast. Op
kamertemperatuur smaakt de salade voller dan koud uit de koelkast. Voeg het
zout, de peper en de olijfolie ook pas vlak voor het serveren toe: zout
onttrekt vocht aan de tomaten, en hoe langer dat kan intrekken, hoe waterig
de salade wordt.}
\begin{steps}
  \item Was de tomaten en snijd ze horizontaal in gelijkmatige, dunne plakken
        met een scherp kartelmes.
  \item Snijd de mozzarella in plakken van ongeveer dezelfde dikte als de
        tomaat.
  \item Schik de plakken tomaat en mozzarella om en om op een mooi plat bord.
  \item Verdeel de basilicumblaadjes tussen of over de plakken.
  \item Bestrooi vlak voor het serveren met het grove zeezout en de
        versgemalen zwarte peper.
  \item Besprenkel de salade met de olijfolie en voeg naar smaak een paar
        druppels balsamicoazijn toe.
\end{steps}
\end{recipe}
