\begin{recipe}{Aardappelen, groente en vlees}
\label{recipe:aardappel-groente-vlees}
\kicker[Hoofdgerecht,Vlees,Doordeweeks]{Hoofdgerecht · vlees naar keuze · doordeweeks}
\index[register]{Aardappelen, groente en vlees}
\index[register]{Krieltjes}
\index[register]{Paprika}
\index[register]{Komkommer}

\meta{30 minuten \dvd Eenvoudig}

\lettrine{H}{eel} basis, maar geliefd: gebakken aardappeltjes, rauwe groente en
vlees naar keuze. Geen vast recept, meer een combinatie die altijd werkt en
makkelijk is aan te passen aan wat er in huis is. In plaats van de rauwe
groente kan ook de groene salade (p.~\pageref{recipe:groene-salade}) erbij.

\blockrule{Ingrediënten}
\noindent\textit{Vlees}

{\small\setlength{\parindent}{0pt}
\ing{5 stuks}{vlees naar keuze: slavink, hamburger, chipolataworstjes, cordon bleu, \dots}\par}

\medskip
\noindent\textit{Gebakken aardappelen}
\begin{ingredients}
  \ing{450 g}{krieltjes, gehalveerd}\index[register]{Krieltjes}
  \ingb{boter}
  \ingb{knoflook(poeder)}
  \ingb{tijm}
  \ingb{rozemarijn}
  \ingb{zout}
  \ingb{ketchup}
  \ingb{yogonaise}
\end{ingredients}

\medskip
\noindent\textit{Groente}
\begin{ingredients}
  \ing{1}{komkommer, in plakjes}\index[register]{Komkommer}
  \ing{1}{paprika, in stukjes}\index[register]{Paprika}
  \ing{300 g}{snackwortels}
\end{ingredients}

\blockrule{Bereiding}
\tip{Kinderen vinden het soms leuk om de groente te dippen in goede olijfolie met versgemalen peper en zout.}
\begin{steps}
  \item Was de komkommer, paprika en snackwortels. Snijd de komkommer in
        plakjes en de paprika in stukjes; snackwortels kunnen heel blijven of
        in stukken.
  \item Bak het vlees volgens het recept op de verpakking.
  \item Was de krieltjes, halveer ze en kook ze 8 tot 10 minuten voor in gezouten
        water tot ze bijna gaar zijn. Giet af en laat goed uitdampen.
  \item Bak de krieltjes droog en met de snijkant naar beneden op middelhoog
        vuur flink aan in ruim boter, tot ze van binnen zacht zijn en van
        buiten een korst hebben.
  \item Voeg de tijm en rozemarijn de laatste paar minuten toe, zodat ze nog
        even kort kunnen meebakken in de boter, en breng pas vlak voor het
        serveren op smaak met zout en knoflook(poeder), zodat die niet
        verbranden.
  \item Serveer het vlees met de gebakken krieltjes en de rauwe groente
        erbij. De kinderen doen er graag ketchup bij, wij liever yogonaise.
\end{steps}
\end{recipe}
