\begin{recipe}{Pizzarolletjes}
\kicker[Bakken,Italiaans,Kinderen]{Bakken · Italiaans · kinderen}
\index[register]{Pizzarolletjes}
\index[register]{Mozzarella}
\index[register]{Croissantdeeg}

\meta{25 minuten \dvd 7 stuks \dvd Oven 180\,°C}

\lettrine{M}{et} kant-en-klaar croissantdeeg staan deze pizzarolletjes zo op
tafel: gewoon ontrollen, beleggen, oprollen en snijden. Fijn als snack of
erbij, en een uitgeprobeerd succesje bij de kinderen op het kerstdiner van
de basisschool, ook al zijn ze zelf niet bepaald kerstig.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 blikje}{croissantdeeg}\index[register]{Croissantdeeg}
  \ing{200 g}{geraspte mozzarella}\index[register]{Mozzarella}
  \ing{1 pot (400\,g)}{pastasaus met basilicum}
  \ingb{oregano, om te bestrooien}
\end{ingredients}

\blockrule{Bereiding}
\tip{Het is eigenlijk een opgerolde pizza, dus je kunt de vulling aanpassen
naar wat je lekker vindt: een paar plakjes salami of ham erdoor kan
bijvoorbeeld ook heel goed.}
\begin{steps}
  \item Verwarm de oven voor op ca.\ 180\,°C (boven- en onderwarmte), of
        volgens de temperatuur op de verpakking van het croissantdeeg.
  \item Ontrol het croissantdeeg uit de verpakking. Dit zijn drie paar
        driehoeken die telkens aan elkaar vastzitten; druk elke naad dicht
        zodat je drie rechthoeken deeg krijgt.
  \item Smeer de pastasaus gelijkmatig over het deeg en bestrooi met de
        mozzarella.
  \item Rol elke rechthoek vanaf de lange zijde strak op tot een rol.
  \item Snijd de rollen in gelijke plakken.
  \item Leg de rolletjes met wat afstand van elkaar in een ingevette
        ovenschaal, zodat ze tijdens het bakken kunnen uitzetten. Bestrooi
        met een beetje oregano.
  \item Bak 15--20 minuten tot de rolletjes goudbruin zijn en de kaas
        gesmolten is.
\end{steps}
\end{recipe}
