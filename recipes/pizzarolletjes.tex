\begin{recipe}{Pizzarolletjes}
\kicker[Bakken,Snel,Italiaans]{Bakken · snel · Italiaans}
\index[register]{Pizzarolletjes}
\index[register]{Mozzarella}
\index[register]{Croissantdeeg}

\meta{25 minuten \dvd 6--8 stuks \dvd Oven ca.\ 180\,°C}

\lettrine{D}{ezelfde} truc als de kaneelrolletjes, maar dan hartig: met
pastasaus en mozzarella in plaats van kaneel en suiker. Fijn als snack of
erbij.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 blikje}{croissantdeeg}\index[register]{Croissantdeeg}
  \ing{200 g}{geraspte mozzarella}\index[register]{Mozzarella}
  \ing{1 pot}{pastasaus met basilicum}
  \ingb{oregano, om te bestrooien}
\end{ingredients}

\blockrule{Bereiding}
\tip{Rol het deeg uit tot één vierkant in plaats van de driehoeken waar het
blikje voor bedoeld is, dan krijg je een mooie, gelijkmatige rol.}
\begin{steps}
  \item Verwarm de oven voor op ca.\ 180\,°C (boven- en onderwarmte), of
        volgens de temperatuur op de verpakking van het croissantdeeg.
  \item Rol het croissantdeeg uit tot één vierkant, druk de naden dicht.
  \item Smeer de pastasaus gelijkmatig over het deeg en bestrooi met de
        mozzarella.
  \item Rol het vierkant strak op tot een rol.
  \item Snijd de rol in gelijke plakken.
  \item Leg de rolletjes met wat afstand van elkaar in een ingevette
        ovenschaal, zodat ze tijdens het bakken kunnen uitzetten. Bestrooi
        met een beetje oregano.
  \item Bak 15--20 minuten tot de rolletjes goudbruin zijn en de kaas
        gesmolten is.
\end{steps}
\end{recipe}
