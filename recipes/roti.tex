\begin{recipe}{Roti met kip}
\kicker{Hoofdgerecht · Surinaams · doordeweeks}
\index[register]{Roti met kip}
\index[register]{Kip}
\index[register]{Krieltjes}
\index[register]{Sperziebonen}

\meta{45 minuten \dvd Surinaams}

\lettrine{R}{oti} is een Surinaams gerecht dat je met volkoren wraps goed thuis maakt.
De saus van gekruide kip met krieltjes en tomaat is de avond van tevoren te bereiden,
zodat het op de dag zelf snel op tafel staat.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{2}{kipdijfilets, in blokjes}\index[register]{Kipdijfilet}
  \ing{300 g}{krielftjes met schil, gehalveerd}\index[register]{Krieltjes}
  \ing{1 bak}{roma tomaten, in blokjes}\index[register]{Tomaten}
  \ing{1}{ui, gesnipperd}
  \ing{1--2}{knoflooktenen, fijngehakt}
  \ing{1}{kipbouillonblokje}
  \ingb{bakboter}
  \ing{1 zak}{voorgesneden sperziebonen}\index[register]{Sperziebonen}
  \ing{5}{eieren}
  \ing{5}{volkoren wraps}
  \ing{1 el}{kerrie}
  \ing{1 el}{komijn}
\end{ingredients}

\blockrule{Bereiding}
\tip{De saus kan de avond van tevoren worden gemaakt. Op de dag zelf hoef je hem dan alleen nog op te warmen.}
\begin{steps}
  \item Halveer de krieltjes en kook ze 10 minuten voor in gezouten water, zodat ze
        bijna gaar zijn. Giet af en zet apart.
  \item Snijd de kipdijfilets in blokjes, snipper de ui, hak de knoflook fijn en snijd
        de tomaten in blokjes.
  \item Verhit een ruime pan en smelt daarin een klont bakboter. Bak de kip op
        middelhoog vuur rondom bruin.
  \item Voeg de ui toe en bak tot glazig. Voeg de knoflook, kerrie en komijn toe en
        roer een minuut mee.
  \item Voeg de krieltjes toe en bak ze 1--2 minuten mee. Voeg dan de tomatenblokjes,
        het bouillonblokje en een scheut water toe.
  \item Leg het deksel op de pan en laat 10 minuten zachtjes pruttelen.
  \item Kook de sperziebonen volgens de verpakking, maar houd ze iets knapperig.
        Kook de eieren tegelijk in een aparte pan (10 minuten voor hardgekookt).
  \item Warm de wraps op in een droge koekenpan. Warm de saus op als je die de dag
        ervoor hebt gemaakt.
  \item Dien op: leg een wrap op het bord, schep de saus erop en schik de sperziebonen
        en een gepeld ei erbij.
\end{steps}
\end{recipe}
