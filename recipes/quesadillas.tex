\begin{recipe}{Quesadilla's}
\kicker[Hoofdgerecht,Vegetarisch,Mexicaans]{Hoofdgerecht · vegetarisch · Mexicaans}
\index[register]{Quesadilla's}
\index[register]{Wraps}
\index[register]{Zwarte bonen}
\index[register]{Mais}

\meta{30 minuten}

\lettrine{W}{eer} een makkelijk recept voor op de camping: er hoeft niets in de
koeling behalve de kaas. De vulling is op smaak met komijn en koriander, en de
wrap wordt in de droge koekenpan goudbruin gebakken tot de kaas erin gesmolten is.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{200 g}{geraspte belegen kaas}\index[register]{Kaas}
  \ing{400 g}{zwarte bonen of kidneybonen}\index[register]{Zwarte bonen}
  \ing{6}{volkoren wraps}
  \ing{1}{rode paprika}\index[register]{Paprika}
  \ing{150 g}{mais}\index[register]{Mais}
  \ing{1}{losse tomaat}
  \ing{1}{ui}
  \ing{3 stengels}{lente-ui}
  \ing{1}{knoflookteentje}\index[register]{Knoflook}
  \ingb{olijfolie}
  \ingb{peterselie, naar smaak}
  \ingb{komijn}
  \ingb{koriander, gemalen}
  \ingb{paprikapoeder}
  \ingb{chilipoeder}
  \ingb{versgemalen peper en zout}
  \ingb{zure room of crème fraîche met een beetje sambal, om erbij te serveren}
\end{ingredients}

\blockrule{Bereiding}
\tip{Het sausje van zure room of crème fraîche met een beetje sambal zorgt voor
frisheid tegenover de romige kaas en bonen. Heb je wat meer tijd? Maak de wraps
dan zelf (p.~\pageref{recipe:wraps}).}
\begin{steps}
  \item Was de paprika en de tomaat. Snijd de paprika klein en snipper de ui. Hak de knoflook fijn. Snijd de
        lente-ui in ringetjes en de peterselie fijn. Snijd de tomaat klein. Giet
        de bonen en de mais af en spoel ze om.
  \item Bak de ui, de paprika en de knoflook op middelhoog vuur in een hapjespan
        in olijfolie tot de ui glazig is. Voeg de kruiden toe.
  \item Voeg na een minuutje de bonen en de mais toe. Zodra ze warm zijn, voeg de
        tomaat toe en roerbak nog een minuutje. Zet het vuur uit.
  \item Zet een andere koekenpan op laag vuur. Leg een wrap in de pan en verdeel
        een zesde van de kaas erover.
  \item Schep een derde van de vulling erop en verdeel nog een zesde van de kaas
        erover. Leg de tweede wrap erbovenop en druk licht aan.
  \item Laat een paar minuten bakken, leg dan een bord ondersteboven op de pan en
        draai de pan om. Laat de quesadilla terug in de pan glijden om de andere
        kant te bakken. Nog niet bruin genoeg? Draai hem gerust nog een keer om.
  \item Herhaal dit voor de andere twee quesadilla's.
  \item Snijd elke quesadilla in parten, zoals een pizza, zodat je ze samen
        kunt delen.
  \item Bestrooi voor het serveren met wat peterselie.
\end{steps}
