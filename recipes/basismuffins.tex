\begin{recipe}{Basismuffins}
\kicker{Bakken · basisrecept}
\index[register]{Basismuffins}
\index[register]{Bloem}
\index[register]{Eieren}

\meta{35 minuten \dvd Voor 8 muffins}

\lettrine{M}{et} dit basisrecept kun je alle kanten op. De muffins worden
luchtig en zijn zo klaar om naar smaak aan te vullen, met chocolade, fruit of
noten.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{120 g}{fijne kristalsuiker}
  \ing{300 g}{bloem}\index[register]{Bloem}
  \ing{1 tl}{bakpoeder}
  \ing{1 tl}{baking soda}
  \ingb{snuf zout}
  \ing{2}{eieren (M)}\index[register]{Eieren}
  \ing{250 ml}{melk}
  \ing{120 ml}{zonnebloemolie}
  \ing{2 tl}{vanille-extract}
\end{ingredients}

\blockrule{Bereiding}
\tip{Zijn de muffins niet smeuïg of luchtig genoeg uit de oven gekomen? Dan is
het beslag waarschijnlijk te lang gemixt, of hebben de muffins te lang of te
heet in de oven gestaan.}
\begin{steps}
  \item Doe de suiker, bloem, bakpoeder, baking soda en het zout bij elkaar in
        een kom en roer even door met een spatel.
  \item Meng in een andere kom de eieren, melk, zonnebloemolie en het
        vanille-extract met een garde door elkaar.
  \item Voeg de inhoud van beide kommen samen en spatel tot één geheel. Dit
        mag best een beetje klonterig zijn.
  \item Bekleed een muffinvorm met papieren muffinvormpjes. Verdeel het
        beslag met een ijslepel over de vormpjes, tot ongeveer een halve
        centimeter onder de rand.
  \item Bak de muffins 20 minuten op 200\,°C (boven- en onderwarmte) gaar en
        goudbruin.
\end{steps}
\end{recipe}
