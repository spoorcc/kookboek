\begin{recipe}{Spinaziequiche}
\kicker[Hoofdgerecht,Vegetarisch,Hartig]{Hoofdgerecht · vegetarisch · hartig}
\index[register]{Spinaziequiche}
\index[register]{Spinazie}
\index[register]{Zongedroogde tomaten}

\meta{1 uur 15 min}

\lettrine{E}{en} quiche met een vulling van spinazie, zongedroogde tomaten en crème
fraîche. Reken op een halfuur rusttijd voor het deeg voor je kunt beginnen.

\blockrule{Ingrediënten}

\noindent\textit{Deeg}
\begin{ingredients}
  \ing{150 g}{bloem}
  \ing{75 g}{boter}
  \ing{1}{ei}
  \ing{1 snuf}{zout}
\end{ingredients}

\medskip
\noindent\textit{Vulling}
\begin{ingredients}
  \ing{450 g}{bladspinazie (diepvries), ontdooid}\index[register]{Spinazie}
  \ing{1}{rode ui, gesnipperd}
  \ing{2}{knoflookteentjes, fijngehakt}
  \ing{1 potje}{zongedroogde tomaten, uitgelekt en grof gesneden}\index[register]{Zongedroogde tomaten}
  \ing{125 ml}{crème fraîche}
  \ing{2}{eieren}
  \ingb{cherrytomaatjes, gehalveerd}
  \ingb{handvol geraspte kaas}
  \ingb{versgemalen peper, zout en nootmuskaat}
\end{ingredients}

\blockrule{Bereiding}
\tip{Geen zin in zelfgemaakt deeg? Een vel bladerdeeg uit de koeling werkt ook goed
en scheelt je de rusttijd.}
\begin{steps}
  \item Meng bloem, boter, ei en zout tot een samenhangend deeg. Kneed kort door,
        maak een bol en laat 30 minuten rusten in de koelkast.
  \item Knijp het overtollige vocht uit de ontdooide spinazie.
  \item Verwarm de oven voor op 200\,°C (boven- en onderwarmte).
  \item Fruit de rode ui en knoflook in een scheutje olie. Voeg de spinazie toe en
        bak het resterende vocht er zo veel mogelijk uit. Roer de zongedroogde tomaten
        erdoor en breng op smaak met peper, zout en nootmuskaat.
  \item Klop de crème fraîche en de 2 eieren los in een kom. Meng het
        spinaziemengsel erdoor.
  \item Rol het deeg uit en bekleed een springvorm ermee. Giet de vulling erin.
        Leg de cherrytomaatjes bovenop en strooi de geraspte kaas erover.
  \item Bak de quiche 45 minuten op 200\,°C tot de vulling gestold is en de bovenkant
        goudbruin.
\end{steps}
\end{recipe}
