\begin{recipe}{Nasi goreng}
\kicker{Hoofdgerecht · Indonesisch · doordeweeks}
\index[register]{Nasi goreng}
\index[register]{Kip}
\index[register]{Rijst}

\meta{25 minuten \dvd Indonesisch}

\lettrine{N}{asi} goreng met kip, een gebakken ei en pindasaus. Dit recept gebruikt
een boemboe uit een pakje, wat het op een doordeweekse avond een stuk sneller maakt.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{300 g}{langkorrelige rijst}\index[register]{Rijst}
  \ing{2}{kipfilets, in blokjes}\index[register]{Kip}
  \ing{1 zak}{nasi-bamigroenten}
  \ing{1 pakje}{nasi boemboe (b.v.\ Lidl)}
  \ingb{kokosolie}
  \ing{5}{eieren (1 per persoon)}
  \ingb{pindasaus (pakje of zelfgemaakt)}
  \ingb{gebakken uitjes}
  \ingb{kroepoek}
  \ingb{sambal, voor de liefhebber}
\end{ingredients}

\blockrule{Bereiding}
\tip{Rijst van de dag ervoor werkt het best. Koude rijst bakt droger en minder
plakkerig.}
\begin{steps}
  \item Kook de rijst en laat afkoelen.
  \item Verhit een wokpan goed voor en doe een scheut kokosolie erin. Bak de
        kipblokjes tot er een bruine kant op zit.
  \item Voeg de groenten toe en bak kort mee.
  \item Voeg de boemboe toe en schep goed door.
  \item Voeg de rijst toe en roer op laag vuur door tot alles goed gemengd en warm is.
  \item Bak ondertussen een ei per persoon in een koekenpan en maak de pindasaus
        klaar.
  \item Serveer met een gebakken ei erop en gebakken uitjes, kroepoek en sambal
        erbij.
\end{steps}
\end{recipe}
