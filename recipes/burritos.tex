\begin{recipe}{Burrito's}
\kicker[Hoofdgerecht,Vlees,Mexicaans,Doordeweeks]{Hoofdgerecht · vlees · Mexicaans · doordeweeks}
\index[register]{Burrito's}
\index[register]{Gehakt}
\index[register]{Kidneybonen}
\index[register]{Wraps}

\meta{30 minuten \dvd Voor 4 personen \dvd Oven 200\,°C}

\lettrine{E}{en} gevulde wrap uit de oven, met een pittige mix van gehakt, bonen
en groente. De vulling maak je op het fornuis, daarna gaat alles samen de oven
in tot de kaas gesmolten is.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{200 g}{gehakt}
  \ing{4}{volkoren wraps}\index[register]{Wraps}
  \ing{1}{rode paprika}\index[register]{Paprika}
  \ing{1}{ui}
  \ing{400 g}{rode kidneybonen}\index[register]{Kidneybonen}
  \ing{1 handje}{cherrytomaatjes}\index[register]{Tomaten}
  \ing{1}{knoflookteentje}
  \ing{150 g}{mais}\index[register]{Mais}
  \ing{75 g}{geraspte kaas}\index[register]{Kaas}
  \ing{1 tl}{komijnpoeder}
  \ing{1 tl}{koriander, gemalen}
  \ing{125 g}{zure room of crème fraîche met een beetje sambal, om erbij te serveren}
  \ingb{zout}
  \ingb{bakboter}
  \ingb{salsa, om erbij te serveren}
\end{ingredients}

\blockrule{Bereiding}
\tip{Lekker met salsa erbij, en het sausje van zure room of crème fraîche met
een beetje sambal zorgt voor frisheid tegenover de romige kaas en bonen. Heb
je wat meer tijd? Maak de wraps dan zelf (p.~\pageref{recipe:wraps}).}
\begin{steps}
  \item Verwarm de oven voor op 200 °C (boven- en onderwarmte).
  \item Snipper de ui en hak de knoflook fijn. Was de paprika en de cherrytomaatjes.
        Snijd de paprika in blokjes en de cherrytomaatjes klein. Giet de bonen af en spoel ze om.
  \item Bak het gehakt in een hapjespan op hoog vuur rul in wat bakboter. Voeg de ui,
        de knoflook en de kruiden toe.
  \item Bak de paprika en de tomaatjes een paar minuten mee.
  \item Voeg de bonen en de mais toe en bak alles goed door elkaar tot het
        gemengd is. Haal de pan van het vuur.
  \item Vet een ovenschaal in, anders plakken de wraps vast. Verdeel de vulling
        over de wraps, rol ze op en leg ze naast elkaar in de schaal. Bestrooi
        met de geraspte kaas.
  \item Zet de schaal in de oven tot de kaas gesmolten is.
\end{steps}
\end{recipe}
