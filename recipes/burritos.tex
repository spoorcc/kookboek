\begin{recipe}{Burrito's}
\kicker[Hoofdgerecht,Vlees,Mexicaans,Doordeweeks]{Hoofdgerecht · vlees · Mexicaans · doordeweeks}
\index[register]{Burrito's}
\index[register]{Rundergehakt}
\index[register]{Kidneybonen}
\index[register]{Wraps}

\meta{30 minuten \dvd Oven 200\,°C}

\lettrine{E}{en} gevulde wrap uit de oven, met een pittige mix van gehakt, bonen
en groente. De vulling maak je op het fornuis, daarna gaat alles samen de oven
in tot de kaas gesmolten is.

\blockrule{Ingrediënten}

\noindent\textit{Gerecht}
\begin{ingredients}
  \ing{200 g}{rundergehakt}
  \ing{5}{volkoren wraps}\index[register]{Wraps}
  \ing{1}{rode paprika}\index[register]{Paprika}
  \ing{1}{ui}
  \ing{2}{lente-ui}\index[register]{Lente-ui}
  \ing{400 g}{rode kidneybonen}\index[register]{Kidneybonen}
  \ing{1 handje}{cherrytomaatjes}\index[register]{Tomaat}
  \ing{1}{knoflookteentje}
  \ing{150 g}{maïs}\index[register]{Maïs}
  \ing{150--200 g}{bladspinazie (diepvries), ontdooid}\index[register]{Spinazie}
  \ing{75 g}{jong belegen geraspte kaas}\index[register]{Kaas}
  \ing{2 el}{salsa, door de vulling}
  \ing{125 g}{zure room of crème fraîche met sambal, om erbij te serveren}
  \ingb{bakboter}
  \ingb{salsa, om erbij te serveren}
\end{ingredients}

\medskip
\noindent\textit{Kruidenmix}
\begin{ingredients}
  \ing{1 tl}{zout}
  \ing{1 tl}{komijnpoeder}
  \ing{1 tl}{koriander, gemalen}
  \ing{1 tl}{paprikapoeder}
  \ingb{snufje cayennepeper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Lekker met salsa of het frisse sausje van zure room met sambal erbij. Heb
je wat meer tijd? Maak de wraps dan zelf (p.~\pageref{recipe:wraps}).}
\begin{steps}
  \item Verwarm de oven voor op 200 °C (boven- en onderwarmte).
  \item Snipper de ui en hak de knoflook fijn. Was de lente-ui, snijd in ringetjes
        en houd het witte deel apart van het groene deel. Was de paprika en de
        cherrytomaatjes. Snijd de paprika in blokjes en de cherrytomaatjes klein.
        Giet de bonen af en spoel ze om. Knijp het vocht stevig uit de ontdooide
        spinazie.
  \item Bak het gehakt in een hapjespan op hoog vuur rul in wat bakboter. Voeg de
        ui, het witte deel van de lente-ui, de knoflook en de kruidenmix toe.
        Roer de salsa erdoor.
  \item Bak de paprika en de tomaatjes een paar minuten mee, op hoog vuur zodat
        de vulling droog blijft en de wraps niet zompig worden.
  \item Voeg de bonen en de maïs toe en bak de uitgeknepen spinazie de laatste
        2 minuten mee tot alles gemengd is. Haal de pan van het vuur.
  \item Vet een ovenschaal in, anders plakken de wraps vast. Verdeel de vulling
        over de wraps, rol ze op en leg ze naast elkaar in de schaal. Bestrooi
        met de geraspte kaas en zet de schaal in de oven tot de kaas gesmolten
        is.
  \item Strooi het groene deel van de lente-ui over de burritos zodra ze goudbruin
        uit de oven komen.
\end{steps}

\end{recipe}
