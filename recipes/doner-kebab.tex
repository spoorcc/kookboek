\begin{recipe}{Döner kebab}
\kicker[Hoofdgerecht,Vlees,Turks]{Hoofdgerecht · vlees · Turks}
\index[register]{Döner kebab}
\index[register]{Gehakt}
\index[register]{Vlees}

\meta{50 minuten \dvd Voor 5 personen}

\lettrine{E}{en Turkse klassieker zonder draaiende spit: het gehakt wordt flinterdun uitgerold tussen bakpapier en in de oven gaar en krokant gebakken. Serveer het warm in pita met knoflooksaus en een fris ui-sumaksalade.}

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{500 g}{half-om-half gehakt}\index[register]{Half-om-half gehakt}
  \ing{1}{ui, in grove stukken}
  \ing{2}{knoflookteentjes, fijngehakt}
  \ing{2 el}{Griekse yoghurt}
  \ing{2 el}{olijfolie}
  \ing{1 tl}{zout}
  \ing{1,5 tl}{paprikapoeder}
  \ing{1 tl}{gedroogde oregano}
  \ing{1 tl}{komijnpoeder}
  \ing{1,5 tl}{rode peppervlokken}
  \ing{0,5 tl}{zwarte peper}
\end{ingredients}

\blockrule{Voor de vulling}
\begin{ingredients}
  \ing{5}{shoarmabroodjes (zelfgebakken, zie blz.\,\pageref{recipe:shoarmabroodjes}) of pita}\index[register]{Shoarmabroodjes}
  \ing{1}{tomaat, in stukken}
  \ing{½ krop}{sla}
  \ing{1}{rode ui, in ringen}
  \ing{1}{komkommer, in blokjes}
  \ingb{knoflooksaus}
  \ingb{sambal, voor liefhebber}
\end{ingredients}

\blockrule{Bereiding}
\tip{Kneed het gehakt 3 tot 5 minuten intensief: het vlees moet glad en kleverig worden en bijna als paté voelen, zodat je straks nette plakken kunt snijden. Voor zelfgebakken broodjes: zie blz.\,\pageref{recipe:shoarmabroodjes}.}
\begin{steps}
  \item Blend de ui tot fijn puree. Meng het gehakt, yoghurt, olijfolie, uipuree, knoflook en kruiden in een kom. Kneed alles 3 tot 5 minuten met je handen tot een glad mengsel, of gebruik een keukenmachine.
  \item Verdeel de vleesmassa in 2 gelijke porties. Leg bakpapier op het aanrecht, plaats één portie in het midden en dek af met een tweede vel bakpapier.
  \item Rol het vlees met een deegroller flinterdun uit tot 2--3 mm dik.
  \item Haal het bovenste vel papier eraf. Pak het onderste vel aan de lange zijde, rol het vleesblad strak op tot een cilinder (zonder papier mee) en wikkel in het papier als een snoepje. Herhaal met de tweede portie.
  \item Verwarm de oven voor op 200°C. Leg de twee rollen op een bakplaat en bak 20 minuten. Bereid ondertussen de groente voor: snipper de tomaat, ui en komkommer, en scheur de sla grof.
  \item Haal de bakplaat eruit, giet het vet en vocht voorzichtig weg.
  \item Haal de rollen uit het papier, leg ze terug op de (droge) bakplaat en bak nog 15 minuten tot de buitenkant goudbruin is.
  \item Laat 5 minuten rusten en snijd in flinterdunne plakjes.
  \item Bak de plakjes 3--4 minuten in een droge pan of onder de grill tot de randjes donkerbruin en krokant worden.
  \item Snijd de broodjes open en vul met döner, groente en sauzen naar voorkeur.
\end{steps}
\end{recipe}
