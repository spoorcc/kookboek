\begin{recipe}{Pasta carbonara}
\kicker[Hoofdgerecht,Vlees,Italiaans]{Hoofdgerecht · vlees · Italiaans}
\index[register]{Pasta carbonara}
\index[register]{Spaghetti}
\index[register]{Spek}
\index[register]{Parmezaanse kaas}

\meta{25 minuten}

\lettrine{C}{arbonara} is een Romeins gerecht dat zijn romigheid alleen aan ei en
kaas dankt, zonder room. Traditioneel gaat er pecorino romano in, of half-om-half
met Parmezaan: het zoute, hartige randje van deze schapenkaas is kenmerkend voor
het gerecht. De rucola is zeker niet traditioneel, maar de lichte pepersmaak past
goed bij het gerecht. Vertel het maar niet aan een Italiaan.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{4}{dooiers}\index[register]{Eieren}
  \ing{150 g}{guanciale (varkenswang) of ongerookt spek}\index[register]{Spek}\index[register]{Guanciale}
  \ing{350 g}{spaghetti (geen verse pasta)}\index[register]{Spaghetti}
  \ing{75 g}{Pecorino Romano en Parmezaanse kaas (half-om-half), geraspt}\index[register]{Pecorino}\index[register]{Parmezaanse kaas}
  \ingb{handje rucola}
  \ingb{zwarte peperkorrels}
\end{ingredients}

\blockrule{Bereiding}
\tip{Gebruik bij voorkeur guanciale (varkenswang): soms te vinden bij de AH, verder
nauwelijks. Ongerookt spek werkt ook. De overgebleven eiwitten kun je gebruiken voor
meringues (p.~\pageref{recipe:meringues}).}
\begin{steps}
  \item Kook de spaghetti een minuut korter dan de verpakking aangeeft. De pasta
        gaart straks verder in de hapjespan met de saus. Bewaar een kopje kookvocht
        voordat je afgiet.
  \item Bak het spek uit in een hapjespan op middelhoog vuur tot het knapperig is.
  \item Rooster de peperkorrels op laag vuur 30 seconden in een droog pannetje en maal ze dan:
        dat is het verschil tussen peper als prikkel en peper als smaak.
  \item Klop de dooiers los, rasp er een flinke hoeveelheid van de kaas door en
        maal er royaal van de geroosterde peper over. Roer daarna een schep heet
        kookvocht door dit mengsel zodat de kaas niet draderig wordt.
  \item Voeg de afgegoten pasta toe aan het spek en zet het vuur uit. Giet het
        ei-kaasmengsel erover en roer snel door. De restwarmte gaart het ei tot een
        romige saus. Voeg indien nodig nog een scheutje kookvocht toe.
  \item Roer een hand rucola door de pasta. Schep op een diep bord en rasp nog wat
        extra kaas erover.
\end{steps}
\end{recipe}
