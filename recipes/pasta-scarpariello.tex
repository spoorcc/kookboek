\begin{recipe}{Pasta allo scarpariello}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans,Pittig]{Hoofdgerecht · vegetarisch · Italiaans · pittig}
\index[register]{Pasta allo scarpariello}
\index[register]{Tomaat}
\index[register]{Basilicum}
\index[register]{Spaghetti}

\meta{25 minuten}

\lettrine{S}{carpariello} is een Napolitaanse techniek waarbij cherry tomaatjes op laag
vuur langzaam in olijfolie opengaan. Het sap mengt zich met de olie tot een lichte,
smaakvolle saus. Een zomers recept met weinig ingrediënten, ook prima te maken op de camping.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{350 g}{spaghetti of andere pasta}\index[register]{Spaghetti}
  \ing{250 g}{cherry tomaten}\index[register]{Tomaat}
  \ing{2}{knoflookteentjes}\index[register]{Knoflook}
  \ing{3 el}{Parmezaanse kaas, geraspt}\index[register]{Parmezaanse kaas}
  \ing{1 plantje}{basilicum}\index[register]{Basilicum}
  \ingb{olijfolie}
  \ingb{chilivlokken}
\end{ingredients}

\blockrule{Bereiding}
\tip{Gebruik pasta trafilata al bronzo, herkenbaar aan het ruwe, matte oppervlak. Die neemt de saus beter op dan de gladde variant.}
\begin{steps}
  \item Was de cherry tomaatjes en halveer ze. Pel de knoflookteentjes maar laat ze heel.
  \item Verhit een ruime scheut olijfolie in een brede hapjespan op laag vuur. Voeg de
        knoflookteentjes en een snuf chilivlokken toe.
  \item Zet ondertussen water op voor de pasta.
  \item Als de knoflook lichtjes begint te kleuren, leg dan de tomaatjes met de
        snijkant naar beneden in de olie.
  \item Als het water kookt, doe de pasta erin en kook volgens de verpakking.
  \item Als de tomaatjes beginnen te smelten, schep een soeplepel pastawater in de
        saus. Voeg af en toe een klein scheutje extra pastawater toe als de saus te
        droog wordt, maar houd haar dik.
  \item Leg 3 hele takken basilicum in de pan.
  \item Haal vlak voor de pasta gaar is met een tang de knoflook en basilicumtakken
        eruit.
  \item Giet de pasta af en schep door de saus.
  \item Rasp de parmezaan boven de pan, meng goed door en blijf de pan bewegen om de
        saus te laten binden.
  \item Verdeel over borden en garneer met een paar verse basilicumblaadjes.
\end{steps}
