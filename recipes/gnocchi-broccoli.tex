\begin{recipe}{Gnocchi met broccoli en spek}
\kicker[Hoofdgerecht,Vlees,Italiaans]{Hoofdgerecht · vlees · Italiaans}
\index[register]{Gnocchi met broccoli en spek}
\index[register]{Gnocchi}
\index[register]{Broccoli}
\index[register]{Spek}
\index[register]{Parmezaanse kaas}

\meta{35 minuten}

\lettrine{E}{en} snelle pastaschotel met romige bouillonsaus,
knapperig spek en broccoli. Met verse gnocchi staat dit gerecht in ruim een half uur op tafel.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{2 pakken}{verse gnocchi}\index[register]{Gnocchi}
  \ing{1 stronk}{broccoli}\index[register]{Broccoli}
  \ing{250 g}{gerookte spekblokjes}\index[register]{Spek}
  \ing{1}{knoflookteentje}\index[register]{Knoflook}
  \ing{150 ml}{room}
  \ing{5 el}{geraspte Parmezaanse kaas}\index[register]{Parmezaanse kaas}
  \ing{350 ml}{kippenbouillon}
  \ingb{basilicum (+ extra voor de afwerking)}\index[register]{Basilicum}
  \ingb{citroensap, om af te maken}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Heb je wat meer tijd? Maak de gnocchi dan zelf (p.~\pageref{recipe:gnocchi}).}
\begin{steps}
  \item Pel en pers de knoflook. Verdeel de broccoli in roosjes.
  \item Bak de spekblokjes krokant in een droge hapjespan: het spek laat zelf genoeg vet
        los. Voeg de knoflook toe en roerbak 1 minuut mee. 
  \item Giet er de bouillon bij en laat die tot de helft
        inkoken. Voeg de broccoliroosjes, de room en de Parmezaanse kaas toe.
        Roer tot de broccoli beetgaar is.
  \item Kook de gnocchi volgens de aanwijzingen op de verpakking. Giet het af
        en laat ze goed uitlekken. Schep ze 2 minuten in een hete pan met een
        klontje boter of wat spekvet tot ze goudbruine vlekjes hebben: krokant
        randje buiten, zacht van binnen. Schep de gnocchi samen met het
        basilicum door de saus.
  \item Breng op smaak met peper, zout en een beetje citroensap. Serveer
        meteen met extra basilicum.
\end{steps}
\end{recipe}
