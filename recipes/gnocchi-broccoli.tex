\begin{recipe}{Gnocchi met broccoli en spek}
\kicker{Hoofdgerecht · Italiaans}
\index[register]{Gnocchi met broccoli en spek}
\index[register]{Gnocchi}
\index[register]{Broccoli}
\index[register]{Spek}
\index[register]{Parmezaanse kaas}

\meta{30 minuten \dvd Voor 3 personen \dvd Italiaans}

\lettrine{E}{en} snelle stamppot-achtige pastaschotel met romige bouillonsaus,
knapperig spek en broccoli. Kant-en-klare gnocchi maakt het een gerecht voor
doordeweeks, klaar voordat het water goed en wel kookt.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{400 g}{gnocchi (kant-en-klaar)}\index[register]{Gnocchi}
  \ing{100 g}{gerookte spekblokjes}\index[register]{Spek}
  \ing{1}{broccoli}\index[register]{Broccoli}
  \ing{1 teentje}{knoflook}\index[register]{Knoflook}
  \ingb{basilicum (+ extra voor de afwerking)}\index[register]{Basilicum}
  \ing{150 ml}{room}
  \ing{5 el}{geraspte Parmezaanse kaas}\index[register]{Parmezaanse kaas}
  \ing{350 ml}{kippenbouillon}
  \ingb{peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Lekker met een beetje citroensap erover vlak voor het serveren.}
\begin{steps}
  \item Pel en pers de knoflook. Verdeel de broccoli in roosjes. Bak de
        spekblokjes krokant in een droge koekenpan. Voeg de knoflook toe en
        roerbak 1 minuut mee. Giet er de bouillon bij en laat die tot de helft
        inkoken. Voeg de broccoliroosjes, de room en de Parmezaanse kaas toe.
        Roer tot de broccoli beetgaar is.
  \item Kook de gnocchi volgens de aanwijzingen op de verpakking. Giet ze af
        en laat ze uitlekken. Schep de gnocchi samen met het basilicum onder
        de saus. Breng op smaak met peper en zout. Serveer meteen met extra
        basilicum.
\end{steps}
\end{recipe}
