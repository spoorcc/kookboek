\begin{recipe}{Turkse pide}
\kicker[Hoofdgerecht,Vlees,Turks]{Hoofdgerecht · vlees · Turks}
\index[register]{Turkse pide}
\index[register]{Feta}
\index[register]{Spinazie}
\index[register]{Rundergehakt}

\meta{2 uur \dvd Voor 6 personen}

\lettrine{P}{ide} is Turks bootvormig brood met een gevulde bodem, gebakken tot
de randen goudbruin zijn. Het deeg heeft ruim een uur nodig om te rijzen; begin
daar op tijd mee. Hieronder staan twee vullingen: spinazie met feta, en gehakt
met tomaat.

\blockrule{Ingrediënten}

\noindent\textit{Deeg}
\begin{ingredients}
  \ing{500 g}{patentbloem, plus extra om te bestuiven}
  \ing{7 g}{droge gist (1 zakje)}
  \ing{1 tl}{suiker}
  \ing{300 ml}{lauwwarm water}
  \ing{2 el}{olijfolie}
  \ing{1 tl}{zout}
\end{ingredients}

\medskip
\noindent\textit{Vulling spinazie-feta}
\begin{ingredients}
  \ing{200 g}{verse spinazie}\index[register]{Spinazie}
  \ing{100 g}{feta, verkruimeld}\index[register]{Feta}
  \ing{1}{groene paprika, fijngesneden}
  \ing{1}{tomaat, in blokjes}
  \ing{1 el}{olijfolie}
  \ingb{zout en versgemalen peper}
\end{ingredients}

\medskip
\noindent\textit{Vulling gehakt-tomaat}
\begin{ingredients}
  \ing{250 g}{rundergehakt}\index[register]{Rundergehakt}
  \ing{400 g}{tomatenblokjes, uitgelekt}
  \ing{1}{ui, fijngesnipperd}
  \ing{1}{kleine groene paprika, fijngesneden}
  \ing{5 el}{olijfolie}
  \ing{100 g}{feta, verkruimeld}
  \ing{½ bosje}{peterselie of munt, fijngehakt}
  \ing{1 tl}{gemalen komijn}
  \ing{1 tl}{chilipoeder}
\end{ingredients}

\blockrule{Bereiding}
\tip{Laat de pides na het bakken even rusten voor je ze aansnijdt. Zo loopt de
vulling niet uit en blijft de korst intact.}

\medskip
\noindent\textit{Deeg}
\begin{steps}
  \item Los de gist op in een deel van het lauwwarme water met een snufje van
        de suiker. Laat 5 tot 10 minuten staan tot het gaat schuimen.
  \item Meng de bloem met het zout in een kom. Voeg het gistmengsel, de rest
        van het water en de olijfolie toe en kneed 10 tot 15 minuten tot een
        soepel, niet-plakkerig deeg.
  \item Dek de kom af met een vochtige doek en laat het deeg 1 tot 1{,}5 uur
        rijzen tot het in volume is verdubbeld. De gist heeft die tijd nodig
        om het deeg luchtig te maken.
\end{steps}

\medskip
\noindent\textit{Vulling spinazie-feta}
\begin{steps}
  \item Laat de spinazie in een hete pan slinken, zonder extra vet.
  \item Meng de spinazie met de paprika, tomaat en olijfolie.
  \item Verkruimel de feta erdoorheen en breng op smaak met zout en peper.
\end{steps}

\medskip
\noindent\textit{Vulling gehakt-tomaat}
\begin{steps}
  \item Bak het gehakt rul in de olijfolie.
  \item Voeg de ui, paprika, tomatenblokjes, komijn en chilipoeder toe en laat
        een paar minuten meebakken tot een dik mengsel.
  \item Haal van het vuur, roer de peterselie of munt erdoor en verkruimel de
        feta erover.
\end{steps}

\medskip
\noindent\textit{Vormen en bakken}
\begin{steps}
  \item Verwarm de oven voor op 200\,°C (boven- en onderwarmte), het liefst met een bakplaat of
        pizzasteen erin.
  \item Verdeel het gerezen deeg in 6 stukken en rol elk stuk uit tot een
        ovaal van ongeveer 30\,cm lang.
  \item Verdeel een van de vullingen over het midden van elk ovaal, met een
        rand van ongeveer 2\,cm vrij.
  \item Vouw de lange randen naar binnen over de vulling en knijp de
        uiteinden samen tot een bootvorm.
  \item Bestrijk de randen met olijfolie.
  \item Bak de pides 20 tot 25 minuten op de voorverwarmde bakplaat, tot de
        randen goudbruin zijn.
\end{steps}
\end{recipe}
