\begin{recipe}{IJstaart met karamel en chocolade}
\kicker[Dessert,Bevroren]{Dessert · bevroren}
\index[register]{IJstaart met karamel en chocolade}
\index[register]{Slagroom}
\index[register]{Karamel}
\index[register]{Chocolade}

\meta{30 minuten + 1 nacht invriezen \dvd Voor 8--10 personen}

\lettrine{E}{en} zelfgemaakte ijstaart in de stijl van een viennetta, zonder
ijsmachine. Het mengsel van gecondenseerde melk en slagroom geeft een romig
resultaat.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{397 ml}{gezoete gecondenseerde melk (1 blikje)}
  \ing{500 ml}{slagroom}\index[register]{Slagroom}
  \ing{½}{vanillestokje, het merg}
  \ing{100 g}{pure chocolade, gesmolten}\index[register]{Chocolade}
  \ing{1 potje}{karamelsaus (± 250 g, zelfgemaakt of uit pot)}\index[register]{Karamel}
\end{ingredients}

\blockrule{Bereiding}
\tip{Zet de taart minimaal een nacht in de vriezer. Een paar uur is niet genoeg.}
\begin{steps}
  \item Bekleed een cakevorm van 25 centimeter met plasticfolie en laat dit ruim aan
        de zijkanten overhangen.
  \item Doe de gecondenseerde melk, slagroom en het vanillemerg in een kom en klop
        tot een romig geheel met de dikte van lobbig geklopte slagroom.
  \item Verdeel een dun laagje roommengsel over de bodem van de vorm en strijk glad.
        Sprenkel wat gesmolten chocolade en karamelsaus over dit laagje.
  \item Schep voorzichtig een nieuw laagje roommengsel erop en strijk glad. Herhaal
        dit tot de vorm vol zit. Houd wat karamelsaus over voor de garnering.
  \item Klap het overhangende plasticfolie naar binnen. Zet de vorm minimaal 1 nacht
        in de vriezer.
  \item Haal de taart voor het serveren uit de vorm met behulp van het plasticfolie
        en leg hem op een schaal. Verwijder het folie voorzichtig en druppel de
        overgebleven karamelsaus erover.
\end{steps}
\end{recipe}
