\begin{recipe}{Meringues}
\label{recipe:meringues}
\kicker[Bakken,Vegetarisch,Frans]{Bakken · vegetarisch · Frans}
\index[register]{Meringues}
\index[register]{Eiwit}

\meta{15 minuten + 105 minuten oven \dvd Oven 90\,°C}

\lettrine{D}{it} recept is handig als je eiwit over hebt, na pasta carbonara
bijvoorbeeld. Van één eiwit krijg je al een mooi bakje meringues. Ze zijn
knapperig van buiten en licht van binnen.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1}{eiwit}\index[register]{Eiwit}
  \ing{50 g}{suiker}
  \ing{1 zakje}{vanillesuiker}
\end{ingredients}

\blockrule{Bereiding}
\tip{Zorg dat de kom en de garde volledig vetvrij zijn. Een spoortje vet en het
eiwit klopt niet stijf.}
\begin{steps}
  \item Klop het eiwit op tot het wit en schuimig is.
  \item Voeg de suiker en vanillesuiker lepel voor lepel toe en blijf kloppen tot
        alle korrels opgelost zijn en het mengsel stijf en glanzend is.
  \item Verwarm de oven voor op 90\,°C (boven- en onderwarmte). Spuit of schep kleine toefjes op een met
        bakpapier beklede bakplaat.
  \item Bak 105 minuten. Zet de oven uit en laat de meringues in de oven afkoelen.
        Na het afkoelen zijn ze stevig.
\end{steps}
\end{recipe}
