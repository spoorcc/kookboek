\begin{recipe}{Lahmacun}
\kicker[Hoofdgerecht,Vlees,Turks,Pittig]{Hoofdgerecht · vlees · Turks · pittig}
\index[register]{Lahmacun}
\index[register]{Lamsgehakt}
\index[register]{Paprika}
\index[register]{Peterselie}

\meta{1{,}5 uur}

\lettrine{L}{ahmacun} is een dun Turks flatbread met een laagje gekruid gehakt,
opgediend met verse toppings en opgerold. Het deeg heeft 45 minuten nodig om te
rijzen; begin daar op tijd mee. De pizza's worden per stuk gebakken in een droge
koekenpan.

\blockrule{Ingrediënten}

\noindent\textit{Deeg}
\begin{ingredients}
  \ing{250 g}{bloem, plus extra om te bestuiven}
  \ing{3{,}5 g}{instant gist (½ zakje)}
  \ing{75 ml}{melk}
  \ing{15 g}{margarine}
  \ing{1 tl}{zout}
\end{ingredients}

\medskip
\noindent\textit{Gehaktmengsel}
\begin{ingredients}
  \ing{150 g}{lamsgehakt of rundergehakt}\index[register]{Lamsgehakt}
  \ing{1}{paprika}\index[register]{Paprika}
  \ing{1}{tomaat}
  \ing{1}{rode ui}
  \ing{1 el}{tomatenpuree}
  \ing{1 el}{zonnebloemolie}
  \ing{50 ml}{water}
  \ing{½}{rode peper}
  \ing{1 tl}{paprikapoeder}
  \ingb{bosje verse peterselie}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\medskip
\noindent\textit{Toppings}
\begin{ingredients}
  \ingb{tomaten, fijngesneden}
  \ingb{ijsbergsla}
  \ingb{yoghurt of knoflooksaus}
\end{ingredients}

\blockrule{Bereiding}
\tip{Leg de klare pizza's onder een omgekeerde deksel of schone theedoek. Door de condens blijft de bodem soepel genoeg om op te rollen.}

\medskip
\noindent\textit{Deeg}
\begin{steps}
  \item Meng bloem, gist, zout, melk, 75 ml water en margarine in een kom en kneed
        tot een gladde deegbal.
  \item Dek de kom af en laat het deeg 45 minuten rijzen op kamertemperatuur.
        In een oven op 30--35\,°C gaat het ook goed.
\end{steps}

\medskip
\noindent\textit{Gehaktmengsel}
\begin{steps}
  \item Snijd de tomaat, paprika (zonder steel en zaadlijsten), rode peper en
        rode ui in stukken. Doe ze samen met de peterselie in een keukenmachine
        en maal fijn.
  \item Meng het groentemengsel door het gehakt. Voeg tomatenpuree,
        zonnebloemolie, paprikapoeder, water, peper en zout toe en meng goed.
        Druk het gehakt daarna wat fijn zodat er geen grote stukken meer in zitten.
\end{steps}

\medskip
\noindent\textit{Bakken (herhaal voor elk bolletje)}
\begin{steps}
  \item Verdeel het gerezen deeg in 5 bolletjes. Dek de rest af terwijl je er
        één verwerkt.
  \item Rol een bolletje heel dun uit op een met bloem bestoven werkvlak.
  \item Verhit een droge koekenpan op middelhoog vuur. Leg het deeg in de pan
        en besmeer de bovenzijde direct met een vijfde van het gehaktmengsel.
        Dek de pan af.
  \item Bak ca.\ 5 minuten. Beweeg af en toe met de pan zodat de pizza niet
        vastkomt. Controleer of het gehakt bovenop goed gaar is.
  \item Leg de klare pizza onder een deksel of theedoek en ga door met het
        volgende bolletje.
\end{steps}

\medskip
\noindent\textit{Serveren}
\begin{steps}
  \item Verdeel de toppings over de pizza's, rol ze op en serveer met yoghurt
        of knoflooksaus.
\end{steps}
\end{recipe}
