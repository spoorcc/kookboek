\begin{recipe}{Puffy pita}
\label{recipe:pita}
\kicker[Brood,Zelf bakken]{Brood \& basis · zelf bakken}
\index[register]{Puffy pita}
\index[register]{Gist}

\meta{Rijzen 1 uur \dvd 5 stuks \dvd Oven 240\,°C}

\lettrine{Z}{elfgebakken} pita die in een paar minuten opbolt tot een luchtig,
hol broodje. Plakkerig deeg, een gloeiend hete steen, en het komt helemaal
goed. Lekker om in de shakshuka te dippen.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{300 g}{tarwebloem}
  \ing{7 g}{instant gist}
  \ing{200 ml}{lauw water (67\%)}
  \ing{1 el}{suiker (ca.\ 4\%)}
  \ing{1 el}{olijfolie (ca.\ 5\%)}
  \ing{2 tl}{zout (ca.\ 3\%)}
\end{ingredients}

\blockrule{Bereiding}
\tip{Deeg te plakkerig? Een extra lepel bloem en nog even doorkneden lost het op.}
\begin{steps}
  \item Roer gist en suiker door het lauwe water en laat het 8–9 minuten tot
        leven komen. Meng de bloem met het zout in een grote kom.
  \item Roer de olijfolie door het gistmengsel, giet alles bij de bloem en meng
        tot er een deeg ontstaat.
  \item Kiep het deeg op de werkbank en kneed 2–3 minuten. Ja, het is plakkerig,
        en ja, het komt goed; gooi voor de zekerheid nog 1 el bloem erbij. Doe
        het in een ingevette kom, dek af met (ingeolied) vershoudfolie en laat 1 uur
        rijzen op een warme plek, zodat de gist zijn werk kan doen en het deeg
        luchtig wordt.
  \item Zet een bakplaat of pizzasteen in de oven en verhit minstens 1 uur op
        240\,°C. Hoe heter de steen, hoe mooier je pita's straks opbollen.
  \item Sla de lucht uit het deeg en verdeel in 5 bolletjes van ruim 100 g. Bol ze
        op, leg op een bebloemde plaat en laat een half uur rusten, zodat het
        gluten ontspant en het deeg zich straks makkelijker laat uitrollen.
  \item Knip het bakpapier in vier gelijke rechthoeken. Rol elk bolletje uit tot
        een schijf van 15 cm die net over de randen van zijn stukje bakpapier
        uitsteekt, dat maakt het straks makkelijker om de pita van het papier af
        op de hete plaat te schuiven.
  \item Bak de pita's 2–3 minuten op de hete plaat tot ze mooi puffy worden en
        draai ze dan nog even om.
  \item Haal de pita's uit de oven en leg ze in een mandje in een theedoek. Zo
        blijven ze warm en stomen ze lekker soepel na tot het serveren.
\end{steps}
\end{recipe}
