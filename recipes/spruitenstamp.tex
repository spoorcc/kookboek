\begin{recipe}{Spruitenstamp met brie en spekjes}
\kicker[Hoofdgerecht,Vlees,Nederlands,Doordeweeks]{Hoofdgerecht · vlees · Nederlands · doordeweeks}
\index[register]{Spruitenstamp met brie en spekjes}
\index[register]{Spruitjes}
\index[register]{Brie}
\index[register]{Stamppot}

\meta{35 minuten}

\lettrine{S}{pruitenstamp} met spekjes en brie is een winterse stamppot. De brie
smelt licht door de warme puree en de spekjes zorgen voor wat bite. De
kinderen dachten lang dat ze geen spruitjes lustten, tot we ze
toverballen noemden: wie er eentje opat kon ons daarna even in een dier
toveren. Ineens waren ze wel lekker en interessant.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{500 g}{spruitjes, schoongemaakt}\index[register]{Spruitjes}
  \ing{1 stuk}{brie (± 125 g)}\index[register]{Brie}
  \ing{250 g}{spekblokjes}\index[register]{Spekjes}
  \ing{1 kg}{kruimige aardappelen, geschild en in stukken}\index[register]{Aardappelen}
  \ing{1}{knoflookteentje, fijngehakt}
  \ing{1 tl}{kerrie}
  \ing{1 tl}{uienpoeder}
  \ing{1 tl}{versgemalen peper}
  \ing{1 el}{mosterd}
  \ingb{scheutje melk}
  \ingb{klontje boter}
  \ingb{zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Voeg de brie op het allerlaatst toe en roer snel door. Hij hoeft niet volledig
te smelten: losse stukjes brie door de stamp zijn juist lekker.}
\begin{steps}
  \item Kook de aardappelen in gezouten water gaar, ongeveer 20 minuten. Kook de spruitjes in apart water 5 minuten tot ze beetgaar zijn. Giet af en
        laat even uitstomen zodat ze verder garen.
  \item Bak de spekjes op middelhoog vuur in een droge koekenpan knapperig. Voeg de spruitjes en de
        knoflook toe en bak een paar minuten mee. Voeg de kerrie en het uienpoeder
        toe en roer goed door.
  \item Giet de aardappelen af en laat goed droogstomen, dat geeft een romiger
        stamp in plaats van een waterige. Stamp ze met een scheut melk,
        een klontje boter en de mosterd tot een stevige puree. Breng op smaak met zout en peper.
  \item Roer het spruitjes-spekmengsel door de puree. Snijd de brie in stukken en roer rustig door de stamppot. Serveer direct.
\end{steps}
\end{recipe}
