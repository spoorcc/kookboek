\begin{recipe}{Quiche met chorizo en geitenkaas}
\kicker[Hoofdgerecht,Vlees,Hartig]{Hoofdgerecht · vlees · hartig}
\index[register]{Quiche met chorizo en geitenkaas}
\index[register]{Chorizo}
\index[register]{Geitenkaas}
\index[register]{Prei}

\meta{1 uur \dvd Voor 6 porties \dvd Oven 200\,°C}

\lettrine{E}{en} hartige quiche met uitgebakken chorizo, prei, walnoten en
zongedroogde tomaat. De geitenkaas wordt er aan het einde over verkruimeld.
Deze variant heeft geen voorbakstap nodig.

\blockrule{Ingrediënten}

\noindent\textit{Deeg}
\begin{ingredients}
  \ing{300 g}{bloem}
  \ing{150 g}{roomboter}
  \ing{2}{eieren}
  \ing{1 tl}{zout}
\end{ingredients}

\medskip
\noindent\textit{Vulling}
\begin{ingredients}
  \ing{125 g}{chorizo, in stukjes}\index[register]{Chorizo}
  \ing{100 g}{geitenkaas}\index[register]{Geitenkaas}
  \ing{250 g}{prei (ca.\ 2 stelen), fijngesneden}\index[register]{Prei}
  \ing{75 g}{zongedroogde tomaat}\index[register]{Zongedroogde tomaten}
  \ing{50 g}{walnoten}\index[register]{Walnoten}
  \ing{5}{eieren (M)}
  \ing{200 g}{crème fraîche}
  \ingb{rozemarijn (gedroogd of vers)}
  \ingb{zout, versgemalen peper}
  \ingb{paneermeel (optioneel, voor op de bodem)}
\end{ingredients}

\blockrule{Bereiding}
\tip{Strooi een dun laagje paneermeel over de deegbodem voordat je de vulling erin
doet. Dat voorkomt dat de bodem vochtig wordt.}
\begin{steps}
  \item Kneed de deegingrediënten tot een samenhangend deeg. Maak een bol, wikkel
        in folie en laat 15 minuten rusten in de koelkast.
  \item Bak de prei in een koekenpan met een scheutje olijfolie tot hij zacht is.
        Giet overtollig vocht af en laat uitlekken.
  \item Bak de chorizo in een droge koekenpan tot hij knapperig is: hij laat zelf genoeg
        vet los. Giet het vet af en laat even uitdruipen op keukenpapier.
  \item Klop de eieren met de crème fraîche. Breng op smaak met rozemarijn, zout
        en peper.
  \item Verwarm de oven voor op 200\,°C (boven- en onderwarmte). Rol het deeg uit en bekleed de springvorm.
        Prik met een vork wat gaatjes in de bodem: dat voorkomt dat het deeg
        tijdens het bakken opbolt. Strooi eventueel een dun laagje paneermeel
        over de bodem.
  \item Verdeel de prei, chorizo, walnoten en zongedroogde tomaat over de bodem.
        Giet het eimengsel erover en verdeel het gelijkmatig. Verkruimel de
        geitenkaas erover.
  \item Bak 30--35 minuten op 200\,°C tot de quiche goudbruin en gaar is.
\end{steps}
\end{recipe}
