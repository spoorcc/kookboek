\begin{recipe}{Quiche met chorizo en geitenkaas}
\kicker[Hoofdgerecht,Vlees,Hartig]{Hoofdgerecht · vlees · hartig}
\index[register]{Quiche met chorizo en geitenkaas}
\index[register]{Chorizo}
\index[register]{Geitenkaas}
\index[register]{Prei}

\meta{90 minuten \dvd Voor 6 porties \dvd Oven 180\,°C}

\lettrine{E}{en} hartige quiche met uitgebakken chorizo, prei, walnoten en
zongedroogde tomaat. De geitenkaas wordt er aan het einde over verkruimeld.
Het deeg is hetzelfde als bij de andere quiches in dit boek.

\blockrule{Ingrediënten}

\noindent\textit{Deeg}
\begin{ingredients}
  \ing{300 g}{tarwebloem}
  \ing{150 g}{ongezouten roomboter}
  \ing{2}{eieren}
  \ing{1 tl}{zout}
\end{ingredients}

\medskip
\noindent\textit{Vulling}
\begin{ingredients}
  \ing{125 g}{chorizo, in stukjes}\index[register]{Chorizo}
  \ing{100 g}{geitenkaas}\index[register]{Geitenkaas}
  \ing{250 g}{prei (ca.\ 2 stelen), fijngesneden}\index[register]{Prei}
  \ing{75 g}{zongedroogde tomaat, grof gesneden}\index[register]{Zongedroogde tomaten}
  \ing{50 g}{walnoten}\index[register]{Walnoten}
  \ing{4}{eieren (M)}
  \ing{200 g}{crème fraîche}
  \ingb{rozemarijn (gedroogd of vers)}
  \ingb{zout, versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Laat de chorizo goed uitlekken voor je hem in de vulling verwerkt: te veel
vet maakt de bodem vochtig in plaats van stevig.}
\begin{steps}
  \item Kneed de deegingrediënten tot een samenhangend deeg. Maak een bol, wikkel
        in vershoudfolie en laat 15 minuten rusten in de koelkast.
  \item Verwarm de oven voor op 180\,°C (boven- en onderwarmte). Rol het deeg uit en bekleed de springvorm
        ermee. Gebruik je bladerdeeg, rol de plakken dan niet uit: druk 8
        plakken tegen de rand en 2 tegen de bodem, en laat het deeg goed over
        de rand hangen zodat het niet inzakt. Prik met een vork wat gaatjes
        in de bodem: dat voorkomt dat het deeg tijdens het voorbakken opbolt.
        Bedek de zijkanten eventueel met aluminiumfolie.
  \item Bak de bodem 15 minuten voor. Gebruik je bladerdeeg? Bak dan in
        plaats daarvan even voor op 220\,°C, tot de randen net beginnen te
        kleuren.
  \item Bak de prei op middelhoog vuur in een koekenpan met een scheutje olijfolie
        tot hij zacht is. Giet overtollig vocht af en laat uitlekken.
  \item Bak de chorizo op middelhoog vuur in een droge koekenpan tot hij knapperig
        is: hij laat zelf genoeg vet los. Giet het vet af en laat even uitdruipen op
        keukenpapier.
  \item Klop de eieren met de crème fraîche. Breng op smaak met rozemarijn, zout
        en peper.
  \item Verdeel de prei, chorizo, walnoten en zongedroogde tomaat over de
        voorgebakken bodem. Giet het eimengsel erover en verdeel het gelijkmatig.
        Verkruimel de geitenkaas erover.
  \item Bak de quiche 45 minuten op 180\,°C tot hij goudbruin en gaar is.
  \item Laat de quiche 10 minuten rusten voor het aansnijden.
\end{steps}

\end{recipe}
