\begin{recipe}{Quiche met ham en prei}
\kicker[Hoofdgerecht,Vlees,Hartig]{Hoofdgerecht · vlees · hartig}
\index[register]{Quiche met ham en prei}
\index[register]{Prei}
\index[register]{Ham}
\index[register]{Feta}

\meta{1{,}5 uur \dvd Voor 6 porties}

\lettrine{E}{en} stevige quiche met prei, ham en witte kaas. Het deeg is hetzelfde
als bij de andere quiches. Wie het snel wil doen gebruikt bladerdeeg.

\blockrule{Ingrediënten}

\noindent\textit{Deeg}
\begin{ingredients}
  \ing{300 g}{bloem}
  \ing{150 g}{roomboter}
  \ing{2}{eieren}
  \ing{1 tl}{zout}
\end{ingredients}

\medskip
\noindent\textit{Vulling}
\begin{ingredients}
  \ing{3--4}{preien, in ringen}\index[register]{Prei}
  \ing{250 g}{hamblokjes}\index[register]{Ham}
  \ing{150 g}{witte kaas (bijv.\ feta), verkruimeld}\index[register]{Feta}
  \ing{1}{knoflookteentje}
  \ing{4}{eieren}
  \ing{200 g}{crème fraîche}
  \ingb{kerrie, zout, versgemalen peper, knoflookpoeder}
\end{ingredients}

\blockrule{Bereiding}
\tip{Bedek de zijkanten van het deeg tijdens het voorbakken met aluminiumfolie
zodat ze niet naar beneden zakken.}
\begin{steps}
  \item Kneed de deegingrediënten tot een samenhangend deeg. Maak een bol, wikkel
        in folie en laat 15 minuten rusten in de koelkast.
  \item Verwarm de oven voor op 180\,°C (boven- en onderwarmte). Rol het deeg uit, bekleed de springvorm en
        prik met een vork wat gaatjes in de bodem: dat voorkomt dat het deeg
        tijdens het voorbakken opbolt. Bak de bodem 15 minuten voor. Bedek de zijkanten eventueel met aluminiumfolie.
  \item Bak de hamblokjes in een koekenpan tot ze beginnen te kleuren. Pers de
        knoflook erboven uit en voeg de prei in ringen toe.
  \item Roerbak de prei tot hij iets slap is. Hij gaart nauwelijks verder in de oven,
        dus zorg dat hij al redelijk gaar is.
  \item Roer de kruiden erdoor en haal de pan van het vuur. Laat het mengsel iets
        afkoelen en roer dan de witte kaas erdoor.
  \item Schep de vulling in de voorgebakken bodem. Klop de 4 eieren met de crème
        fraîche tot een glad mengsel en giet dit erover.
  \item Bak de quiche 45 minuten op 180\,°C tot de vulling gestold is.
\end{steps}
\end{recipe}
