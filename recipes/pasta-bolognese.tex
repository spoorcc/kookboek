\begin{recipe}{Pasta bolognese}
\kicker[Hoofdgerecht,Vlees,Italiaans,Doordeweeks]{Hoofdgerecht · vlees · Italiaans · doordeweeks}
\index[register]{Pasta bolognese}
\index[register]{Gehakt}
\index[register]{Soffrito}
\index[register]{Tomaten}

\meta{90 minuten}

\lettrine{D}{e} kinderen noemen dit pasta rode saus, hoewel het zeker niet de
enige rode saus in huis is. De saus draait om een fijn gesneden soffrito van
bleekselderij, winterpeen en ui in gelijke hoeveelheden, die naast het gehakt
voor body zorgt. Handig om in een grote pan te maken: in porties van 750 gram
ingevroren staat er op een drukke avond binnen een kwartier een warme
maaltijd op tafel.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{200 g}{half-om-half gehakt}\index[register]{Gehakt}
  \ing{350 g}{pasta naar keuze}\index[register]{Pasta}
  \ing{3 stengels}{bleekselderij}
  \ing{1}{winterpeen}
  \ing{1 grote}{ui}
  \ing{1 blik (400\,g)}{gepelde tomaten}\index[register]{Tomaten}
  \ing{70 g}{tomatenpuree}
  \ing{1}{runderbouillonblokje}
  \ing{50 g}{Parmezaanse kaas, geraspt}\index[register]{Parmezaanse kaas}
  \ingb{scheut witte wijn, voor afblussen}
  \ingb{olijfolie}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Deze saus is gemaakt om in te vriezen. Verdeel hem in porties van 750
gram en laat op een drukke dag rechtstreeks vanuit de vriezer opwarmen in een
hapjespan, terwijl de pasta al dente kookt.}
\begin{steps}
  \item Schil de winterpeen en snijd deze samen met de bleekselderij en de ui
        heel fijn, in gelijke hoeveelheden. Verhit een eetlepel olijfolie op
        laag vuur in een hapjespan en voeg de groente toe: dit
        heet soffrito, of mirepoix in de Franse keuken.
  \item Laat de soffrito rustig bakken op laag vuur. De groente hoeft niet
        bruin te worden: eerst komt het vocht eruit, waarna de smaak sterker
        wordt. Na 10 tot 15 minuten is de ui helemaal glazig. Haal de
        soffrito uit de pan.\index[register]{Soffrito}
  \item Zet het vuur flink hoger. Bak het gehakt rul met het
        bouillonblokje: het zout trekt vocht uit het vlees en zorgt voor
        bruine, gebakken stukjes die smaak geven.
  \item Voeg de tomatenpuree toe en bak deze onder goed roeren ongeveer een
        minuut mee. Dit ontzuurt de puree en maakt de saus zoeter.
  \item Blus af met een scheut witte wijn en wacht tot het vocht grotendeels
        is verdampt: dan is ook het grootste deel van de alcohol weg. Voeg
        de tomaten toe met een half blik water.
  \item Voeg de soffrito weer toe en roer goed door.
  \item Doe het deksel op de pan, zet het vuur laag en laat de saus minstens
        een uur heel zachtjes doorsudderen, af en toe een bubbeltje. Is de
        saus aan het eind nog te nat, laat hem dan nog even zonder deksel
        doorgaan. Is hij te droog, voeg dan een scheutje water toe. Wil je
        de saus direct gebruiken, houd hem dan liever iets aan de droge
        kant, zolang hij maar niet aanbrandt.
  \item Voor de vriezer: haal de pan van het vuur en laat de saus afkoelen
        tot kamertemperatuur voordat je hem in porties verdeelt over
        ziplockzakjes.
  \item Kook de pasta al dente. Meng de saus door de pasta, eventueel met een
        scheutje kookvocht, en roer er ruim Parmezaanse kaas door.
\end{steps}
\end{recipe}
