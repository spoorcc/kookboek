\begin{recipe}{Tomatensoep van geroosterde tomaten}
\kicker[Voorgerecht,Soep,Italiaans]{Voorgerecht · soep · Italiaans}
\index[register]{Tomatensoep van geroosterde tomaten}
\index[register]{Tomaten}
\index[register]{Kippenbouillon}

\meta{50 minuten \dvd Voor 4 personen}

\lettrine{G}{eroosterde} tomaten geven deze soep een diepe, geconcentreerde smaak.
De tomaten karamelliseren in de oven, en daarna heb je weinig extra nodig.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 kg}{pruimtomaten, gehalveerd}\index[register]{Tomaten}
  \ing{4 el}{olijfolie}
  \ing{1}{ui, in dunne ringen}
  \ing{2 grote}{knoflookteentjes, gehalveerd}
  \ing{1 liter}{kippenbouillon}
  \ing{3}{gerookte zongedroogde tomaten op olijfolie, uitgelekt}
  \ingb{kleine handvol tijmtakjes}
  \ingb{kleine handvol basilicumtakjes}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\noindent\textit{Garnering:}
\begin{ingredients}
  \ing{2 el}{olijfolie}
  \ing{200--300 g}{troskerstomaatjes}
  \ingb{kleine basilicumblaadjes}
\end{ingredients}

\blockrule{Bereiding}
\tip{Kun je geen gerookte zongedroogde tomaten vinden, vervang ze dan door 2 half
zongedroogde tomaten en een eetlepel barbecuesaus.}
\begin{steps}
  \item Verwarm de oven voor op 220\,°C (boven- en onderwarmte).
  \item Schenk de olijfolie in een ovenschaal. Schep de tomaten, uienringen en
        knoflook erin en schep goed om. Strooi de tijmtakjes, peper en zout erover.
        Rooster 20--25 minuten
        onder één of twee keer omscheppen, tot alles mooi gekaramelliseerd is. Voeg de
        basilicumtakjes toe in de laatste paar minuten.
  \item Breng de bouillon in een soeppan aan de kook. Schep de geroosterde groenten erin
        (verwijder de tijmtakjes) en voeg de gerookte tomaten toe. Laat 5 minuten
        koken.
  \item Pureer het mengsel in de pan glad met een staafmixer. Door een fijne zeef
        wrijven met de achterkant van een lepel is optioneel, voor een nog gladdere
        soep. Breng op smaak met peper en zout.
  \item Verhit voor de garnering wat olijfolie in een koekenpan. Snijd de tomatentros
        in vier stukken en bak de tomaten 1 minuut aan de tros.
  \item Warm de soep op en schep in verwarmde kommen. Leg een gebakken trostomaat
        erop, sprenkel wat bakvocht eromheen en strooi basilicumblaadjes erover.
\end{steps}
\end{recipe}
