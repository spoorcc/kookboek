\begin{recipe}{Gehaktballetjes}
\kicker[Hoofdgerecht,Vlees,Nederlands,Basisrecept]{Hoofdgerecht · vlees · Nederlands · basisrecept}
\index[register]{Gehaktballetjes}
\index[register]{Gehakt}
\index[register]{Jus}

\meta{40 minuten}

\lettrine{D}{eze} gehaktballetjes zijn snel gemaakt en smakelijk. De basis
is simpel: half-om-half gehakt, wat geweekte beschuit en een ei om alles
bij elkaar te houden. Mosterd of ketjap maakt de bal smeuïger, en van het
bakvocht maak je meteen een jus.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{500 g}{gehakt half-om-half}\index[register]{Gehakt}
  \ing{4}{beschuiten}
  \ing{1}{ei}
  \ing{1 el}{mosterd (of ketjap)}
  \ing{1 el}{maizena, voor de jus}
  \ing{400 ml}{runderbouillon (van blokje of poeder)}\index[register]{Runderbouillon}
  \ingb{scheutje melk, om de beschuit in te weken}
  \ingb{zout}
  \ingb{versgemalen peper}
  \ingb{nootmuskaat}
  \ingb{boter en olijfolie, om te bakken}
\end{ingredients}

\blockrule{Bereiding}
\tip{Wil je zeker weten dat de kruiden goed zitten? Bak een klein balletje
apart en proef. Breng het gehakt zo nodig verder op smaak voor je de rest
bakt, rauw gehakt proef je immers niet.}
\begin{steps}
  \item Los het bouillonblokje (of de bouillonpoeder) op in 400 ml kokend
        water. Houd de bouillon warm tot je hem nodig hebt.
  \item Week de beschuiten kort in een scheutje melk en knijp ze goed uit.
        Meng met het gehakt, het ei, de mosterd (of ketjap), zout,
        versgemalen peper en nootmuskaat, en kneed goed door.
  \item Rol met natte handen gelijke balletjes, bijvoorbeeld met een
        ijsschep, en rol ze niet langer dan nodig: hoe strakker de bal, hoe
        steviger de beet na het garen.
  \item Verhit een klontje boter met een scheutje olijfolie in een
        hapjespan op hoog vuur en schroei de gehaktballetjes rondom dicht.
  \item Zet het vuur laag en voeg de warme bouillon toe tot de balletjes
        voor de helft onder staan. Laat ze twintig minuten zachtjes garen en
        haal ze dan uit de pan. Roer de maizena los in koud water, voeg toe
        aan het bakvocht en laat kort doorkoken tot een jus.
\end{steps}
\end{recipe}
