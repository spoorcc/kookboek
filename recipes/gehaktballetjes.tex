\begin{recipe}{Gehaktballetjes}
\kicker[Hoofdgerecht,Vlees,Nederlands,Basisrecept]{Hoofdgerecht · vlees · Nederlands · basisrecept}
\index[register]{Gehaktballetjes}
\index[register]{Gehakt}
\index[register]{Jus}

\meta{40 minuten}

\lettrine{E}{lke} oma heeft haar eigen gehaktballetjes, met net iets andere
kruiden. De basis is steeds hetzelfde: half-om-half gehakt, een beetje
geweekt brood voor de zachtheid en een ei om alles bij elkaar te houden.
Mosterd of ketjap maakt de bal smeuïger, en van het bakvocht maak je meteen
een jus.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{500 g}{gehakt half-om-half}\index[register]{Gehakt}
  \ing{2 sneetjes}{oud witbrood, zonder korst}
  \ing{1}{ei}
  \ing{1 eetlepel}{mosterd (of ketjap)}
  \ing{1 eetlepel}{maizena, voor de jus}
  \ingb{scheutje melk, om het brood in te weken}
  \ingb{zout, peper en nootmuskaat}
  \ingb{boter en olijfolie, om te bakken}
  \ingb{warm water of bouillon}
\end{ingredients}

\blockrule{Bereiding}
\tip{Wil je zeker weten dat de kruiden goed zitten? Bak een klein balletje
apart en proef. Breng het gehakt zo nodig verder op smaak voor je de rest
bakt, rauw gehakt proef je immers niet.}
\begin{steps}
  \item Week het brood een paar minuten in een scheutje melk en knijp het
        goed uit.
  \item Meng het gehakt met het brood, het ei, de mosterd (of ketjap), zout,
        peper en nootmuskaat, en kneed goed door.
  \item Rol met natte handen balletjes van gelijke grootte, bijvoorbeeld met
        een ijsschep.
  \item Verhit een klontje boter met een scheutje olijfolie in een
        (gietijzeren) braadpan op hoog vuur en schroei de gehaktballetjes
        rondom dicht.
  \item Zet het vuur laag, voeg warm water of bouillon toe tot de balletjes
        voor de helft onder staan en laat ze ongeveer twintig minuten
        zachtjes garen.
  \item Haal de gehaktballetjes uit de pan. Roer de maizena los in een
        scheutje koud water, voeg het toe aan het bakvocht en laat kort
        doorkoken tot een jus.
\end{steps}
\end{recipe}
