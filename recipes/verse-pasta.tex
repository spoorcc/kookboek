\begin{recipe}{Verse pasta}
\label{recipe:verse-pasta}
\kicker[Hoofdgerecht,Italiaans,Basisrecept]{Hoofdgerecht · Italiaans · basisrecept}
\index[register]{Verse pasta}
\index[register]{Bloem}
\index[register]{Eidooiers}

\meta{45 minuten \dvd Pastamachine}

\lettrine{M}{et} maar twee ingrediënten maak je een soepel, goudgeel pastadeeg
dat zachter en rijker smaakt dan pasta uit de supermarkt. Verse pasta doet het
vooral goed met een simpele boter- of roomsaus; een stevige, klonterige saus
past beter bij gedroogde pasta.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{4}{eidooiers}\index[register]{Eidooiers}
  \ing{200 g}{bloem tipo 00}\index[register]{Bloem}
  \ingb{extra bloem of stuifbloem, om te bestuiven}
\end{ingredients}

\blockrule{Bereiding}
\tip{Geen pastamachine? Rol het deeg met een deegroller flinterdun uit. Het
kost meer moeite, maar het lukt prima.}
\begin{steps}
  \item Doe de bloem op een schone werkbank, maak in het midden een kuiltje
        en doe de eidooiers erin.
  \item Klop de dooiers met een vork los en meng ze geleidelijk door de bloem
        tot er een grof deeg ontstaat.
  \item Kneed het deeg 8–10 minuten stevig door tot het glad en elastisch
        aanvoelt, als klei, niet als korrels. Te droog? Kneed er een paar
        druppels water door. Te nat? Een lepel bloem lost het op. Druk met
        een vinger in het deeg: veert het terug, dan is het elastisch genoeg.
  \item Wikkel het deeg in vershoudfolie en laat 20 minuten rusten op
        kamertemperatuur, zodat de gluten ontspannen en het deeg makkelijker
        uitrolt.
  \item Verdeel het deeg in stukken en rol elk stuk met de pastamachine steeds
        dunner uit, met af en toe wat stuifbloem zodat het niet plakt.
  \item Snijd de uitgerolde lappen tot de gewenste pastavorm.
  \item Kook de pasta 3 minuten in ruim gezouten water, verse pasta is sneller
        gaar dan gedroogde. Giet direct af en meng meteen door de saus naar keuze.
\end{steps}
\end{recipe}
