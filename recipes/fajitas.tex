\begin{recipe}{Fajita's}
\kicker{Hoofdgerecht · met kip}
\index[register]{Fajita's}
\index[register]{Kip}
\index[register]{Wraps}
\index[register]{Mais}

\meta{30 minuten \dvd Voor 5 personen \dvd Mexicaans}

\lettrine{I}{edereen} vult zijn eigen wrap: een gerecht waar je bij tafel zelf
de baas bent. De kip gaat gemarineerd in de pan, de groenten komen rauw erbij,
en dan is het vouwen maar.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1}{kipfilet, in blokjes}\index[register]{Kipfilet}
  \ing{1 tl}{komijn}
  \ing{1 tl}{koriander, gemalen}
  \ing{1 tl}{paprikapoeder}
  \ing{1 tl}{knoflookpoeder}
  \ing{1 el}{olijfolie}
  \ing{—}{zout, naar smaak}
  \ing{5}{volkoren wraps}
  \ing{1}{paprika, in reepjes}\index[register]{Paprika}
  \ing{1}{komkommer, in blokjes}\index[register]{Komkommer}
  \ing{1 handje}{cherrytomaatjes, gehalveerd}\index[register]{Tomaten}
  \ing{½ krop}{sla, grof gesneden}
  \ing{1 klein blikje}{mais (± 150 g)}\index[register]{Mais}
  \ing{75 g}{geraspte kaas}\index[register]{Kaas}
  \ing{1 bakje}{crème fraîche of zure room}
\end{ingredients}
\ingb{sambal, naar smaak}
\ingb{avocado of lente-ui, wat je in huis hebt}

\blockrule{Bereiding}
\tip{Bak de zesde wrap iets langer en harder in de droge pan — hij wordt krokant en is lekker om erbij te knabbelen. Losse restjes groente eten de kinderen graag op als slaatje.}
\ingredientsketch{kipblokjes}
\begin{steps}
  \item Snijd de kipfilet in blokjes. Meng ze met komijn, koriander, paprikapoeder,
        knoflookpoeder, zout en olijfolie en laat even intrekken.
  \item Snijd de paprika in reepjes, de komkommer in blokjes en de tomaatjes doormidden.
        Snijd de sla grof. Laat de mais uitlekken en rasp de kaas.
  \item Bak de kip op middelhoog vuur in ± 8 minuten gaar en bruin.
  \item Warm de wraps in een droge koekenpan. Leg ze op elkaar en draai ze om zodra
        ze beginnen te bobbelen — alle vijf passen tegelijk.
  \item Bestrijk elke wrap met crème fraîche of zure room. Wie wil doet er een
        lepeltje sambal bij.
  \item Leg de gebakken kip in een streep dwars over het midden, over ongeveer
        driekwart van de breedte. Strooi de geraspte kaas erover, dan de groenten.
  \item Vouw de wrap op: klap het smalle kwart zonder vulling eerst recht omhoog,
        vouw dan de twee lange zijkanten erover. Klaar.
\end{steps}
\end{recipe}
