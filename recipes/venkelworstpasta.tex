\begin{recipe}{Pasta met venkelworst \& kool}
\kicker[Hoofdgerecht,Vlees,Italiaans]{Hoofdgerecht · vlees · Italiaans}
\index[register]{Pasta met venkelworst en kool}

\meta{40 minuten}

\lettrine{S}{tevige} pasta met varkensworst, geurig venkelzaad en zoete
spitskool. Rustiek comfortfood uit één pan, een vaste favoriet bij ons thuis.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{350 g}{tagliatelle (of pasta naar keuze)}\index[register]{Tagliatelle}
  \ing{200 g}{varkensworst}\index[register]{Varkensworst}
  \ing{300 g}{spitskool, in reepjes}\index[register]{Spitskool}
  \ing{1}{ui, gesnipperd}\index[register]{Ui}
  \ing{250 ml}{groentebouillon}
  \ing{2 el}{extra vierge olijfolie}
  \ing{4 el}{Parmigiano Reggiano, geraspt}\index[register]{Parmigiano Reggiano}
  \ing{1 tl}{venkelzaad}\index[register]{Venkelzaad}
  \ing{1 el}{verse kruiden (peterselie, tijm, marjolein)}
  \ingb{zout \& versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Kook de pasta in het kookwater van de kool. Dat geeft extra smaak en bindt
de saus mooi.}
\begin{steps}
  \item Was en snijd de spitskool in reepjes. Verwijder het vel van de worst
        en snijd in stukjes; snipper de ui en hak de kruiden fijn.
  \item Bak de worststukjes met wat olijfolie en het venkelzaad in een grote hapjespan.
        Schep ze eruit en zet apart.
  \item Kook de kool 3–5 minuten in gezouten water en giet af, maar bewaar het
        kookwater voor de pasta.
  \item Fruit de ui in dezelfde pan (extra olijfolie indien nodig) en voeg dan de
        kool toe.
  \item Schenk de groentebouillon erbij en laat alles 10 minuten zachtjes stoven.
  \item Kook ondertussen de pasta beetgaar in het bewaarde kookwater van de kool
        en giet af.
  \item Doe de worst terug bij de kool en breng op smaak met zout en peper. Meng
        de pasta en verse kruiden erdoor, bestrooi met Parmigiano en serveer
        direct.
\end{steps}
\end{recipe}
