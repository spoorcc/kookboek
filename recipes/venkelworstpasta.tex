\begin{recipe}{Pasta met venkelworst en kool}
\kicker[Hoofdgerecht,Vlees,Italiaans]{Hoofdgerecht · vlees · Italiaans}
\index[register]{Pasta met venkelworst en kool}

\meta{45 minuten}

\lettrine{S}{tevige} pasta met varkensworst, geurig venkelzaad en zoete
spitskool. Rustiek comfortfood uit één pan, een vaste favoriet bij ons thuis.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{200 g}{varkensworst}\index[register]{Varkensworst}
  \ing{300 g}{spitskool, in reepjes}\index[register]{Spitskool}
  \ing{350 g}{tagliatelle (of pasta naar keuze)}\index[register]{Tagliatelle}
  \ing{1}{ui, gesnipperd}\index[register]{Ui}
  \ing{250 ml}{groentebouillon}
  \ing{2 el}{extra vergine olijfolie}
  \ing{4 el}{Parmigiano Reggiano, geraspt}\index[register]{Parmezaanse kaas}
  \ing{1 tl}{venkelzaad}\index[register]{Venkelzaad}
  \ingb{scheutje witte wijn}
  \ingb{zout \& versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Als je echte Italiaanse worsten kan krijgen met venkel is het helemaal lekker.}
\begin{steps}
  \item Was en snijd de spitskool in reepjes. Verwijder het vel van de worst
        en snijd in stukjes; snipper de ui. Kneus het venkelzaad met de platte
        kant van een mes of in een vijzel: gekneusd zaad geeft veel meer smaak
        af dan heel zaad.
  \item Bak de worststukjes met wat olijfolie en het gekneusde venkelzaad 5 minuten in een
        grote hapjespan tot ze rondom bruin zijn. Schep ze eruit en zet apart.
  \item Fruit de ui in dezelfde pan (extra olijfolie indien nodig) en voeg dan de
        kool toe.
  \item Blus af met een scheutje witte wijn en laat het grotendeels inkoken: de
        kool en worst zijn zoet en rijk; het zuur maakt dat minder zwaar. Schenk
        dan de groentebouillon erbij en laat alles 10 minuten zachtjes stoven.
  \item Kook ondertussen de pasta beetgaar in ruim gezouten water. Bewaar een
        kop kookvocht en giet daarna af.
  \item Doe de worst terug bij de kool en breng op smaak met zout en peper. Schep
        de pasta met een scheut kookvocht door de saus in de pan en laat nog even
        meegaren zodat de saus bindt. Bestrooi met Parmigiano en serveer direct.
\end{steps}
\end{recipe}
