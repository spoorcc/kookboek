\begin{recipe}{Ovenfriet}
\label{recipe:ovenfriet}
\kicker[Bijgerecht,Vegan,Nederlands]{Bijgerecht · vegan · Nederlands}
\index[register]{Ovenfriet}
\index[register]{Aardappelen}

\meta{45 minuten \dvd Oven 220\,°C}

\lettrine{K}{ruimige} aardappelen met de schil eraan, voorgekookt en dan
afgebakken in de oven met olijfolie, tijm en rozemarijn. Het voorkoken
zorgt voor een zachte binnenkant, de oven voor een krokante buitenkant.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{600 g}{kruimige aardappelen, met schil}\index[register]{Aardappelen}
  \ingb{olijfolie}
  \ingb{zout en peper}
  \ingb{takjes tijm}
  \ingb{takjes rozemarijn}
\end{ingredients}

\blockrule{Bereiding}
\tip{Schud de braadslee halverwege even door, dan kleuren de frieten
gelijkmatiger.}
\begin{steps}
  \item Verwarm de oven voor op 220\,°C (boven- en onderwarmte).
  \item Snijd de aardappelen met schil en al in frietvorm.
  \item Kook de frieten een paar minuten voor in ruim water, tot ze nog
        een beetje hard zijn.
  \item Giet af en spoel de frieten af onder koud water om het zetmeel
        eraf te spoelen. Dep ze droog en schud ze dan stevig om in het
        vergiet tot de randen wat opruwen: dat ruwe oppervlak wordt in
        de oven de krokante korst.
  \item Leg de frieten in een braadslee, besprenkel met olijfolie en
        breng op smaak met peper. Schep goed door elkaar.
  \item Bak 15 minuten, voeg dan de tijm en rozemarijn toe en bak nog 5
        tot 10 minuten tot de frieten goudbruin zijn: op 220 graden
        verbranden kruiden na 25 minuten en worden bitter.
  \item Bestrooi de frieten met zout zodra ze uit de oven komen, nog heet:
        vooraf gezouten friet trekt vocht aan en wordt minder krokant.
\end{steps}
\end{recipe}
