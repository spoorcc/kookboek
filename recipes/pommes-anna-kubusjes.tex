\begin{recipe}{Pommes Anna Kubusjes}
\kicker[Bijgerecht,Vegetarisch,Frans]{Bijgerecht · vegetarisch · Frans}
\index[register]{Pommes Anna Kubusjes}
\index[register]{Aardappelen}

\meta{40 minuten}

\lettrine{P}{ommes Anna is een klassiek Frans gerecht van dunne aardappelplakjes.} Deze kubusjes-versie geeft dezelfde boterige smaak maar met veel meer krokante randjes. Lekker als bijgerecht bij vlees, vis of ei.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 kg}{vastkokende aardappelen}
  \ing{100 g}{boter (gesmolten)}
  \ing{2}{teentjes knoflook (geperst)}
  \ing{1 tl}{gedroogde rozemarijn}
  \ingb{zout}
  \ingb{versgemalen zwarte peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Goed droog deppen na spoelen is essentieel voor maximale krokantheit.}
\begin{steps}
  \item Schil de aardappelen en snijd ze met een mandoline (kaasschaaf of heel scherp mes) in dunne plakjes van ongeveer 2 mm dik. Spoel kort onder koud water om overtollig zetmeel weg te halen en dep ze heel goed droog met een schone theedoek.
  \item Meng de plakjes in een grote kom met de gesmolten boter, knoflook, zout, peper en eventuele kruiden. Zorg dat elk plakje goed bedekt is.
  \item Spreid de plakjes uit op een bakplaat met bakpapier in losse, licht overlappende laagjes (niet hoger dan 3-4 laagjes). Of maak kleine stapeltjes van ± 4-5 cm breed en snijd ze na een korte voorbakbeurt in kubusjes van ± 2-3 cm. Alternatief: bak de laagjes eerst kort in een hete koekenpan met extra boter zodat ze aan elkaar plakken, snijd dan in kubussen.
  \item Verwarm de oven voor op 225 °C (hete lucht). Bak 25-35 minuten tot ze goudbruin en krokant zijn. Halverwege omscheppen voor gelijkmatige kleur. Voor extra krokant: zet in de laatste 5-8 minuten de grillstand aan of verhoog de temperatuur naar 230-235 °C.
  \item Serveer meteen, eventueel met wat zeezout erover.
\end{steps}
\end{recipe}
