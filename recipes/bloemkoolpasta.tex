\begin{recipe}{Bloemkoolpasta}
\kicker[Hoofdgerecht,Vlees,Doordeweeks]{Hoofdgerecht · vlees · doordeweeks}
\index[register]{Bloemkoolpasta}

\meta{25 minuten \dvd Snel \dvd Op gevoel}

\lettrine{R}{omige} pasta met knapperige spekjes, zoete erwtjes en licht
beetgare bloemkool. Snel gemaakt en fijn voor doordeweeks. Je kan erwtjes 
en bloemkool al voorbereiden, zodat je het gerecht nog sneller op tafel zet.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1}{bloemkool, in roosjes}\index[register]{Bloemkool}
  \ing{250 g}{spekblokjes}\index[register]{Spekjes}
  \ing{350 g}{pasta (zoals penne of fusilli)}\index[register]{Pasta}
  \ing{1}{ui}\index[register]{Ui}
  \ing{1}{knoflookteentje}\index[register]{Knoflook}
  \ing{250 ml}{kookroom}\index[register]{Kookroom}
  \ing{200 g}{diepvrieserwtjes}\index[register]{Erwtjes}
  \ing{50 g}{parmezaan of pecorino}\index[register]{Parmezaanse kaas}
  \ingb{zout en versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Bewaar altijd een kopje pastawater. Die kan je gebruiken om de saus te binden. }
\begin{steps}
  \item Snipper de ui en knoflook, en snijd de bloemkool in kleine roosjes: zo
        garen ze sneller en mengen ze beter door de pasta.
  \item Breng een grote pan water aan de kook, voor de bloemkool. 
        Kook ook water voor het ontdooien in de volgende stap.
  \item Doe de diepvrieserwtjes in een ruime schaal en overgiet met kokend water tot
        ze net onder water staan. Hiermee ontdooien ze.
  \item Blancheer de bloemkool 2–3 minuten in het kokende water tot hij net
        zacht is maar nog een beetje stevig. Schep eruit
        met een schuimspaan en laat goed uitlekken in een vergiet.
  \item Kook de pasta in pan van bloemkool met nieuw water met wat zout,
        1 minuut korter dan de verpakking aangeeft; in de saus wordt hij 
        nog wat zachter. Bewaar voor het afgieten een kopje pastawater.
  \item Verhit ondertussen een hapjespan en bak de spekjes
        knapperig op laag vuur. 
  \item Voeg de ui toe zodra de spekjes bijna bruin zijn en bak
        mee tot de ui glazig is. 
  \item Voeg de knoflook en de bloemkool toe, zet het vuur hoger en
        roerbak nog 1–2 minuten mee.
  \item Voeg de ontdooide erwtjes en de kookroom toe, zet het vuur weer laag en
        doe geen deksel op de pan, saus mag wat inkoken; laat dit 2–3 minuten 
        sudderen tot saus wat dikker is geworden.
  \item Voeg de afgegoten pasta toe aan de saus. Rasp de Parmezaan of
        pecorino erdoor, breng op smaak met zout en versgemalen peper, en
        maak de saus losser met een scheutje pastawater als hij te dik is.
\end{steps}

\end{recipe}
