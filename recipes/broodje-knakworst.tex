\begin{recipe}{Broodje knakworst}
\kicker[Hoofdgerecht,Vlees,Lunch]{Hoofdgerecht · vlees · lunch}
\index[register]{Broodje knakworst}
\index[register]{Knakworst}
\index[register]{Hotdog}

\meta{10 minuten}

\lettrine{E}{en} broodje knakworst staat bij ons in het weekend vaak in
plaats van een avondmaaltijd, na een lunch die al laat was. Snel gemaakt en
iedereen bouwt zijn eigen broodje met wat hij lekker vindt.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{10}{knakworsten (blikje)}\index[register]{Knakworst}
  \ing{5}{witte puntjes}
  \ing{5}{blaadjes sla}
  \ingb{ketchup}
  \ingb{mosterd}
  \ingb{gebakken uitjes}
\end{ingredients}

\blockrule{Bereiding}
\tip{Warm de knakworsten op in het kookvocht uit het blikje, dan blijven ze
sappig in plaats van taai.}
\begin{steps}
  \item Warm de knakworsten op in het kookvocht uit het blikje, of kort in
        een pannetje water.
  \item Snijd de puntjes open en leg in elk broodje een blaadje sla.
  \item Leg twee knakworsten in elk broodje.
  \item Laat iedereen zelf zijn broodje afmaken met ketchup, mosterd en
        gebakken uitjes.
\end{steps}
\end{recipe}
