\begin{recipe}{Jambalaya}
\kicker{Hoofdgerecht · met vlees}
\index[register]{Jambalaya}

\meta{Voor 4 personen \dvd Eén pan \dvd Cajun}

\lettrine{V}{ol} en kruidig eenpansgerecht met chorizo, kip en een stevige
cajunkruidenmix. Was de rijst tot het water helder is en laat alles samen garen
tot een rokerig, pittig bord vol smaak.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{chorizo}\index[register]{Chorizo}
  \ing{250 g}{kipdijfilet}\index[register]{Kipdijfilet}
  \ing{4}{tenen knoflook}\index[register]{Knoflook}
  \ing{1}{groene paprika}
  \ing{1}{witte ui}\index[register]{Ui}
  \ing{3}{stengels bleekselderij}\index[register]{Bleekselderij}
  \ing{3}{takjes tijm}
  \ing{400 g}{rijst, gewassen}\index[register]{Rijst}
  \ing{2 el}{tomatenpuree}
  \ing{2}{rijpe tomaten}\index[register]{Tomaten}
  \ing{250 ml}{kippenbouillon}
  \ing{1/2 bos}{lente-ui}\index[register]{Lente-ui}
  \ing{2–3 el}{cajunkruiden (zie hieronder)}\index[register]{Cajunkruiden}
\end{ingredients}

\blockrule{Bereiding}
{\itshape\small\color{inkfaint}Onze eigen werkwijze — schrijf 'm hier met de hand op:}
\writelines{8}

{\footnotesize\textsc{Cajunkruiden} {\color{inkfaint}(of kant-en-klaar uit de supermarkt):}
1\,el zout \dvd 2\,el zwarte peper \dvd 1\,el witte peper \dvd 1{,}5\,el paprikapoeder
\dvd 1\,el knoflookpoeder \dvd 1\,el uienpoeder \dvd 1\,el cayennepeper \dvd 1{,}5\,tl komijn.}
\end{recipe}
