\begin{recipe}{Jambalaya}
\kicker[Hoofdgerecht,Vlees,Cajun,Pittig]{Hoofdgerecht · vlees · Cajun · pittig}
\index[register]{Jambalaya}
\index[register]{Tomatenpuree}

\meta{45 minuten \dvd Eén pan}

\lettrine{J}{ambalaya} is een traditioneel Cajun gerecht uit New Orleans, de stad
waar ook de jazz vandaan komt. Denk aan paella, maar dan zonder vis: chorizo,
kip en een flinke cajunkruidenmix, samen gegaard tot een rokerig, stevig bord.

\blockrule{Ingrediënten}

\noindent\textit{Gerecht}
\begin{ingredients}
  \ing{250 g}{chorizo}\index[register]{Chorizo}
  \ing{250 g}{kipdijfilet}\index[register]{Kip}
  \ing{180 g}{langkorrelige rijst}\index[register]{Rijst}
  \ing{1}{groene paprika}
  \ing{4}{knoflookteentjes, fijngehakt}\index[register]{Knoflook}
  \ing{1}{ui}\index[register]{Ui}
  \ing{3}{stengels bleekselderij}\index[register]{Bleekselderij}
  \ing{2 el}{tomatenpuree}
  \ing{2}{rijpe tomaten, gewassen en in blokjes}\index[register]{Tomaat}
  \ing{225 ml}{kippenbouillon}
  \ing{1/2 bos}{lente-ui}\index[register]{Lente-ui}
  \ing{3}{takjes tijm}
  \ing{1}{handje verse peterselie}\index[register]{Peterselie}
  \ing{1–2 el}{cajunkruiden}\index[register]{Cajunkruiden}
\end{ingredients}

\medskip
\noindent\textit{Cajunkruiden {\normalfont\color{inkfaint}(of kant-en-klaar uit de supermarkt)}}
\begin{ingredients}
  \ing{1 el}{zout}
  \ing{2 el}{zwarte peper}
  \ing{1 el}{witte peper}
  \ing{1{,}5 el}{paprikapoeder}
  \ing{1 el}{knoflookpoeder}
  \ing{1 el}{uienpoeder}
  \ing{1 el}{cayennepeper}
  \ing{1{,}5 tl}{komijn}
\end{ingredients}

\blockrule{Bereiding}
\tip{Begin met 1 eetlepel cajunkruiden en proef voor je meer toevoegt: de mix heeft al aardig wat pit van de cayennepeper.}
\begin{steps}
  \item Meng zout, zwarte peper, witte peper, paprikapoeder, knoflookpoeder,
        uienpoeder, cayennepeper en komijn tot de cajunkruiden (of sla deze
        stap over bij kant-en-klare).
  \item Snijd de chorizo in blokjes en bak ze op middelhoog vuur droog in de
        hapjespan tot licht krokant, het vet komt vanzelf vrij. Schep uit de
        pan en zet apart.
  \item Bak de kipdijfilet op hoog vuur in het chorizovet tot mooi bruin
        (gaar door en door hoeft nog niet) en schep bij de chorizo.
  \item Was de paprika en bleekselderij en snijd ze samen met de ui zo fijn
        mogelijk, de holy trinity. Fruit ze op laag vuur ± 10 minuten tot
        zacht en glazig.
  \item Was ondertussen de rijst tot het water helder is en rooster hem op
        laag vuur droog in een koekenpan tot de korrels licht goudgeel kleuren.
  \item Voeg knoflook en tijm toe, bak 1 minuut mee en zet het vuur omhoog.
  \item Roer tomatenpuree en cajunkruiden erdoor, bak 1–2 minuten om te ontzuren.
  \item Voeg de tomaten, geroosterde rijst en bouillon toe, leg chorizo en
        kip terug en roer alles goed door.
  \item Dek af en laat op laag vuur garen tot de rijst de bouillon heeft
        opgenomen, ± 20 minuten. Lijkt het te droog voordat de rijst helemaal
        gaar is, voeg dan een scheutje extra bouillon of water toe.
  \item Roer vlak voor het serveren de lente-ui en peterselie erdoor zodat ze
        fris en groen blijven.
\end{steps}
