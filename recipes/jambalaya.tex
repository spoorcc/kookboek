\begin{recipe}{Jambalaya}
\kicker[Hoofdgerecht,Vlees,Cajun,Pittig]{Hoofdgerecht · met vlees · Cajun · pittig}
\index[register]{Jambalaya}

\meta{45 minuten \dvd Voor 4 personen \dvd Eén pan}

\lettrine{J}{ambalaya} is een traditioneel Cajun gerecht uit New Orleans, de stad
waar ook de jazz vandaan komt. Denk aan paella, maar dan zonder vis en met meer
comfort food-karakter: chorizo, kip en een flinke cajunkruidenmix in één pan.
Was de rijst tot het water helder is en laat alles samen garen tot een rokerig,
stevig bord.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{chorizo}\index[register]{Chorizo}
  \ing{250 g}{kipdijfilet}\index[register]{Kipdijfilet}
  \ing{4}{knoflookteentjes}\index[register]{Knoflook}
  \ing{1}{groene paprika}
  \ing{1}{ui}\index[register]{Ui}
  \ing{3}{stengels bleekselderij}\index[register]{Bleekselderij}
  \ing{400 g}{risottorijst, gewassen}\index[register]{Rijst}
  \ing{2 el}{tomatenpuree}
  \ing{2}{rijpe tomaten}\index[register]{Tomaten}
  \ing{250 ml}{kippenbouillon}
  \ing{1/2 bos}{lente-ui}\index[register]{Lente-ui}
  \ing{3}{takjes tijm}
  \ing{1}{handje verse peterselie}\index[register]{Peterselie}
  \ing{1–2 el}{cajunkruiden (zie hieronder)}\index[register]{Cajunkruiden}
\end{ingredients}

\blockrule{Bereiding}
\tip{Begin met 1 eetlepel cajunkruiden en proef voor je meer toevoegt. De mix heeft al behoorlijk wat pit van de cayennepeper.}
\begin{steps}
  \item Snijd de chorizo in blokjes en bak ze in een droge koekenpan op middelhoog vuur.
        Het vet komt vanzelf vrij. Bak tot licht krokant, schep dan uit de pan en
        leg apart.
  \item Bak de kipdijfilet in het achtergebleven chorizovet. Een mooie bruining
        is hier het doel, gaar door en door hoeft niet. Schep de kip apart bij de
        chorizo.
  \item Snijd ui, groene paprika en bleekselderij zo fijn mogelijk. Dit is de
        holy trinity, de cajun soffritto. Zet het vuur laag en fruit ze in
        dezelfde pan, rustig, ± 10 minuten tot alles zacht en glazig is.
  \item Rooster ondertussen de gewassen rijst in een droge koekenpan tot de
        korrels licht goudgeel kleuren.
  \item Voeg de knoflook en tijm toe aan de holy trinity en bak nog 1 minuut mee.
        Zet dan het vuur omhoog.
  \item Roer de tomatenpuree erdoor en bak 1–2 minuten om te ontzuren.
  \item Voeg de tomaten, geroosterde rijst, cajunkruiden en bouillon toe. Leg de
        chorizo en kip terug in de pan en roer alles goed door.
  \item Dek af en laat op laag vuur garen tot de rijst de bouillon heeft
        opgenomen, ± 20 minuten.
  \item Roer vlak voor het serveren de lente-ui en peterselie erdoor zodat ze
        fris en groen blijven.
\end{steps}

{\footnotesize\textsc{Cajunkruiden} {\color{inkfaint}(of kant-en-klaar uit de supermarkt):}
1\,el zout \dvd 2\,el zwarte peper \dvd 1\,el witte peper \dvd 1{,}5\,el paprikapoeder
\dvd 1\,el knoflookpoeder \dvd 1\,el uienpoeder \dvd 1\,el cayennepeper \dvd 1{,}5\,tl komijn.}
\end{recipe}
