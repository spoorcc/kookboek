\begin{recipe}{Broodje hamburger}
\kicker{Hoofdgerecht · kinderfavoriet}
\index[register]{Broodje hamburger}
\index[register]{Hamburger}
\index[register]{Ui}

\meta{25 minuten}

\lettrine{B}{roodje} hamburger is bij ons een kinderfavoriet. De winst zit in kleine
dingen: langzaam gekaramelliseerde uien, een geroosterd bolletje en een plakje kaas
dat op de hete burger smelt. De rest laten we door de kinderen zelf samenstellen.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{5}{hamburgerbolletjes}
  \ing{5}{hamburgers (black angus)}\index[register]{Hamburger}
  \ing{1}{ui}\index[register]{Ui}
  \ingb{sla}
  \ing{1}{tomaat}
  \ing{3}{augurken, in plakjes}
  \ing{1}{komkommer, in plakjes}
  \ing{1}{paprika, in reepjes}
  \ing{5}{plakjes kaas}
  \ingb{mayonaise}
  \ingb{tomatenketchup}
  \ingb{burgersaus}
  \ingb{mosterd}
\end{ingredients}

\blockrule{Bereiding}
\tip{Snij de tomaat horizontaal door voor gelijkmatige, ronde plakken. Een vleestomaat
zoals coeur de bœuf geeft mooie grote plakken.}
\begin{steps}
  \item Snij de ui in dunne ringen. Verhit een klontje boter met een scheutje water in
        een pan op laag vuur en laat de uienringen langzaam karamelliseren.
  \item Verdeel de sla in blaadjes en snij de tomaat in plakken.
  \item Verhit bakboter in een tweede pan en bak de hamburgers volgens de verpakking.
  \item Rooster de bolletjes ondertussen onder de gril in de oven.
  \item Leg vlak voor de burgers klaar zijn een plakje kaas op elke hamburger zodat
        die kan smelten.
  \item Laat iedereen zijn eigen broodje opbouwen met wat hij of zij wil.
\end{steps}
\end{recipe}
