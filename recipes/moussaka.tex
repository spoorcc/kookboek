\begin{recipe}{Moussaka}
\kicker[Hoofdgerecht,Vlees,Grieks]{Hoofdgerecht · vlees · Grieks}
\index[register]{Moussaka}
\index[register]{Aubergine}
\index[register]{Rundergehakt}
\index[register]{Bechamelsaus}

\meta{2 uur 30 \dvd Voor 4 personen \dvd Oven 180\,°C}

\lettrine{M}{oussaka} is een Grieks gerecht met laagjes aardappel, gekruid gehakt
en aubergine, afgesloten met een romige bechamelsaus. Er zijn wat stappen, maar
niets ervan is ingewikkeld.

\blockrule{Ingrediënten}

\noindent\textit{Gehaktsaus}
\begin{ingredients}
  \ing{500 g}{rundergehakt (of kalfs- of lamsgehakt)}\index[register]{Rundergehakt}
  \ing{200 g}{gepelde tomaten (½ blik)}
  \ing{30 ml}{droge witte wijn}
  \ing{2 tl}{tomatenpuree}
  \ing{2}{laurierblaadjes}
  \ing{1 tl}{gemalen piment (of 2 pimentkorrels)}
  \ing{2}{knoflookteentjes, geperst}
  \ing{2 kleine}{uien, zeer fijn gesneden}
  \ing{2 tl}{kaneel}
  \ing{3 tl}{oregano}
\end{ingredients}

\medskip
\noindent\textit{Vulling}
\begin{ingredients}
  \ing{8 middelgrote}{aardappelen}\index[register]{Aardappelen}
  \ing{2}{aubergines}\index[register]{Aubergine}
  \ingb{olijfolie}
  \ingb{zout en versgemalen peper}
\end{ingredients}

\medskip
\noindent\textit{Bechamelsaus}
\begin{ingredients}
  \ing{900 ml}{volle melk}
  \ing{2}{eidooiers}
  \ing{50 g}{roomboter}
  \ing{50 g}{bloem}
  \ing{50 g}{Parmezaanse kaas of Grana Padano, geraspt}
  \ing{1 snuf}{nootmuskaat (vers geraspt)}
  \ingb{extra geraspte kaas, voor de bovenkant}
\end{ingredients}

\blockrule{Bereiding}
\tip{Aubergines nemen veel olijfolie op. Gebruik niet te veel en laat de plakken goed
uitlekken op keukenpapier, anders wordt de moussaka te vet.}
\begin{steps}
  \item Verhit een scheutje olijfolie in een pan en fruit de ui met de knoflook
        op middelhoog vuur 5--10 minuten tot glazig, niet bruin.
  \item Voeg het gehakt toe en laat het op laag vuur rullen. Giet het mengsel
        eventueel in een zeef om overtollig vet af te gieten, dat maakt de moussaka
        minder vet, en doe het terug in de pan. Voeg de laurierblaadjes toe.
  \item Roer de tomatenpuree erdoor en bak een minuut mee. Blus af met de wijn en
        laat sudderen tot het vocht is opgenomen.
  \item Voeg de gepelde tomaten, oregano, piment en kaneel toe. Breng zachtjes aan
        de kook en laat inkoken tot een gebonden saus. Verwijder de laurierblaadjes
        (en pimentkorrels). Breng op smaak met zout en peper. Zet de saus opzij.
  \item Schil de aardappelen en snijd ze in de lengte in dunne plakken. Snijd de
        aubergines op dezelfde manier.
  \item Verhit ruim olijfolie in een derde koekenpan en bak de aardappelplakken
        in porties rondom goudbruin. Laat uitlekken op keukenpapier.
  \item Bak de aubergineplakken in de pan met een scheutje olijfolie aan beide
        kanten bruin. Voeg tussendoor olijfolie toe als nodig. Laat ook deze uitlekken op
        keukenpapier.
  \item Vet een diepe ovenschaal (ca.\ 32\,×\,20\,×\,8 cm) in. Bedek de bodem met
        iets minder dan de helft van de aardappelplakken, als een puzzel zodat er
        een egale laag ontstaat. Verdeel de gehaktsaus erover en druk aan. Leg alle
        aubergineplakken erop en sluit af met de rest van de aardappelen. Druk het
        geheel zachtjes aan.
  \item Verwarm de oven voor op 180\,°C (boven- en onderwarmte), zodat hij op
        temperatuur is tegen de tijd dat de moussaka erin gaat. Smelt de boter
        in een hoge pan. Voeg de bloem toe en laat 2 minuten zachtjes
        borrelen. Verwarm de melk apart (niet laten koken, dat voorkomt een vliesje en
        klontjes) en giet een scheut bij het
        boter-bloemmengsel. Roer glad. Voeg de rest van de melk toe en laat koken
        tot een lobbige saus. Haal van het vuur.
  \item Klop de eidooiers door de saus. Roer de geraspte kaas en nootmuskaat erdoor.
  \item Giet de bechamelsaus over de moussaka en
        rasp wat extra kaas erover. Bak 40--45 minuten tot de bovenkant goudbruin is.
        Laat de moussaka daarna 15 tot 30 minuten buiten de oven rusten voor je
        hem aansnijdt: dan zetten de lagen op en snijd je nette stukken in plaats
        van dat de vulling meteen alle kanten op loopt. In Griekenland eet men
        moussaka toch het liefst lauw.
\end{steps}
\end{recipe}
