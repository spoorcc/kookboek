\begin{recipe}{Moussaka}
\kicker[Hoofdgerecht,Vlees,Grieks]{Hoofdgerecht · vlees · Grieks}
\index[register]{Moussaka}
\index[register]{Aubergine}
\index[register]{Rundergehakt}
\index[register]{Bechamelsaus}
\index[register]{Tomatenpuree}

\meta{150 minuten \dvd Oven 180\,°C}

\lettrine{M}{oussaka} is een Grieks gerecht met laagjes aardappel, gekruid gehakt
en aubergine, afgesloten met een romige bechamelsaus. Er zijn wat stappen, maar
niets ervan is ingewikkeld.

\blockrule{Ingrediënten}

\noindent\textit{Gehaktsaus}
\begin{ingredients}
  \ing{500 g}{rundergehakt (of kalfs- of lamsgehakt)}\index[register]{Rundergehakt}
  \ing{200 g}{gepelde tomaten (½ blik)}
  \ing{30 ml}{droge witte wijn}
  \ing{2 tl}{tomatenpuree}
  \ing{2}{laurierblaadjes}
  \ing{1 tl}{gemalen piment (of 2 pimentkorrels)}
  \ing{2}{knoflookteentjes, geperst}
  \ing{2 kleine}{uien, zeer fijn gesneden}
  \ing{2 tl}{kaneel}
  \ing{3 tl}{oregano}
\end{ingredients}

\medskip
\noindent\textit{Vulling}
\begin{ingredients}
  \ing{8 middelgrote}{aardappelen}\index[register]{Aardappelen}
  \ing{2}{aubergines}\index[register]{Aubergine}
  \ingb{olijfolie}
  \ingb{zout en versgemalen peper}
\end{ingredients}

\medskip
\noindent\textit{Bechamelsaus}
\begin{ingredients}
  \ing{900 ml}{volle melk}
  \ing{2}{eidooiers}
  \ing{50 g}{ongezouten roomboter}
  \ing{50 g}{bloem}
  \ing{50 g}{Parmezaanse kaas of Grana Padano, geraspt}
  \ing{1 snuf}{nootmuskaat (vers geraspt)}
  \ingb{extra geraspte kaas, voor de bovenkant}
\end{ingredients}

\blockrule{Bereiding}
\tip{Serveer de moussaka met een horiatiki (Griekse boerensalade,
p.\,\pageref{recipe:horiatiki}) ernaast: het fris-zure van de salade snijdt
door de romige bechamel en het rijke gehakt heen.}
\begin{steps}
  \item Verwarm de oven voor op 200\,°C (boven- en onderwarmte). Schil de aardappelen
        en snijd ze in de lengte in dunne plakken. Was de aubergines en snijd ze op
        dezelfde manier. Meng een scheutje olijfolie met zout en versgemalen peper
        en smeer de plakken hiermee in voor je ze verdeelt over twee met bakpapier
        beklede bakplaten.
  \item Rooster de plakken 25 tot 30 minuten tot ze goudbruin zijn: aubergine zuigt
        in een pan olie op als een spons, in de oven niet, dus dit scheelt meteen
        ook vet. Maak intussen de gehaktsaus (volgende stappen).
  \item Verhit een scheutje olijfolie in een hapjespan en fruit de ui met de knoflook
        op middelhoog vuur 5--10 minuten tot glazig, niet bruin.
  \item Voeg het gehakt toe, verhoog het vuur naar middelhoog tot hoog en bak
        rondom stevig bruin: dat geeft meer smaak dan grijs gestoomd gehakt.
        Giet overtollig vet af en voeg de laurierblaadjes toe.
  \item Roer de tomatenpuree erdoor en bak een minuut mee. Blus af met de wijn en
        laat sudderen tot het vocht is opgenomen.
  \item Voeg de gepelde tomaten, oregano, piment en kaneel toe. Breng zachtjes aan
        de kook en laat inkoken tot een gebonden saus. Verwijder de laurierblaadjes
        (en pimentkorrels) en breng op smaak met zout en peper.
  \item Vet een diepe ovenschaal (ca.\ 32\,×\,20\,×\,8 cm) in. Bedek de bodem met
        iets minder dan de helft van de aardappelplakken, als een puzzel. Verdeel
        de gehaktsaus erover en druk aan. Leg alle aubergineplakken erop en sluit
        af met de rest van de aardappelen. Druk zachtjes aan.
  \item Verlaag de oven naar 180\,°C (boven- en onderwarmte). Smelt de boter
        in een hoge pan, voeg de bloem toe en laat 2 minuten zachtjes borrelen.
        Verwarm de melk apart (niet laten koken), giet er beetje bij beetje
        bij en roer tot een dikke saus: de bechamel is hier een deksel voor de
        moussaka, dus mag dikker zijn dan gebruikelijk.
  \item Klop de eidooiers, geraspte kaas en nootmuskaat door de saus. Giet de
        bechamelsaus over de moussaka, rasp er wat extra kaas over en bak
        40--45 minuten tot de bovenkant goudbruin is. Laat daarna 15 tot 30
        minuten rusten voor het aansnijden: dan zetten de lagen op.
\end{steps}
\end{recipe}
