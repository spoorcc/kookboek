\begin{recipe}{Chili con carne}
\kicker[Hoofdgerecht,Vlees,Mexicaans,Pittig]{Hoofdgerecht · vlees · Mexicaans · pittig}
\index[register]{Chili con carne}
\index[register]{Rundergehakt}
\index[register]{Kidneybonen}
\index[register]{Rijst}

\meta{45 minuten \dvd Eén pan}

\lettrine{C}{hili} con carne is een stevige stoofpot van gehakt, bonen en
tomaat, op smaak gebracht met een flinke kruidenmix. Alles gaat in één pan en
je serveert het met rijst, zure room en verse kruiden erbovenop.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1}{ui, gesnipperd}
  \ing{2 teentjes}{knoflook, gesnipperd}\index[register]{Knoflook}
  \ing{2}{rode paprika's, in kleine stukjes}\index[register]{Paprika}
  \ing{1 el}{olijfolie}
  \ing{300 g}{rundergehakt}
  \ing{450 g}{kidneybonen, uitgelekt en afgespoeld}
  \ing{70 g}{tomatenpuree}
  \ing{400 g}{gepelde tomaten}
  \ing{150 g}{mais}\index[register]{Mais}
  \ing{200 ml}{zure room}
  \ingb{peper en zout}
  \ingb{peterselie of koriander, fijngehakt}
  \ing{170 g}{lange korrelrijst, om bij te serveren}\index[register]{Rijst}
  \ing{1 tl}{chilipoeder}
  \ing{1 tl}{knoflookpoeder}
  \ing{1 tl}{komijn}
  \ing{2,5 tl}{paprikapoeder}
  \ing{1 tl}{nootmuskaat}
  \ing{1/4 tl}{koriander, gemalen}
  \ing{2 tl}{uienpoeder}
\end{ingredients}

\blockrule{Bereiding}
\begin{steps}
  \item Snipper de ui en de knoflook en snijd de paprika's in kleine stukjes.
        Zo heb je alles bij de hand voor je begint met koken.
  \item Bak de ui en de knoflook in een hapjespan met een beetje olijfolie.
  \item Voeg het gehakt toe en bak dit rul. Voeg de kruidenmix toe en roer
        er goed doorheen.
  \item Bak de paprika een paar minuten mee.
  \item Voeg de tomatenpuree toe en bak deze even kort mee, dat haalt de
        zuurheid eraf. Doe dan de gepelde tomaten erbij en roer alles goed
        door elkaar. Breng op smaak met voldoende peper en zout. Laat dit
        mengsel 10 minuten pruttelen zodat de smaken goed intrekken.
  \item Voeg de afgespoelde bonen en de mais toe en laat nog 10 minuten
        pruttelen. Kook ondertussen de rijst in ongeveer 10 minuten gaar.
  \item Serveer de chili con carne met een flinke schep zure room, wat
        fijngehakte peterselie of koriander en de rijst.
\end{steps}
\end{recipe}
