\begin{recipe}{Rode Rijst}
\kicker[Bijgerecht,Vegan]{Bijgerecht · vegan}
\index[register]{Rode rijst}
\index[register]{Rijst}\index[register]{Tomaat}\index[register]{Knoflook}

\meta{30 minuten \dvd Voor 2–3 personen \dvd Eén pan}

\lettrine{R}{ijst} gegaard in tomatenwater met ui, knoflook en rundsbouillon.
Een simpel gerecht dat zichzelf op laag vuur klaarmaakt terwijl je de rest
van de maaltijd voorbereidt.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{2 el}{olijfolie}
  \ing{1}{ui, fijngesnipperd}\index[register]{Ui}
  \ing{1}{knoflookteentje}\index[register]{Knoflook}
  \ing{1 blik (400\,g)}{gepelde tomaten}\index[register]{Tomaten}
  \ing{1}{rundsbouillonblokje}\index[register]{Bouillon}
  \ing{170 g}{langkorrelige rijst}
\end{ingredients}

\blockrule{Bereiding}
\tip{Roer zo weinig mogelijk. Te veel roeren maakt de rijst papperig.}
\begin{steps}
  \item Snipper de ui fijn en pers de knoflook.
  \item Verhit de olijfolie in een hapjespan met dikke bodem en fruit de ui op
        middelhoog vuur tot glazig, ongeveer 5 minuten.
  \item Voeg de knoflook toe en bak nog 1 minuut mee.
  \item Voeg de tomaten toe. Spoel elk blik na met een blik water en doe
        ook dat water in de pan. Voeg het bouillonblokje toe en roer door.
  \item Breng aan de kook, voeg dan de rijst toe, zet het vuur laag en doe
        de deksel op de pan. Laat minstens 10 minuten garen.
  \item Roer één keer voorzichtig door. Als de rijst te droog lijkt, voeg
        dan een klein scheutje water toe, niet meer dan dat.
  \item Proef of de rijst gaar is en serveer direct.
\end{steps}
\end{recipe}
