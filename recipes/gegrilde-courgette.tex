\begin{recipe}{Gegrilde courgette}
\label{recipe:gegrilde-courgette}
\kicker[Bijgerecht,Vegan,Zomer]{Bijgerecht · vegan · zomer}
\index[register]{Courgette}
\index[register]{Knoflook}

\meta{15 minuten marineren \dvd Bbq}

\lettrine{D}{it} gerecht proefde ik voor het eerst tijdens een zeezeilrace,
als student, ergens in Griekenland. Simpeler kan bijna niet: courgette,
zout, peper en knoflook, en de bbq doet de rest.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{2}{courgettes, in plakken van 1 cm}\index[register]{Courgette}
  \ing{2}{teentjes knoflook, geperst}\index[register]{Knoflook}
  \ingb{ruim zout en peper}
  \ingb{olijfolie}
\end{ingredients}

\blockrule{Bereiding}
\tip{Dit werkt eigenlijk met bijna elke groente. Puntpaprika en aubergine
zijn net zo lekker van de bbq.}
\begin{steps}
  \item Snijd de courgettes in de lengte in plakken van ongeveer 1 cm dik en 
        leg ze in een ondiepe schaal.
  \item Smeer beide kanten van de plakken ruim in met olijfolie en
        bestrooi ze met zout en peper. Pers de knoflook erover.
  \item Laat 15 minuten marineren.
  \item Gril de plakken op de bbq tot je mooie grillstrepen hebt.
\end{steps}

\end{recipe}
