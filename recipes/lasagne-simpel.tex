\begin{recipe}{Simpele lasagne met gehakt}
\kicker{Hoofdgerecht · doordeweeks}
\index[register]{Simpele lasagne met gehakt}
\index[register]{Gehakt}
\index[register]{Lasagnebladen}
\index[register]{Cherrytomaatjes}
\index[register]{Courgette}
\index[register]{Paprika}

\meta{45 minuten \dvd Voor 5 personen \dvd Italiaans}

\lettrine{E}{en} lasagne zonder gedoe, klaar in drie kwartier en met weinig
losse kruiden. Ui, paprika en courgette zijn de enige groente die je hoeft
te snijden, de kant-en-klare lasagnesaus doet de rest van het werk. Dit is
een recept voor een drukke doordeweekse avond. Zonder gehakt werkt het net
zo goed.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{125 g}{lasagnebladen}\index[register]{Lasagnebladen}
  \ing{300 g}{rundergehakt}\index[register]{Gehakt}
  \ing{1}{ui, gesnipperd}
  \ing{1}{rode paprika, in stukjes}\index[register]{Paprika}
  \ing{1}{courgette, in plakjes}\index[register]{Courgette}
  \ing{250 g}{cherrytomaatjes, gehalveerd}\index[register]{Cherrytomaatjes}
  \ing{500 ml}{halfvolle melk}
  \ing{500 ml}{water}
  \ing{1 zakje}{Honig basis voor lasagnesaus}
  \ing{50 g}{geraspte kaas}
  \ingb{1 eetlepel olijfolie}
\end{ingredients}

\blockrule{Bereiding}
\tip{Gebruik lasagnebladen van eierpasta in plaats van gewone: dat smaakt
een stuk voller. Geen zin in gehakt? Laat het gewoon weg en bak de groente
in de olijfolie: de lasagne wordt er niet minder vol van.}
\begin{steps}
  \item Verwarm de oven voor op 200\,°C (gasovenstand 4). Vet een ovenschaal
        van ongeveer 1500 ml in.
  \item Verhit de olijfolie in een pan. Bak het gehakt rul en bruin. Voeg de
        ui, paprika, courgette en cherrytomaatjes toe en bak dit kort mee.
  \item Voeg het water, de melk en de Honig basis voor lasagnesaus toe. Breng
        alles al roerend aan de kook en laat ongeveer een minuut zachtjes
        doorkoken.
  \item Bouw de lasagne op: begin met een laag saus op de bodem van de
        ovenschaal, bedek met lasagnebladen en herhaal dit tot de saus op
        is. Eindig met een ruime laag saus.
  \item Bestrooi met de kaas. Bak de lasagne in het midden van de oven
        minstens 20 minuten gaar en zet de laatste 5 minuten de grillstand
        aan, zodat de kaas mooi bruin wordt. Laat hem 5 minuten staan
        voordat je hem aansnijdt.
\end{steps}
\end{recipe}
