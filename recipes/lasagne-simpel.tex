\begin{recipe}{Simpele lasagne met gehakt}
\kicker[Hoofdgerecht,Vlees,Italiaans,Doordeweeks]{Hoofdgerecht · vlees · Italiaans · doordeweeks}
\index[register]{Simpele lasagne met gehakt}
\index[register]{Gehakt}
\index[register]{Lasagnebladen}
\index[register]{Courgette}
\index[register]{Paprika}

\meta{45 minuten \dvd Oven 200\,°C}

\lettrine{E}{en} lasagne zonder gedoe en met weinig losse kruiden. Ui, paprika en
courgette zijn de enige groente die je hoeft te snijden. De kant-en-klare
lasagnesaus doet de rest van het werk, precies wat je wil op een drukke
doordeweekse avond.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{300 g}{rundergehakt}\index[register]{Gehakt}
  \ing{125 g}{lasagnebladen}\index[register]{Lasagnebladen}
  \ing{1}{rode paprika, in stukjes}\index[register]{Paprika}
  \ing{1}{courgette, in plakjes}\index[register]{Courgette}
  \ing{1}{ui, gesnipperd}
  \ing{500 ml}{halfvolle melk}
  \ing{500 ml}{water}
  \ing{1 verpakking}{Honig basis voor lasagnesaus}
  \ing{50 g}{geraspte kaas}
  \ing{1 eetlepel}{olijfolie}
\end{ingredients}

\blockrule{Bereiding}
\tip{Gebruik lasagnebladen van eierpasta in plaats van gewone: dat smaakt
een stuk voller.}
\begin{steps}
  \item Verwarm de oven voor op 200\,°C (boven- en onderwarmte). Vet een ovenschaal
        van ongeveer 1500 ml in.
  \item Verhit de olijfolie in een hapjespan. Bak het gehakt op middelhoog vuur en
        roer niet te vaak: laat het een paar minuten met rust liggen zodat het
        écht bruin wordt in plaats van grijs, dat aanbaksel is hier de enige
        smaakmaker die je zelf in de hand hebt. Bak dan de ui mee tot deze glazig
        is. Voeg dan de paprika en courgette toe en bak mee tot ze bruin beginnen
        te kleuren.
  \item Voeg het water, de melk en de Honig basis voor lasagnesaus toe. Breng
        alles al roerend aan de kook en laat ongeveer een minuut zachtjes
        doorkoken. Proef de saus: pakketsaus is vaak aan de zoete kant, een
        scheutje azijn of wat extra peper trekt hem recht.
  \item Bouw de lasagne op: begin met een laag saus op de bodem van de
        ovenschaal, bedek met lasagnebladen en herhaal dit tot de saus op
        is. Eindig met een ruime laag saus.
  \item Bestrooi met de kaas. Bak de lasagne in het midden van de oven
        minstens 20 minuten gaar en zet de laatste 5 minuten de grillstand
        aan, zodat de kaas mooi bruin wordt. Laat hem 10 minuten staan
        voordat je hem aansnijdt: dan blijven de lagen staan bij het
        opscheppen.
\end{steps}
\end{recipe}
