\begin{recipe}{Gnocchi met zongedroogde tomatensaus en spinazie}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans]{Hoofdgerecht · vegetarisch · Italiaans}
\index[register]{Gnocchi met zongedroogde tomatensaus en spinazie}
\index[register]{Gnocchi}
\index[register]{Zongedroogde tomaten}
\index[register]{Spinazie}
\index[register]{Mozzarella}

\meta{30 minuten}

\lettrine{E}{en} romige pastaschotel zonder dat er echt room aan te pas komt:
de saus wordt gemaakt van zongedroogde tomaten, ricotta en een scheutje
kookvocht. Met spinazie en gescheurde mozzarella erdoorheen is het zo klaar,
en in de oven krijgt het net wat meer korstje.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{400 g}{gnocchi}\index[register]{Gnocchi}
  \ing{100 g}{zongedroogde tomaten op olie, goed uitgelekt}\index[register]{Zongedroogde tomaten}
  \ing{1 klein}{uitje, in stukken}\index[register]{Ui}
  \ing{150 g}{ricotta}\index[register]{Ricotta}
  \ing{1 teen}{knoflook}\index[register]{Knoflook}
  \ing{20--30 g}{Parmezaanse kaas (optioneel)}\index[register]{Parmezaanse kaas}
  \ing{1--2 el}{olie uit het potje (niet meer, anders wordt de saus zwaar)}
  \ing{2--4 el}{heet water (of kookvocht van de gnocchi)}
  \ing{200 g}{verse spinazie}\index[register]{Spinazie}
  \ing{1 bol}{mozzarella, in stukken gescheurd}\index[register]{Mozzarella}
  \ingb{versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Schep alles in een ovenschaal, bestrooi met wat extra Parmezaanse kaas
en zet het 10--15 minuten in een voorverwarmde oven op 200\textdegree{}C,
eventueel de laatste 2--3 minuten onder de grill voor een goudbruin korstje.
Rechtstreeks uit de pan is het al romig genoeg, de ovenschotelvariant geeft
gewoon net wat meer gratinsmaak.}
\begin{steps}
  \item Zet een pan water met zout op voor de gnocchi.
  \item Pureer ondertussen het uitje met de uitgelekte zongedroogde
        tomaten, ricotta, knoflook, eventueel de Parmezaanse kaas, 1--2
        eetlepels tomatenolie en het hete water tot een gladde saus.
  \item Bak de puree kort aan in een hapjespan. Voeg de spinazie toe en
        laat die slinken terwijl het water voor de gnocchi kookt.
  \item Kook de verse gnocchi kort mee, dat duurt maar een paar minuten: ze
        zijn klaar zodra ze bovendrijven. Schep ze er met een schuimspaan
        direct uit in de saus.
  \item Schep de gnocchi door de saus en verdeel de gescheurde mozzarella
        erover.
\end{steps}
\end{recipe}
