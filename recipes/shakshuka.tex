\begin{recipe}{Shakshuka}
\kicker[Hoofdgerecht,Vegetarisch,Doordeweeks]{Hoofdgerecht · vegetarisch · doordeweeks}
\index[register]{Shakshuka}

\meta{30 minuten \dvd Voor 5 personen \dvd Eén pan}

\lettrine{E}{ieren} gepocheerd in een pittige tomatensaus met paprika, komijn en
gerookte paprika. Lepel er feta en peterselie overheen en dip er een stuk verse
pita in.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{6}{eieren}\index[register]{Eieren}
  \ing{200 g}{feta, in blokjes}\index[register]{Feta}
  \ing{750 g}{tomaten, gewassen en in blokjes}\index[register]{Tomaat}
  \ing{2 el}{extra vergine olijfolie}
  \ing{1}{ui, in stukken}\index[register]{Ui}
  \ing{1}{rode paprika, in stukken}\index[register]{Paprika}
  \ing{2}{knoflookteentjes, fijngehakt}\index[register]{Knoflook}
  \ing{1 tl}{komijnpoeder}
  \ing{1 tl}{gerookt paprikapoeder}
  \ing{1 tl}{chilivlokken}
  \ing{1}{handje verse peterselie}\index[register]{Peterselie}
  \ingb{zout \& versgemalen peper, naar smaak}
  \ingb{pita, om bij te serveren}
\end{ingredients}

\blockrule{Bereiding}
\tip{Wordt de saus te droog tijdens het pruttelen, voeg dan een scheutje water toe. Zelfgebakken pita erbij? Zie blz.\,\pageref{recipe:pita}.}
\begin{steps}
  \item Was de paprika en snijd hem samen met de ui in stukken. Hak de
        knoflook en de peterselie fijn.
  \item Zet een droge hapjespan op middelhoog vuur en rooster de chilivlokken
        al roerend ± 30 seconden tot ze geurig worden.
  \item Schenk de olijfolie in de pan en laat heet worden.
  \item Bak de ui en paprika op middelhoog vuur ± 5 minuten tot de ui glazig
        wordt.
  \item Voeg de knoflook, gerookt paprikapoeder en komijn toe en bak 1 minuut mee.
  \item Voeg de tomaten met zout en peper toe, zet het vuur laag en laat 15
        minuten pruttelen.
  \item Breek een ei in een pollepel en gebruik de lepel meteen om een kuiltje in
        de saus te maken; laat het ei erin glijden. Herhaal met de overige
        eieren. Deksel erop en pocheer ± 6 minuten. Zo blijft het eigeel mooi
        lopend.
  \item Zet het vuur uit, strooi de feta over de shakshuka en doe de deksel er
        weer op. Laat 3 minuten staan, zo gaart het ei nog even na.
  \item Strooi vlak voor het serveren de peterselie en een beetje peper en zout
        erover.
  \item Pak een stuk pita en dip erin.
\end{steps}

\vspace{1em}{\footnotesize\itshape\color{terracotta}Met dank aan Nick van Dumpert,
naar zijn shakshuka (zie de bibliografie achterin).}
\end{recipe}
