\begin{recipe}{Shakshuka}
\kicker{Hoofdgerecht · vegetarisch}
\index[register]{Shakshuka}

\heroplaceholder{handgetekende illustratie van de pan shakshuka}
\meta{20 minuten \dvd Voor 4–6 personen \dvd Eén pan}

\lettrine{E}{ieren} gepocheerd in een pittige tomatensaus met paprika, komijn en
gerookte paprika. Lepel er feta en koriander overheen en dip er een stuk verse
pita in — lekker man.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{2 el}{extra vierge olijfolie}
  \ing{1}{ui, in stukken}\index[register]{Ui}
  \ing{1}{rode paprika, in stukken}\index[register]{Paprika, rode}
  \ing{4}{tenen knoflook, fijngehakt}\index[register]{Knoflook}
  \ing{1 tl}{komijnpoeder}
  \ing{1 tl}{gerookt paprikapoeder}
  \ing{1 tl}{Koreaanse pepervlokken}
  \ing{750 g}{tomaten, in blokjes}\index[register]{Tomaten}
  \ing{4–6}{eieren}\index[register]{Eieren}
  \ing{1}{blok feta, in blokjes}\index[register]{Feta}
  \ing{1}{handje verse koriander}\index[register]{Koriander}
  \ing{—}{zout \& peper, naar smaak}
\end{ingredients}
\ingb{2 el Raijmakers Tranquilizer (habanero \& zoete aardappel)}

\blockrule{Bereiding}
\tip{Eieren ± 5 minuten met de deksel erop — dan blijft het eigeel mooi lopend.}
\ingredientsketch{rode paprika}
\begin{steps}
  \item Zet een koekenpan op het vuur, schenk de olijfolie erin en laat heet
        worden.
  \item Snijd de ui en paprika in stukken en bak ze ± 5 minuten tot de ui glazig
        wordt.
  \item Voeg de knoflook, gerookt paprikapoeder en komijn toe en bak 1 minuut mee.
  \item Voeg de tomaten met zout en peper toe, zet het vuur laag en laat 15
        minuten pruttelen.
  \item Maak 6 kuiltjes in de saus en breek er één voor één de eieren in. Deksel
        erop en pocheer naar smaak — ± 5 minuten voor een lopend eigeel.
  \item Hak ondertussen de koriander fijn en snijd de feta in blokjes.
  \item Strooi de koriander en feta over de shakshuka.
  \item Pak een stuk pita en dip erin. Lekker man.
\end{steps}
\ingredientsketch{gepocheerd ei}
\end{recipe}
