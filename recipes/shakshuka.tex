\begin{recipe}{Shakshuka}
\kicker[Hoofdgerecht,Vegetarisch,Doordeweeks]{Hoofdgerecht · vegetarisch · doordeweeks}
\index[register]{Shakshuka}

\meta{20 minuten \dvd Voor 4–6 personen \dvd Eén pan}

\lettrine{E}{ieren} gepocheerd in een pittige tomatensaus met paprika, komijn en
gerookte paprika. Lepel er feta en peterselie overheen en dip er een stuk verse
pita in.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{6}{eieren}\index[register]{Eieren}
  \ing{200 g}{feta, in blokjes}\index[register]{Feta}
  \ing{750 g}{tomaten, in blokjes}\index[register]{Tomaten}
  \ing{2 el}{extra vierge olijfolie}
  \ing{1}{ui, in stukken}\index[register]{Ui}
  \ing{1}{rode paprika, in stukken}\index[register]{Paprika, rode}
  \ing{2}{knoflookteentjes, fijngehakt}\index[register]{Knoflook}
  \ing{1 tl}{komijnpoeder}
  \ing{1 tl}{gerookt paprikapoeder}
  \ing{1 tl}{Koreaanse pepervlokken}
  \ing{1}{handje verse peterselie}\index[register]{Peterselie}
  \ingb{zout \& versgemalen peper, naar smaak}
  \ingb{pita, om bij te serveren}
\end{ingredients}

\blockrule{Bereiding}
\tip{Wordt de saus te droog tijdens het pruttelen, voeg dan een scheutje water toe. Zelfgebakken pita erbij? Zie blz.\,\pageref{recipe:pita}.}
\begin{steps}
  \item Zet een hapjespan op het vuur, schenk de olijfolie erin en laat heet
        worden.
  \item Snijd de ui en paprika in stukken en bak ze ± 5 minuten tot de ui glazig
        wordt.
  \item Voeg de knoflook, gerookt paprikapoeder en komijn toe en bak 1 minuut mee.
  \item Voeg de tomaten met zout en peper toe, zet het vuur laag en laat 15
        minuten pruttelen.
  \item Maak 6 kuiltjes in de saus en breek er één voor één de eieren in. Deksel
        erop en pocheer ± 5 minuten. Zo blijft het eigeel mooi lopend.
  \item Hak ondertussen de peterselie fijn.
  \item Strooi de peterselie en feta over de shakshuka.
  \item Pak een stuk pita en dip erin.
\end{steps}
\end{recipe}
