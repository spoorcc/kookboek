\begin{recipe}{Paella}
\kicker[Hoofdgerecht,Vis,Spaans]{Hoofdgerecht · vis · Spaans}
\index[register]{Paella}
\index[register]{Zeevruchten}
\index[register]{Saffraan}

\meta{40 minuten \dvd Voor 4 personen \dvd Eén pan}

\lettrine{P}{aella} gaart net als risotto lepel voor lepel, met saffraan en
zeevruchten in plaats van kaas en boter. Een licht krokant laagje op de
bodem hoort erbij: dat lukt door de laatste minuten niet meer te roeren.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{750 ml}{water}
  \ing{2}{blokjes kippenbouillon}
  \ing{1}{plukje saffraan}
  \ing{1}{ui, gesnipperd}\index[register]{Ui}
  \ing{2 teentjes}{knoflook, fijngehakt}\index[register]{Knoflook}
  \ing{200 g}{risottorijst}
  \ing{1}{rode paprika, in blokjes}\index[register]{Paprika}
  \ing{1 blik (400 g)}{tomatenblokjes}
  \ing{1 tl}{gerookt paprikapoeder}
  \ing{200 g}{doperwten (diepvries)}
  \ing{400 g}{zeevruchtenmix, ontdooid}\index[register]{Zeevruchten}
  \ing{1 el}{verse peterselie, fijngehakt}
  \ingb{peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Een goede paella heeft een licht krokant laagje aan de onderkant. Dat
lukt door de laatste 5 tot 10 minuten niet meer om te scheppen en het vuur
op middelhoog te houden.}
\begin{steps}
  \item Schenk het water in een pan en voeg de bouillonblokjes en de
        saffraan toe. Breng aan de kook en roer tot de bouillonblokjes zijn
        opgelost. Houd de bouillon daarna op laag vuur warm.
  \item Fruit de ui en de knoflook in een hapjespan tot de ui glazig ziet.
  \item Voeg de risottorijst toe en bak de korrels al roerend een paar
        minuten mee, tot ze glazig worden.
  \item Voeg de paprika, de tomatenblokjes en het gerookte paprikapoeder
        toe en roer alles goed door.
  \item Schenk een flinke soeplepel bouillon bij de rijst. Roer goed door
        en laat op laag vuur garen. Voeg steeds een nieuwe lepel bouillon
        toe zodra de vorige bijna is opgenomen.
  \item Voeg bij de laatste lepel bouillon de doperwten en de helft van de
        zeevruchtenmix toe, zodat ze mee kunnen garen. Breng op smaak met
        peper en zout.
  \item Verdeel de rest van de zeevruchtenmix over de paella zodra de rijst
        gaar is. Doe het deksel op de pan, zodat ook deze zeevruchten warm
        worden.
  \item Werk af met de verse peterselie.
\end{steps}

\end{recipe}
