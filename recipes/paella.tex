\begin{recipe}{Paella}
\kicker[Hoofdgerecht,Vlees,Spaans]{Hoofdgerecht · vlees · Spaans}
\index[register]{Paella}
\index[register]{Saffraan}
\index[register]{Kipdijfilet}
\index[register]{Surimi}

\meta{40 minuten \dvd Eén pan}

\lettrine{P}{aella} gaart net als risotto lepel voor lepel. Jambalaya is er
familie van. Hier lust alleen Ben mosselen, dus die blijven in de kast, en
de kinderen lusten geen vis: deze paella maken we daarom met kipdijfilet,
met een bakje warme surimi sticks en eventueel wat garnalen los erbij. Een
licht krokant laagje op de bodem hoort erbij: dat lukt door de laatste
minuten niet meer te roeren.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{750 ml}{water}
  \ing{2}{blokjes kippenbouillon}
  \ing{1}{plukje saffraan}
  \ing{2 el}{olijfolie}
  \ing{200 g}{kipdijfilet, in stukken}\index[register]{Kipdijfilet}
  \ing{1}{ui, gesnipperd}\index[register]{Ui}
  \ing{2 teentjes}{knoflook, fijngehakt}\index[register]{Knoflook}
  \ing{200 g}{risottorijst}
  \ing{1}{rode paprika, in blokjes}\index[register]{Paprika}
  \ing{1 blik (400 g)}{tomatenblokjes}
  \ing{1 tl}{gerookt paprikapoeder}
  \ing{200 g}{doperwten (diepvries)}
  \ing{1 el}{verse peterselie, fijngehakt}
  \ingb{peper en zout}
  \ing{200 g}{surimi sticks, apart warm te maken}\index[register]{Surimi}
  \ing{200 g}{garnalen, eventueel apart warm te maken}
\end{ingredients}

\blockrule{Bereiding}
\tip{Saffraan hoort er traditioneel bij, maar is prijzig. Vervang het
gerust door een snufje kurkuma en wat extra gerookt paprikapoeder.}
\begin{steps}
  \item Schenk het water in een steelpannetje en voeg de bouillonblokjes en
        de saffraan toe. Breng aan de kook en roer tot de bouillonblokjes
        zijn opgelost. Houd de bouillon daarna op laag vuur warm.
  \item Verhit de olijfolie in een hapjespan en bak de kipdijfilet aan tot
        hij mooi bruin is. Schep de kip uit de pan en zet apart.
  \item Fruit de ui en de knoflook in dezelfde hapjespan tot de ui glazig
        ziet.
  \item Voeg de risottorijst toe en bak de korrels al roerend een paar
        minuten mee, tot ze glazig worden.
  \item Voeg de paprika, de tomatenblokjes en het gerookte paprikapoeder
        toe en roer alles goed door.
  \item Schenk een flinke soeplepel bouillon bij de rijst. Roer goed door
        en laat op laag vuur garen. Voeg steeds een nieuwe lepel bouillon
        toe zodra de vorige bijna is opgenomen.
  \item Voeg bij de laatste lepel bouillon de doperwten en de kip terug
        toe, zodat ze mee kunnen garen. Breng op smaak met peper en zout.
  \item Verhit ondertussen de surimi sticks, en de garnalen als je die
        gebruikt, kort in een klein pannetje met een klein beetje boter.
  \item Werk de paella af met de verse peterselie en zet de surimi en
        garnalen er los bij, zodat iedereen zelf kiest wat erop gaat.
\end{steps}

\vspace{1em}{\footnotesize\itshape\color{terracotta}Naar een recept van Sabine.}
\end{recipe}
