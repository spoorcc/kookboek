\begin{recipe}{Paella}
\kicker[Hoofdgerecht,Vlees,Spaans]{Hoofdgerecht · vlees · Spaans}
\index[register]{Paella}
\index[register]{Saffraan}
\index[register]{Kipdijfilet}
\index[register]{Surimi}

\meta{40 minuten \dvd Eén pan}

\lettrine{P}{aella} is het tegenovergestelde van risotto: de bouillon gaat in
één keer in de pan in plaats van lepel voor lepel, en daarna wordt er niet
meer geroerd. Jambalaya is er familie van. Hier lust alleen ik mosselen,
dus die blijven in de kast, en de kinderen lusten geen vis: deze paella
maken we daarom met kipdijfilet, met een bakje warme surimi sticks en
eventueel wat garnalen los erbij. Een licht krokant laagje op de bodem
hoort erbij, de socarrat: dat lukt alleen als je na het bijgieten van de
bouillon niet meer roert.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{200 g}{kipdijfilet, in stukken}\index[register]{Kipdijfilet}
  \ing{200 g}{surimi sticks, apart warm te maken}\index[register]{Surimi}
  \ing{200 g}{garnalen, eventueel apart warm te maken}
  \ing{200 g}{risottorijst}
  \ing{1}{rode paprika, in blokjes}\index[register]{Paprika}
  \ing{750 ml}{water}
  \ing{2}{blokjes kippenbouillon}
  \ing{1}{plukje saffraan}
  \ing{2 el}{olijfolie}
  \ing{1}{ui, gesnipperd}\index[register]{Ui}
  \ing{2 teentjes}{knoflook, fijngehakt}\index[register]{Knoflook}
  \ing{1 blik (400 g)}{tomatenblokjes}
  \ing{1 tl}{gerookt paprikapoeder}
  \ing{200 g}{doperwten (diepvries)}
  \ing{1 el}{verse peterselie, fijngehakt}
  \ingb{peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Saffraan hoort er traditioneel bij, maar is prijzig. Vervang het
gerust door een snufje kurkuma en wat extra gerookt paprikapoeder.}
\begin{steps}
  \item Los de bouillonblokjes en de saffraan op in het water in een
        steelpannetje. Breng aan de kook en houd daarna op laag vuur warm.
  \item Bak de kipdijfilet op middelhoog tot hoog vuur in de hapjespan in de
        olijfolie mooi bruin. Haal de kip uit de pan en zet apart.
  \item Bak in dezelfde pan de ui en de knoflook glazig.
  \item Roer de risottorijst erdoor en laat de korrels een paar minuten
        glazig bakken.
  \item Meng de paprika, de tomatenblokjes en het gerookte paprikapoeder
        erdoor.
  \item Leg de kip terug in de pan en verdeel de doperwten erover. Maak op
        smaak met peper en zout: dit is het laatste moment om te roeren.
  \item Giet de hete bouillon in één keer over de rijst en schud de pan
        (niet roeren) tot alles gelijkmatig verdeeld is. Zet het vuur laag
        en laat de rijst ± 15 minuten ongestoord de bouillon opnemen. Zet de
        laatste 2 tot 3 minuten het vuur wat hoger en laat de pan
        onaangeroerd staan: zo krijg je de socarrat, het krokante
        bodemlaagje.
  \item Warm ondertussen op laag vuur in een apart pannetje met een klein beetje
        boter de surimi sticks, en de garnalen als je die gebruikt.
  \item Bestrooi de paella met peterselie en zet de surimi en de garnalen
        los erbij, zodat iedereen zelf kan kiezen.
\end{steps}
