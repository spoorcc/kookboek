\begin{recipe}{Couscous met geroosterde tomaten en witte kaas}
\kicker[Hoofdgerecht,Vegetarisch,Arabisch,Doordeweeks]{Hoofdgerecht · vegetarisch · Arabisch · doordeweeks}
\index[register]{Couscous met geroosterde tomaten}
\index[register]{Couscous}
\index[register]{Olijven}
\index[register]{Witte kaas}

\meta{25 minuten \dvd Arabisch}

\lettrine{D}{it} was jaren een kampeergerecht: vegetarisch, snel klaar op een kleine
brander en maar één pan. Kaneel en kardemom in een hartig gerecht klinkt ongewoon,
maar het past hier goed bij de geroosterde tomaten.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{cherrytomaten}\index[register]{Tomaten}
  \ing{1}{knoflookteentje}
  \ing{1 blikje}{maïs, uitgelekt}\index[register]{Maïs}
  \ing{⅔ pot}{zwarte olijven, gehalveerd of in ringen}\index[register]{Olijven}
  \ing{1 tl}{kardemom}
  \ing{160 g}{couscous}\index[register]{Couscous}
  \ing{160 ml}{kokend water}
  \ing{½ blok}{witte kaas (b.v.\ feta), verkruimeld}\index[register]{Witte kaas}
  \ingb{olijfolie}
  \ing{1 el}{verse munt (of blaadjes van 3 takjes)}
  \ing{1 el}{paprikapoeder}
  \ing{1 tl}{kaneel}
  \ing{1 tl}{uienpoeder}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\begin{steps}
  \item Halveer de olijven of snijd ze in ringen. Zet water op.
  \item Verhit een hapjespan goed voor met een scheut olijfolie. Doe de cherrytomaten
        erin en schud de pan regelmatig. Laat ze op hoog vuur stukgaan en licht bruin
        kleuren.
  \item Zet het vuur lager. Pers de knoflook boven de pan uit en roer een minuut
        door de tomaten.
  \item Voeg de maïs toe en bak tot het vocht grotendeels weg is.
  \item Voeg de olijven, munt, paprikapoeder, kardemom, kaneel en uienpoeder toe.
        Roer een minuut door en zet het vuur uit.
  \item Schep de couscous door het mengsel en giet het kokende water erover.
        Verkruimel de witte kaas erop. Leg het deksel op de pan en laat 5 minuten
        staan.
  \item Roer los en breng op smaak met peper en zout.
\end{steps}
\end{recipe}
