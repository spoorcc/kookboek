\begin{recipe}{Zelfgemaakte vla}
\kicker[Dessert,Zoet,Basisrecept]{Dessert · zoet · basisrecept}
\index[register]{Zelfgemaakte vla}
\index[register]{Vanille}
\index[register]{Eidooiers}
\index[register]{Maïzena}

\meta{15 minuten \dvd Ongeveer 1 liter \dvd Eenvoudig}

\lettrine{W}{e} hebben deze vla ooit gemaakt om te zien hoe dat eigenlijk gaat, en
het bleek best makkelijk met een paar basisingrediënten. Reken wel op een paar
uur afkoeltijd voor een romige vanillevla.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 liter}{volle melk}
  \ing{75 g}{kristalsuiker}
  \ing{40 g}{maïzena}\index[register]{Maïzena}
  \ing{2}{eidooiers}\index[register]{Eidooiers}
  \ing{1}{vanillestokje}\index[register]{Vanille}
  \ingb{snufje zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Leg de plasticfolie echt rechtstreeks op de vla, zonder lucht ertussen. Zo
voorkom je een taai velletje.}
\begin{steps}
  \item Giet 900 ml van de melk in een pan en houd de resterende 100 ml apart.
  \item Snijd het vanillestokje overlangs open, schraap het merg eruit en
        voeg merg en stokje toe aan de melk. Breng op middelhoog vuur
        langzaam tegen de kook aan en laat op laag vuur 10 minuten trekken.
  \item Meng in een kom de suiker, maïzena en het snufje zout. Roer de apart
        gehouden koude melk erdoor tot een glad papje en klop er de
        eidooiers doorheen.
  \item Haal het vanillestokje uit de warme melk. Giet al roerend een scheut
        warme melk bij het eimengsel en roer meteen goed door, zodat de
        eieren niet stollen.
  \item Giet het verdunde eimengsel terug bij de rest van de warme melk.
        Breng op laag vuur onder voortdurend roeren met een garde zachtjes
        aan de kook en laat 1 tot 2 minuten doorkoken tot de vla dikker
        wordt en bindt.
  \item Giet de warme vla door een fijne zeef in een schaal en dek meteen af
        met een stuk plasticfolie, rechtstreeks op de vla, zodat er geen vel
        ontstaat. Laat minimaal 2 uur afkoelen in de koelkast.
  \item Klop de vla voor het serveren nog kort door met een garde voor een
        extra glad resultaat.
\end{steps}
\end{recipe}
