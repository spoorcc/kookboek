\begin{recipe}{Venkelrisotto met geitenkaas}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans]{Hoofdgerecht · vegetarisch · Italiaans}
\index[register]{Venkelrisotto met geitenkaas}
\index[register]{Venkel}
\index[register]{Geitenkaas}
\index[register]{Risotto}

\meta{35 minuten}

\lettrine{V}{enkelrisotto} vraagt iets meer aandacht dan een gewone pasta, met al
dat roeren en bouillon toevoegen, maar de geitenkaas en citroen geven de risotto
er een frisse smaak voor terug die goed past bij de anijsachtige venkel.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{2}{venkelknollen}\index[register]{Venkel}
  \ing{120 g}{geitenkaasschijfjes naturel}\index[register]{Geitenkaas}
  \ing{170 g}{risottorijst}\index[register]{Risottorijst}
  \ing{700 ml}{kraanwater}
  \ing{2}{groentebouillonblokjes}
  \ing{1}{middelgrote ui, gesnipperd}
  \ing{2}{knoflookteentjes, fijngehakt}
  \ing{30 g}{ongezouten roomboter}
  \ing{100 ml}{droge witte wijn}
  \ing{1}{citroen}
  \ingb{versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Voeg het citroensap beetje bij beetje toe en proef tussendoor. Risotto heeft meer zuur nodig dan je verwacht.}
\begin{steps}
  \item Breng het water met de bouillonblokjes aan de kook en zet het vuur laag.
        Snipper de ui en snijd de knoflook fijn. Snijd de onderkant en de stelen van
        de venkelknollen en bewaar het venkelgroen apart. Snijd de venkel in blokjes
        van 1 cm. Hou de stelen apart en snij die in kleine ringetjes.
  \item Verhit de boter in een hapjespan met dikke bodem op laag vuur. Voeg de ui en de
        knoflook toe en fruit ze 3 minuten tot de ui glazig is. Voeg de venkel zonder stelen
        toe en bak 2 minuten mee.
  \item Voeg de risottorijst toe en bak al roerend 3 minuten mee tot de korrels glazig
        zijn. Blus af met de witte wijn en roer tot die is opgenomen.
  \item Voeg een scheut hete bouillon toe en roer tot die is opgenomen. Voeg dan de volgende
        schep toe. Blijf dit herhalen.
  \item Na ongeveer 15 minuten voeg je de venkelstelen toe. Blijf bouillon toevoegen en roeren
        tot de rijst gaar is, ongeveer 20 minuten. De risotto hoort dik vloeibaar te
        zijn. Je hoeft niet alle bouillon te gebruiken.
  \item Rasp ondertussen de gele schil van de citroen en pers de vrucht uit.
        Verkruimel de geitenkaas.
  \item Neem de pan van het vuur en roer de helft van de geitenkaas erdoor. Breng op
        smaak met het citroenrasp, 2 el citroensap en versgemalen peper. Garneer
        met de rest van de geitenkaas en het venkelgroen. Serveer direct.
\end{steps}

\end{recipe}
