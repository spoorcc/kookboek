\begin{recipe}{Venkelrisotto met geitenkaas}
\kicker{Hoofdgerecht · vegetarisch}
\index[register]{Venkelrisotto met geitenkaas}
\index[register]{Venkel}
\index[register]{Geitenkaas}
\index[register]{Risotto}

\meta{35 minuten \dvd Voor 5 personen \dvd Italiaans}

\lettrine{V}{enkelrisotto} vraagt iets meer aandacht dan een gewone pasta, maar dat
heeft zijn redenen. De geitenkaas en citroen geven de risotto een frisse smaak die
goed past bij de anijsachtige venkel.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1¼ liter}{kraanwater}
  \ing{2}{groentebouillontabletten}
  \ing{1}{middelgrote ui, gesnipperd}
  \ing{2}{knoflooktenen, fijngehakt}
  \ing{2}{venkelknollen}\index[register]{Venkel}
  \ing{30 g}{ongezouten roomboter}
  \ing{300 g}{risottorijst}\index[register]{Risottorijst}
  \ing{100 ml}{droge witte wijn}
  \ing{1}{citroen}
  \ing{120 g}{geitenkaas blanc 50+}\index[register]{Geitenkaas}
  \ingb{versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Voeg het citroensap beetje bij beetje toe en proef tussendoor. Risotto heeft meer zuur nodig dan je verwacht.}
\ingredientsketch{venkel}
\begin{steps}
  \item Breng het water met de bouillontabletten aan de kook en zet het vuur laag.
        Snipper de ui en snijd de knoflook fijn. Snijd de onderkant en de stelen van
        de venkelknollen en bewaar het venkelgroen apart. Snijd de venkel in blokjes
        van 1 cm.
  \item Verhit de boter in een pan met dikke bodem op laag vuur. Voeg de ui en de
        knoflook toe en fruit ze 3 minuten tot de ui glazig is. Voeg de venkel toe en
        bak 2 minuten mee.
  \item Voeg de risottorijst toe en bak al roerend 3 minuten mee tot de korrels glazig
        zijn. Voeg de witte wijn toe en roer tot die is opgenomen.
  \item Voeg een scheut hete bouillon toe en roer tot die is opgenomen. Herhaal dit
        tot de rijst gaar is, ongeveer 20 minuten. De risotto hoort dik vloeibaar te
        zijn. Je hoeft niet alle bouillon te gebruiken.
  \item Rasp ondertussen de gele schil van de citroen en pers de vrucht uit.
        Verkruimel de geitenkaas.
  \item Neem de pan van het vuur en roer de helft van de geitenkaas erdoor. Breng op
        smaak met het citroenrasp, 2 el citroensap en versgemalen peper. Garneer
        met de rest van de geitenkaas en het venkelgroen. Serveer direct.
\end{steps}
\end{recipe}
