\begin{recipe}{Tortellini al forno met courgette en spek}
\kicker[Hoofdgerecht,Vlees,Italiaans]{Hoofdgerecht · vlees · Italiaans}
\index[register]{Tortellini al forno}
\index[register]{Tortellini}
\index[register]{Courgette}
\index[register]{Spek}

\meta{45 minuten \dvd Oven 200\,°C}

\lettrine{T}{ortellini} gebakken in de oven met een knapperig korstje van
mozzarella, parmezaan en paneermeel. De basis is een simpele pan met
uitgebakken spek, ui en courgette. Verse tortellini werkt goed, maar gedroogde
kan ook.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1}{courgette, in blokjes}\index[register]{Courgette}
  \ing{250 g}{spekblokjes}\index[register]{Spek}
  \ing{500 g}{verse tortellini (of gedroogd)}\index[register]{Tortellini}
  \ing{1}{ui}
  \ing{250 g}{ricotta}\index[register]{Ricotta}
  \ing{50 g}{Parmezaanse kaas, geraspt}\index[register]{Parmezaanse kaas}
  \ing{200 g}{mozzarella}
  \ingb{paneermeel}
  \ingb{versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Je kunt dit gerecht van tevoren maken en vlak voor het eten opwarmen in de oven op 180\,°C.
Zelf ricotta maken kan ook, zie het basisrecept op p.~\pageref{recipe:ricotta}.}
\begin{steps}
  \item Verwarm de oven voor op 200\,°C (boven- en onderwarmte). Zet water op voor de tortellini.
  \item Snipper de ui. Was de courgette en snijd ze in blokjes.
  \item Bak de spekjes op middelhoog vuur uit in een grote hapjespan. Voeg de ui
        toe en bak tot glazig.
  \item Voeg de courgette toe en bak op hoog vuur tot hij mooi lichtbruin
        kleurt: dat is smaak, bleke courgette is alleen vulling. Zet dan het
        vuur laag.
  \item Kook de tortellini gaar en schep een flinke scheut kookvocht apart voor
        je afgiet: het zetmeel daarin helpt straks de saus binden.
  \item Schep de tortellini door de pan met spek en groente. Voeg de ricotta,
        parmezaanse kaas en een scheut kookvocht toe en roer goed door tot een
        romige saus. Breng op smaak met peper. Het spek zorgt voor het zout.
  \item Schep alles in een ovenschaal. Scheur de mozzarella er losjes over,
        strooi nog wat parmezaan en een laagje paneermeel erover.
  \item Zet in de oven tot de bovenkant goed bruin is, ca.\ 15--20 minuten.
\end{steps}

\end{recipe}
