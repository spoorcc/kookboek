\begin{recipe}{Asperges}
\kicker[Hoofdgerecht,Vlees,Brabantse Wal]{Hoofdgerecht · vlees · Brabantse Wal}
\index[register]{Asperges}
\index[register]{Krieltjes}
\index[register]{Ham}

\meta{30 minuten \dvd Voor 4 personen}

\lettrine{A}{sperges} zijn hier op de Brabantse Wal een traditie. In het
seizoen koop je ze het liefst vers bij de boer, bijvoorbeeld bij
boerderijcamping De Meet, waar ze meteen voor je geschild worden.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 kg}{verse witte asperges}\index[register]{Asperges}
  \ing{1 kg}{krieltjes}\index[register]{Krieltjes}
  \ing{6}{middelgrote scharreleieren}
  \ing{1 bosje}{bieslook}
  \ing{4 el}{mayonaise}
  \ing{8}{plakjes achterham}\index[register]{Ham}
\end{ingredients}

\blockrule{Bereiding}
\tip{Bij ons hoort de traditie van de drie A's erbij: aardappelen, asperges
en aardbeien. Sluit af met aardbeien en een likje vanille yoghurt als
toetje, als het aspergeseizoen dat toelaat.}
\begin{steps}
  \item Knip de bieslook alvast fijn, zodat je daar straks niet meer naar
        om hoeft te kijken.
  \item Halveer de grotere krieltjes. Geschilde asperges (van de boer of
        kant-en-klaar gekocht) hoef je verder niet meer te schillen.
  \item Zet drie pannen met ruim water op, eventueel gezouten: een
        aspergepan of gewoon een hoge pan voor de asperges, een gewone pan
        voor de krieltjes en een steelpannetje voor de eieren. Breng alle
        drie aan de kook.
  \item Kook de asperges 10 minuten beetgaar, de krieltjes in de andere
        pan 8 tot 10 minuten gaar, en de eieren in het steelpannetje
        8 minuten hard. Zet ze tegelijk op, dan zijn ze ook tegelijk klaar.
  \item Giet de eieren af zodra ze gaar zijn, koel ze onder koud water en
        pel ze. Snijd ze in parten.
  \item Giet de asperges en de krieltjes af en verdeel ze over de borden.
        Verdeel de eierpartjes en de bieslook over de asperges en maal er
        goed wat peper overheen. Zet een dot mayonaise ernaast. Verdeel de
        ham in mooie roosjes en leg ze erbij.
\end{steps}
\end{recipe}
