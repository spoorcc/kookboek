\begin{recipe}{Asperges}
\kicker[Hoofdgerecht,Vlees,Brabantse Wal]{Hoofdgerecht · vlees · Brabantse Wal}
\index[register]{Asperges}
\index[register]{Krieltjes}
\index[register]{Ham}

\meta{30 minuten \dvd Voor 4 personen}

\lettrine{A}{sperges} zijn hier op de Brabantse Wal een traditie. In het
seizoen koop je ze het liefst vers bij de boer, die ze meteen voor je
schilt.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 kg}{verse witte asperges}\index[register]{Asperges}
  \ing{6}{middelgrote scharreleieren}
  \ing{8}{plakjes achterham}\index[register]{Ham}
  \ing{700 g}{krieltjes}\index[register]{Krieltjes}
  \ing{1 bosje}{bieslook}
  \ing{4 el}{mayonaise}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Een echte Bergse traditie: de drie A's, aardappelen, asperges en
aardbeien, horen bij elkaar. Sluit de maaltijd af met aardbeien als
toetje.}
\begin{steps}
  \item Knip de bieslook alvast fijn. Was de krieltjes en halveer de grotere
        exemplaren. Geschilde asperges (van de boer of kant-en-klaar
        gekocht) hoef je verder niet meer te schillen.
  \item Zet drie pannen met ruim water op, eventueel gezouten: een
        aspergepan of gewoon een hoge pan voor de asperges, een gewone pan
        voor de krieltjes en een steelpannetje voor de eieren. Breng alle
        drie aan de kook.
  \item Kook de asperges 8 minuten, zet dan het vuur uit en laat ze nog 5 tot
        10 minuten met de deksel erop in het hete water staan: zo garen ze
        zachtjes na zonder ooit slap te koken, de veiligste cuisson voor
        witte asperges. Zet de asperges daarom als eerste op, en de krieltjes
        (8 tot 10 minuten gaar) en de eieren (8 minuten hard) een paar
        minuten later, dan zijn ze ook ongeveer tegelijk klaar.
  \item Giet de eieren af zodra ze gaar zijn, koel ze onder koud water, pel
        ze en snijd ze in parten.
  \item Giet de asperges en de krieltjes af en verdeel ze over de borden.
        Verdeel de eierpartjes en de bieslook over de asperges en maal er
        goed wat peper overheen. Zet een dot mayonaise ernaast, verdeel de
        ham in mooie roosjes erbij en breng krieltjes en ham op smaak met
        een beetje zout.
\end{steps}
\end{recipe}
