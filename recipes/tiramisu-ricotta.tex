\begin{recipe}{Tiramisu met ricotta}
\kicker[Dessert,Zoet,Italiaans]{Dessert · zoet · Italiaans}
\index[register]{Tiramisu met ricotta}
\index[register]{Ricotta}
\index[register]{Lange vingers}
\index[register]{Espresso}

\meta{30 minuten + 4 uur koelen}

\lettrine{D}{eze} tiramisu gebruikt ricotta in plaats van mascarpone, wat hem wat
lichter maakt. De basis is een luchtig mengsel van opgeklopt eiwit en eigeel met
suiker, waardoor je ook zonder slagroom een mooie stevige laag krijgt.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{250 g}{ricotta}\index[register]{Ricotta}
  \ing{3}{eieren}\index[register]{Eieren}
  \ing{2 zakjes}{vanillesuiker}
  \ing{1 pak}{lange vingers}\index[register]{Lange vingers}
  \ing{180 g}{suiker}
  \ing{1 kop}{sterke espresso}\index[register]{Espresso}
  \ingb{cacaopoeder, voor het bestuiven}
  \ingb{scheut drank of siroop, naar smaak}
\end{ingredients}

\blockrule{Bereiding}
\tip{Dip de lange vingers heel kort door de koffie: een seconde per kant is genoeg.
Te lang weken en ze worden zompig. Zelf ricotta maken kan ook, zie het
basisrecept op p.~\pageref{recipe:ricotta}.}
\begin{steps}
  \item Zet een sterke kop koffie en laat die afkoelen. Voeg naar smaak een scheut
        drank of siroop toe en schenk het mengsel in een ondiepe schaal.
  \item Splits de eieren. Klop in een grote kom het eigeel met de suiker tot een
        bleke, romige massa.
  \item Klop in een andere kom de eiwitten met de vanillesuiker op tot stijve pieken.
  \item Meng de ricotta goed door het eigeelmengsel. Spatel daarna het eiwit
        voorzichtig erdoor, zodat er zoveel mogelijk lucht in blijft.
  \item Haal de lange vingers kort door de koffie en leg een laag op de bodem van een
        schaal. Verdeel de helft van de vulling erover. Leg een tweede laag lange
        vingers erop en schep de rest van de vulling eroverheen. Strijk glad.
  \item Dek af en zet minstens 4 uur in de koelkast om op te stijven. Zeef vlak voor
        het serveren een laagje cacaopoeder over de bovenkant.
\end{steps}
\end{recipe}
