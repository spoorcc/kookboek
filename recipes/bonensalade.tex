\begin{recipe}{Bonensalade met bleekselderij en citrus}
\kicker[Bijgerecht,Vegetarisch]{Bijgerecht · vegetarisch}
\index[register]{Bonensalade}
\index[register]{Kidneybonen}
\index[register]{Cannellini bonen}
\index[register]{Bleekselderij}

\meta[Amerikaans]{15 minuten \dvd Voor 4 personen \dvd Amerikaans}

\lettrine{D}{rie} soorten bonen uit blik met een frisse citrusdressing. Klaar
in een kwartier en goed mee te nemen op picknick of camping. Van tevoren
maken mag: een nacht in de koelkast verdiept de smaak.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 blik (255 g)}{reuzenbonen}\index[register]{Reuzenbonen}
  \ing{1 blikje (175 g)}{rode kidneybonen}\index[register]{Kidneybonen}
  \ing{1 blikje (175 g)}{cannellini bonen}\index[register]{Cannellini bonen}
  \ing{½}{rode ui, gesnipperd}
  \ing{1 stengel}{bleekselderij, in kleine blokjes}\index[register]{Bleekselderij}
  \ing{3 el}{olijfolie extra vergine}
  \ing{2 el}{citroensap of rode wijnazijn}
  \ing{10 g}{platte peterselie, fijngehakt}
  \ing{1 tl}{gedroogde rozemarijn}
  \ingb{zout en versgemalen zwarte peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Een dag eerder maken werkt prima: de dressing trekt dan goed in de bonen en de smaak wordt voller.}
\begin{steps}
  \item Spoel de drie soorten bonen in een vergiet met stromend water en laat
        goed uitlekken.
  \item Meng de bonen met de rode ui, bleekselderij, peterselie en rozemarijn
        in een kom.
  \item Klop de olijfolie en het citroensap of de rode wijnazijn door elkaar
        met zout en peper. Meng de dressing door de salade. Proef en stuur
        bij met extra zout of peper.
  \item Bewaar afgedekt in de koelkast tot gebruik.
\end{steps}
\end{recipe}
