\begin{recipe}{Broccolipasta met witvis, olijven en rode ui}
\kicker[Hoofdgerecht,Vis,Italiaans,Doordeweeks]{Hoofdgerecht · vis · Italiaans · doordeweeks}
\index[register]{Broccolipasta met witvis, olijven en rode ui}
\index[register]{Broccoli}
\index[register]{Olijven}
\index[register]{Rode ui}

\meta{30 minuten}

\lettrine{E}{en} romige pastaschotel met witvis en broccoli, met de zilte bite
van zwarte olijven en de scherpte van rode ui erdoorheen. De light kookroom
houdt het geheel luchtig zonder dat het gerecht saus tekortkomt.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{300 g}{broccoli, in kleine roosjes}\index[register]{Broccoli}
  \ing{2 filets (± 300 g)}{witvis (bijv. kabeljauw of koolvis)}
  \ing{100 g}{zwarte olijven zonder pit, in schijfjes}\index[register]{Olijven}
  \ing{1}{rode ui, in halve ringen}\index[register]{Rode ui}
  \ing{350 g}{pasta}
  \ing{200 ml}{light kookroom}
  \ing{1}{knoflookteentje}\index[register]{Knoflook}
  \ingb{olijfolie, om te bakken}
  \ingb{versgemalen peper en zout}
  \ingb{Parmezaanse kaas, geraspt, om te serveren}\index[register]{Parmezaanse kaas}
\end{ingredients}

\blockrule{Bereiding}
\tip{Bak de vis niet te lang mee in de saus: hij is al gaar zodra hij
ondoorzichtig en stevig aanvoelt, en valt daarna snel uit elkaar tot
losse vlokken.}
\begin{steps}
  \item Kook de pasta beetgaar volgens de aanwijzingen op de verpakking.
  \item Verdeel ondertussen de broccoli in kleine roosjes. Snijd de rode ui in
        halve ringen en de olijven in schijfjes. Pel en pers de knoflook.
  \item Verhit een scheutje olijfolie op middelhoog vuur in een hapjespan en
        bak de rode ui 3 tot 4 minuten tot hij zacht en glazig is. Voeg de
        knoflook toe en bak 1 minuut mee.
  \item Voeg de broccoliroosjes toe met een scheutje water en laat ze op
        middelhoog vuur met het deksel op de pan zo'n 5 minuten stoven tot ze
        beetgaar zijn.
  \item Dep de vis droog met keukenpapier en snijd hem in grove stukken. Leg
        de stukken tussen de broccoli in de pan en bak op laag vuur 3 tot 4
        minuten tot de vis net gaar is.
  \item Giet de light kookroom en de olijven erbij en laat alles nog 1 minuut
        zachtjes doorwarmen. Breng op smaak met peper en zout.
  \item Giet de pasta af en schep hem voorzichtig door de saus, zodat de
        vis heel blijft. Serveer direct met Parmezaanse kaas erover.
\end{steps}
\end{recipe}
