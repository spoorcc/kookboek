\begin{recipe}{Broccolipasta met witvis, olijven en rode ui}
\kicker[Hoofdgerecht,Vis,Italiaans,Doordeweeks]{Hoofdgerecht · vis · Italiaans · doordeweeks}
\index[register]{Broccolipasta met witvis, olijven en rode ui}
\index[register]{Broccoli}
\index[register]{Olijven}
\index[register]{Rode ui}

\meta{30 minuten}

\lettrine{E}{en} romige pastaschotel met witvis en broccoli, met de zilte bite
van zwarte olijven en de scherpte van rode ui erdoorheen. De light kookroom
houdt het geheel luchtig zonder dat het gerecht saus tekortkomt.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{300 g}{broccoli, in kleine roosjes}\index[register]{Broccoli}
  \ing{2 filets (± 300 g)}{witvis (bijv. kabeljauw of koolvis)}
  \ing{150 g}{zwarte olijven zonder pit, in schijfjes}\index[register]{Olijven}
  \ing{1}{rode ui, in halve ringen}\index[register]{Rode ui}
  \ing{350 g}{pasta}
  \ing{200 ml}{light kookroom}
  \ing{1}{knoflookteentje}\index[register]{Knoflook}
  \ing{1}{citroen, gewassen, voor de rasp}\index[register]{Citroen}
  \ingb{olijfolie, om te bakken}
  \ingb{versgemalen peper en zout}
  \ingb{platte peterselie, fijngehakt, om af te maken}\index[register]{Peterselie}
  \ingb{Parmezaanse kaas, geraspt, om te serveren}\index[register]{Parmezaanse kaas}
\end{ingredients}

\blockrule{Bereiding}
\tip{Bak de vis niet te lang mee in de saus: hij is al gaar zodra hij
ondoorzichtig en stevig aanvoelt, en valt daarna snel uit elkaar tot
losse vlokken.}
\begin{steps}
  \item Kook de pasta al dente: meestal een minuut korter dan de kooktijd op
        de verpakking. Bewaar een kop kookvocht voor je afgiet.
  \item Was ondertussen de broccoli en verdeel in kleine roosjes. Blancheer de
        roosjes 2 tot 3 minuten in ruim kokend water tot ze felgroen en
        beetgaar zijn, giet ze af en zet apart.
  \item Snijd de rode ui in halve ringen en de olijven in schijfjes. Pel en
        pers de knoflook.
  \item Verhit een scheutje olijfolie op hoog vuur in een wokpan of hapjespan
        en roerbak de rode ui ongeveer 1 minuut, tot hij warm is maar nog
        stevig blijft. Voeg de knoflook toe en roerbak een half minuutje mee.
  \item Dep de vis droog met keukenpapier en snijd hem in grove stukken. Bak
        de stukken tussen de ui op laag vuur 3 tot 4 minuten tot de vis net
        gaar is. Schep de vis en de ui uit de pan en zet apart, zodat er
        ruimte is voor de broccoli.
  \item Zet het vuur weer hoog en roerbak de broccoliroosjes in dezelfde pan
        1 tot 2 minuten tot ze een beetje kleur krijgen.
  \item Zet het vuur laag en doe de vis en de ui terug in de pan, samen met de
        olijven en de light kookroom. Laat alles nog 1 minuut zachtjes
        doorwarmen zonder dat de vis verder gaart. Proef voor je zout
        toevoegt: de olijven en de kaas op het eind brengen al genoeg zilt
        mee.
  \item Giet de pasta af en schep hem samen met een scheut kookvocht door de
        saus in de pan. Laat het geheel op laag vuur nog een minuut zachtjes
        binden, zwenk of schep de pan rustig door zodat de saus mooi aan de
        pasta gaat kleven, en pas op dat de vis heel blijft. Rasp er wat
        citroenschil overheen en breng op smaak met peper.
  \item Verdeel over de borden, strooi de peterselie erover en serveer direct
        met Parmezaanse kaas erbij.
\end{steps}
\end{recipe}
