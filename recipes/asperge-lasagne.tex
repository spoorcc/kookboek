\begin{recipe}{Mini-lasagne met asperges, ham en geitenkaas}
\kicker[Hoofdgerecht,Vlees,Hartig]{Hoofdgerecht · met ham · hartig}
\index[register]{Mini-lasagne met asperges, ham en geitenkaas}
\index[register]{Asperges}
\index[register]{Ham}
\index[register]{Geitenkaas}

\meta{35 minuten \dvd Voor 4 personen}

\lettrine{E}{en} gestapeld gerecht van witte asperges en ham met een laagje geitenkaas
en crème fraîche, drie keer op elkaar. Met 20 minuten voorbereiding en een kwartier
oven staat het op tafel.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{400 g}{witte asperges, geschild}\index[register]{Asperges}
  \ing{3}{koelverse lasagnevellen}
  \ing{8 plakken}{been- of achterham}\index[register]{Ham}
  \ing{200 ml}{crème fraîche}
  \ing{100 g}{geitenkaas, verkruimeld}\index[register]{Geitenkaas}
  \ing{1 tl}{Italiaanse kruiden}
  \ing{1 el}{bieslook, gehakt}
  \ingb{versgemalen peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Geen koelverse lasagne voorhanden? Kook gedroogde bladen beetgaar volgens de
verpakking en laat ze uitlekken over de rand van een vergiet.}
\begin{steps}
  \item Verwarm de oven voor op 200\,°C. Vet een kleine ovenschaal in.
  \item Snijd een stukje van de onderkant van de asperges. Kook ze 4--5 minuten
        en laat uitlekken.
  \item Klop de crème fraîche met de geitenkaas en Italiaanse kruiden tot een glad
        mengsel. Breng op smaak met peper.
  \item Leg een lasagnevel in de ovenschaal. Vouw 2--3 plakken ham dubbel en leg
        ze erop. Verdeel de helft van de asperges erover en schep een el van het
        crème-fraîchemengsel erop.
  \item Leg het tweede lasagnevel er schuin op. Herhaal met ham, de rest van de
        asperges en de helft van het resterende mengsel.
  \item Dek af met het laatste lasagnevel en bestrijk met de rest van de crème fraîche.
  \item Bak 10--15 minuten in het midden van de oven. Bestrooi met bieslook voor
        het serveren.
\end{steps}
\end{recipe}
