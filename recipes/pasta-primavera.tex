\begin{recipe}{Pasta Primavera}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans]{Hoofdgerecht · vegetarisch · Italiaans}
\index[register]{Pasta Primavera}
\index[register]{Groene asperges}
\index[register]{Peultjes}
\index[register]{Erwtjes}
\index[register]{Broccoli}
\index[register]{Pasta}

\meta{40 minuten}

\lettrine{D}{eze} pastaprimavera zit vol lentegroen: groene asperges,
peultjes, erwtjes en broccoli geven kleur en frisheid. De saus ontstaat
vanzelf, uit gesnipperde ui en het kookvocht van de pasta, dus room is niet
nodig. Kook de pasta zeer al dente, want hij gaart nog even mee in de pan.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{350 g}{pasta (tagliatelle of fusilli houden de saus goed vast)}\index[register]{Pasta}
  \ing{500 g}{groene asperges}\index[register]{Groene asperges}
  \ing{150 g}{peultjes (of sugarsnaps)}\index[register]{Peultjes}
  \ing{100 g}{erwtjes (vers of diepvries)}\index[register]{Erwtjes}
  \ing{300 g}{broccoli, in kleine roosjes}\index[register]{Broccoli}
  \ing{1}{ui}
  \ing{3 el}{extra vierge olijfolie}
  \ing{50 g}{Parmezaanse kaas, geraspt}\index[register]{Parmezaanse kaas}
  \ingb{zout}
  \ingb{versgemalen zwarte peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Sugarsnaps kunnen prima in plaats van peultjes, ze geven een vergelijkbare frisse bite.
Maak de tagliatelle zelf met het recept voor verse pasta (p.~\pageref{recipe:verse-pasta}).}
\begin{steps}
  \item Snijd de broccoli in kleine roosjes en blancheer ze 2 minuten in
        kokend water. Giet af en zet apart.
  \item Knak elke groene asperge aan de onderkant af: hij breekt vanzelf op de
        grens tussen het houtige en het zachte deel. Houd de punten apart en
        snijd de stelen in stukjes. 
  \item Snijd de peultjes bij en snijd ze schuin in ruitjes van ongeveer 2 cm. 
        Dop de erwtjes als ze vers zijn. Snipper de ui fijn.
  \item Fruit de ui in een ruime hapjespan met de olijfolie ongeveer een minuut. Voeg de
        aspergestelen en de peultjes toe. Blus af met wat water, doe het
        deksel op de pan en laat op middelhoog vuur zo'n 10 minuten
        sudderen. De peultjes en stelen worden zacht en vormen de basis van
        de saus.
  \item Voeg de erwtjes, de apart gehouden aspergepunten en de
        geblancheerde broccoli toe zodra de basis bijna klaar is. Kook nog
        5 minuten zonder deksel: de erwtjes moeten glanzend blijven en de
        punten zacht maar stevig. Plet eventueel een paar aspergestelen met
        een vork voor een romiger saus.
  \item Kook de pasta in gezouten water en giet hem zeer al dente af. Doe de
        pasta bij het groentemengsel en voeg een soeplepel pastakookwater
        toe. Bak op hoog vuur ongeveer een minuut, tot het zetmeel een
        zijdezachte saus vormt die alles bindt.
  \item Roer de parmezaanse kaas erdoor voor de romigheid en maak af met een
        scheut extra vierge olijfolie en versgemalen zwarte peper.
\end{steps}

\end{recipe}
