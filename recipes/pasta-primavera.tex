\begin{recipe}{Pasta Primavera}
\kicker{Hoofdgerecht · vegetarisch}
\index[register]{Pasta Primavera}
\index[register]{Asperges}
\index[register]{Peultjes}
\index[register]{Erwtjes}
\index[register]{Pasta}

\meta{35 minuten \dvd Voor 4 personen \dvd Italiaans}

\lettrine{D}{eze} pastaprimavera zit vol lentegroen: asperges, peultjes en
erwtjes geven kleur en frisheid. De saus ontstaat vanzelf, uit gepureerde ui
en het kookvocht van de pasta, dus room is niet nodig. Kook de pasta zeer al
dente, want hij gaart nog even mee in de pan.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{320 g}{pasta (tagliatelle of fusilli houden de saus goed vast)}\index[register]{Pasta}
  \ing{500 g}{asperges}\index[register]{Asperges}
  \ing{150 g}{peultjes}\index[register]{Peultjes}
  \ing{100 g}{erwtjes (vers of diepvries)}\index[register]{Erwtjes}
  \ing{1}{witte ui}
  \ing{3 el}{extra vierge olijfolie}
  \ingb{enkele takjes marjolein}
  \ingb{zout naar smaak}
  \ingb{versgemalen zwarte peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Geen fan van marjolein? Verse munt geeft de erwtjes ook een frisse smaak.}
\begin{steps}
  \item Maak de asperges schoon en verwijder de houtige uiteinden. Houd de
        punten apart en snijd de stelen in stukjes. Snijd de peultjes bij en
        snijd ze schuin in ruitjes van ongeveer 2 cm. Dop de erwtjes als ze
        vers zijn. Pureer de ui tot een gladde crème.
  \item Fruit de uiencrème in een ruime pan met de olie ongeveer een minuut.
        Voeg de aspergestelen, de peultjes en de marjolein toe. Blus af met
        wat water, doe het deksel op de pan en laat op middelhoog vuur zo'n
        10 minuten sudderen. De peultjes en stelen worden zacht en vormen de
        basis van de saus.
  \item Voeg de erwtjes en de apart gehouden aspergepunten toe zodra de basis
        bijna klaar is. Kook nog 5 minuten zonder deksel: de erwtjes moeten
        glanzend blijven en de punten zacht maar stevig. Plet eventueel een
        paar aspergestelen met een vork voor een romiger saus.
  \item Kook de pasta in gezouten water en giet hem zeer al dente af. Doe de
        pasta bij het groentemengsel en voeg een soeplepel pastakookwater
        toe. Bak op hoog vuur ongeveer een minuut, tot het zetmeel een
        zijdezachte saus vormt die alles bindt.
  \item Maak af met een scheut extra vierge olijfolie en versgemalen zwarte
        peper.
\end{steps}
\end{recipe}
