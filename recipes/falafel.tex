\begin{recipe}{Falafel}
\kicker[Hoofdgerecht,Vegan,Midden-Oosters]{Hoofdgerecht · vegan · Midden-Oosters}
\index[register]{Falafel}
\index[register]{Kikkererwten}
\index[register]{Shoarmabroodjes}
\index[register]{Paprika}
\index[register]{Tomaat}

\meta{Weken 12--24 uur + 1 uur bereiden \dvd 15 stuks \dvd Oven volgens verpakking}

\lettrine{F}{alafel} maak je van gedroogde kikkererwten, niet uit blik en niet
gekookt: alleen geweekt (12 tot 24 uur) binden ze zichzelf en bakken ze droog
en luchtig. Gekookte kikkererwten worden een natte puree die om extra bloem
vraagt. Vers gebakken en warm geserveerd in een broodje. Wil je het
doordeweeks snel houden, dan koop je de balletjes ook gewoon los.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{200 g}{kikkererwten, gedroogd}\index[register]{Kikkererwten}
  \ing{5}{shoarmabroodjes XL (Boons) of zelfgebakken, zie shoarmabroodjes}\index[register]{Shoarmabroodjes}
  \ing{½}{paprika, in reepjes}\index[register]{Paprika}
  \ing{2 teentjes}{knoflook}
  \ing{1}{kleine ui}
  \ing{1/2 tl}{komijn}
  \ing{1/2 tl}{gemalen koriander}
  \ing{1/4 tl}{bakpoeder}
  \ing{1}{tomaat, in blokjes}\index[register]{Tomaat}
  \ing{½ krop}{sla, grof gescheurd}
  \ing{1 pot}{verse peterselie}
  \ingb{snuf peper en zout}
  \ingb{snufje cayennepeper}
  \ingb{zonnebloemolie, om in te bakken}
  \ingb{knoflooksaus, kant-en-klaar of zelfgemaakt}
  \ingb{sambal, voor de liefhebber}
\end{ingredients}

\blockrule{Bereiding}
\tip{Zorg dat de olie goed heet is voor je de falafel erin legt. Is de olie
niet heet genoeg, dan zuigen de balletjes veel olie op in plaats van dat ze
krokant bakken.}

\medskip
\noindent\textit{Avond tevoren}
\begin{steps}
  \item Doe de kikkererwten in een zeef en spoel ze goed af. Laat ze 12 tot
        24 uur weken in een kom met ruim koud water: ze moeten flink in
        volume toenemen.
\end{steps}

\medskip
\noindent\textit{Dag zelf}
\begin{steps}
  \item Giet de geweekte kikkererwten af en dep ze goed droog.
  \item Was de paprika en de tomaat. Snijd de paprika in reepjes, de tomaat in blokjes en scheur de sla
        grof, dan heb je dat straks al klaarstaan.
  \item Doe de kikkererwten met de knoflook, ui, komijn, gemalen koriander,
        peterselie, cayennepeper, peper en zout in de keukenmachine (dus
        niet de paprika, tomaat, sla en zonnebloemolie, die heb je al apart
        klaarstaan) en maal ze door elkaar. De verse peterselie geeft de
        groengevlekte binnenkant en de frisheid van goede falafel. Doe dit
        niet te lang, anders wordt het een te dunne pasta: er moeten nog
        kleine korreltjes zichtbaar zijn.
  \item Zet het mengsel een half uur in de koelkast.
  \item Roer vlak voor het bakken het bakpoeder erdoor: een klassieke truc
        uit de Levant die de falafel lichter maakt. Vorm er balletjes van en
        druk ze iets plat. Is het mengsel toch nog te nat, voeg dan een
        beetje bloem toe.
\end{steps}

\medskip
\noindent\textit{Bakken en serveren}
\begin{steps}
  \item Verwarm de oven (boven- en onderwarmte) alvast voor op de
        temperatuur die op de broodjesverpakking staat.
  \item Verhit een laagje zonnebloemolie van ½ tot 1 cm goed heet in een
        koekenpan en bak de falafel in ongeveer 6 minuten goudbruin. Draai
        ze regelmatig om en laat ze daarna uitlekken.
  \item Leg de broodjes 2--4 minuten in de oven.
  \item Snijd de broodjes open en vul ze met de falafel, de verse groente,
        een flinke lepel knoflooksaus en eventueel wat sambal.
\end{steps}
