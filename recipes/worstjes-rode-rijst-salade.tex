\begin{recipe}{Worstjes, rode rijst en salade}
\label{recipe:worstjes-rode-rijst-salade}
\kicker[Hoofdgerecht,Vlees,Doordeweeks]{Hoofdgerecht · vlees · doordeweeks}
\index[register]{Worstjes, rode rijst en salade}
\index[register]{Chipolataworstjes}
\index[register]{Rode rijst}
\index[register]{Rijst}\index[register]{Tomaat}\index[register]{Knoflook}

\meta{35 minuten}

\lettrine{D}{eze} drie horen bij ons altijd samen: chipolataworstjes, rijst
gegaard in tomatenwater met ui, knoflook en rundsbouillon, en een frisse
salade of snackgroente ernaast. De rijst maakt zichzelf op laag vuur klaar
terwijl de worstjes bakken.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{5}{chipolataworstjes}\index[register]{Chipolataworstjes}
  \ing{170 g}{langkorrelige rijst}
  \ing{400 ml}{water (het leeggespoelde tomatenblik)}
  \ing{2 el}{olijfolie}
  \ing{1}{ui, fijngesnipperd}\index[register]{Ui}
  \ing{1}{knoflookteentje}\index[register]{Knoflook}
  \ing{1 blik (400\,g)}{gepelde tomaten}\index[register]{Tomaten}
  \ing{1}{rundsbouillonblokje}\index[register]{Bouillon}
  \ingb{snackgroente of groene salade, naar keuze}
\end{ingredients}

\blockrule{Bereiding}
\tip{Roer zo weinig mogelijk. Te veel roeren maakt de rijst papperig.}
\begin{steps}
  \item Bak de chipolataworstjes op middelhoog vuur in een koekenpan rondom
        bruin en gaar, ongeveer 10 minuten. Check de verpakking voor de
        precieze tijd.
  \item Snipper ondertussen de ui fijn en pers de knoflook.
  \item Zet ondertussen ook de rijst op: verhit de olijfolie in een hapjespan
        met dikke bodem en fruit de ui op middelhoog vuur tot glazig, ongeveer
        5 minuten.
  \item Voeg de knoflook toe en bak nog 1 minuut mee.
  \item Voeg de tomaten toe. Spoel het blik na met een blik water en doe ook
        dat water in de pan. Voeg het bouillonblokje toe en roer door.
  \item Breng aan de kook, voeg dan de rijst toe, zet het vuur laag en doe
        de deksel op de pan. Laat minstens 10 minuten garen, terwijl de
        worstjes in de andere pan verder bakken tot ze gaar zijn.
  \item Roer één keer voorzichtig door. Als de rijst te droog lijkt, voeg
        dan een klein scheutje water toe, niet meer dan dat.
  \item Serveer de worstjes en de rijst met snackgroente of de groene salade
        (p.~\pageref{recipe:groene-salade}) erbij.
\end{steps}

\end{recipe}
