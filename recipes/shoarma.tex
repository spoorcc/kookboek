\begin{recipe}{Shoarma}
\kicker[Hoofdgerecht,Vlees,Midden-Oosters]{Hoofdgerecht · vlees · Midden-Oosters}
\index[register]{Shoarma}
\index[register]{Kipshoarma}
\index[register]{Shoarmabroodjes}
\index[register]{Paprika}

\meta{30 minuten}

\lettrine{E}{en} maaltijd waarbij iedereen zijn eigen broodje vult. De broodjes
zijn kant-en-klaar te kopen (Boons XL shoarmabroodjes werken goed), of je bakt
ze zelf, zie het recept voor shoarmabroodjes. Zet alle groente en sauzen op
tafel en laat iedereen zijn eigen gang gaan.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{5}{shoarmabroodjes XL (Boons) of zelfgebakken}\index[register]{Shoarmabroodjes}
  \ing{650 g}{kipshoarma}\index[register]{Kipshoarma}
  \ing{½}{paprika, klein gesneden}\index[register]{Paprika}
  \ing{1}{ui}
  \ing{½ krop}{sla}
  \ing{1}{komkommer, in blokjes}
  \ingb{knoflooksaus}
  \ingb{whisky-cocktailsaus}
  \ingb{ketchup}
  \ingb{sambal}
\end{ingredients}

\blockrule{Bereiding}
\tip{Kinderen vinden het leuk om van de losse groente en sla nog een klein slaatje te maken.}
\begin{steps}
  \item Snijd de halve paprika klein, snipper de ui en snijd de komkommer in
        blokjes. Scheur wat sla grof.
  \item Verwarm de oven (boven- en onderwarmte) voor op de temperatuur die op de
        broodjesverpakking staat.
  \item Bak de kipshoarma in een koekenpan. Als het vlees aangebruind is, voeg de ui
        en paprika toe en bak nog een paar minuten mee.
  \item Leg de broodjes 2--4 minuten in de oven.
  \item Snijd de broodjes open. Zet de shoarma, de groente en de sauzen op tafel
        en laat iedereen zelf vullen.
\end{steps}
\end{recipe}
