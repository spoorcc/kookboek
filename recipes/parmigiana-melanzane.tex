\begin{recipe}{Parmigiana di Melanzane}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans]{Hoofdgerecht · vegetarisch · Italiaans}
\index[register]{Parmigiana di Melanzane}
\index[register]{Aubergine}
\index[register]{Mozzarella}
\index[register]{Parmezaanse kaas}

\meta{± 60 minuten \dvd Oven 180\,°C}

\lettrine{A}{ubergine} in laagjes met tomatensaus en kaas: dat is parmigiana in een
zin. De aubergines worden eerst gezouten en gebakken, daarna opgebouwd in een
ovenschaal met mozzarella en Parmezaan.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{4}{aubergines, geschild}\index[register]{Aubergine}
  \ing{2 blik (400\,g)}{gepelde tomaten}
  \ing{400 g}{mozzarella, in plakken}\index[register]{Mozzarella}
  \ing{100\,g}{Parmezaanse kaas, geraspt}\index[register]{Parmezaanse kaas}
  \ing{2--4}{knoflookteentjes, fijngehakt}
  \ing{3 takjes}{basilicum met blaadjes}
  \ingb{olijfolie}
  \ingb{zout en versgemalen peper}
  \ingb{keukenpapier}
\end{ingredients}

\blockrule{Bereiding}
\tip{Geschilde aubergine is zachter en minder herkenbaar. Kinderen die normaal
  nee zeggen tegen aubergine, eten het zo vaak zonder protest.}
\begin{steps}
  \item Schil de aubergines en snijd ze in plakken van ± 1 cm in de lengte. Strooi er voldoende zout over en
        leg ze op een bord, en als ze er allemaal opliggen een ander bord omgekeerd erbovenop.
  \item laat ze 10 minuten staan: dat trekt het vocht en de bitterheid eruit. 
  \item Dep ze droog met keukenpapier.
  \item Bak de aubergineplakken in een opgewarmde hapjespan in olijfolie aan beide kanten goudbruin op middelhoog vuur,
        ± 3--4 minuten per kant. Werk in porties zodat de pan niet te vol staat en de plakken bruinen
        in plaats van stomen. 
  \item Haal ze uit de pan en laat uitlekken op keukenpapier.
  \item Laat de plakken mozzarella intussen uitlekken op keukenpapier. Een
        waterige schotel is de meest gehoorde klacht bij parmigiana, en dat is
        bijna altijd de mozzarella.
  \item Verhit een scheut olijfolie in dezelfde hapjespan en fruit de knoflook 30 seconden op
        laag vuur, tot hij geurig is. Als de knoflook bruin wordt, wordt hij bitter.
  \item Voeg de gepelde tomaten toe en wat water en maak ze wat kleiner met een lepel. Voeg ook de basilicum toe.
        Breng op smaak met zout en peper en laat 15--20 minuten zacht koken tot de saus indikt. 
        Als je met een siliconen spatel een streep trekt over de bodem van de pan, blijft er een spoor zichtbaar,
        dan weet je dat de saus dik genoeg is.
  \item Verwarm de oven voor op 180\,°C (boven- en onderwarmte). 
  \item Verwijder de takjes basilicum, de smaak zit in de saus.
  \item Schep een laag tomatensaus op de bodem van een ovenschaal. Leg er aubergineplakken
        op, gevolgd door wat mozzarella en een beetje Parmezaan. Herhaal de lagen, maar
        gebruik de mozzarella alleen in de binnenste lagen: mozzarella bovenop verbrandt
        en wordt taai. Eindig daarom met alleen een laag tomatensaus en Parmezaan.
  \item Bak 25--30 minuten tot de kaas goudbruin is en de saus pruttelt.
  \item Laat minstens 10 minuten rusten voor het serveren; langer mag ook, parmigiana is
        op zijn best lauw, zo komt hij ook in Napels en op Sicilië op tafel.
\end{steps}
