\begin{recipe}{Parmigiana di Melanzane}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans]{Hoofdgerecht · vegetarisch · Italiaans}
\index[register]{Parmigiana di Melanzane}
\index[register]{Aubergine}
\index[register]{Mozzarella}
\index[register]{Parmezaanse kaas}

\meta{± 60 minuten \dvd Oven 180\,°C}

\lettrine{A}{ubergine} in laagjes met tomatensaus en kaas: dat is parmigiana in een
zin. De aubergines worden eerst gezouten en gebakken, daarna opgebouwd in een
ovenschaal met mozzarella en Parmezaan.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{4}{aubergines, geschild}\index[register]{Aubergine}
  \ing{2 blik (400\,g)}{gepelde tomaten}
  \ing{400 g}{mozzarella, in plakken}\index[register]{Mozzarella}
  \ing{100\,g}{Parmezaanse kaas, geraspt}\index[register]{Parmezaanse kaas}
  \ing{2--4}{knoflookteentjes, fijngehakt}
  \ingb{olijfolie}
  \ingb{zout en versgemalen peper}
  \ingb{basilicum}
\end{ingredients}

\blockrule{Bereiding}
\tip{Geschilde aubergine is zachter en minder herkenbaar. Kinderen die normaal
  nee zeggen tegen aubergine, eten het zo vaak zonder protest.}
\begin{steps}
  \item Snijd de geschilde aubergines in plakken van ± 1 cm. Strooi er zout over en
        laat 10 minuten staan: dat trekt het vocht en de bitterheid eruit. Dep ze
        droog met keukenpapier.
  \item Bak de aubergineplakken in een koekenpan in olijfolie aan beide kanten goudbruin, ± 3--4 minuten
        per kant. Werk in porties zodat de pan niet te vol staat en de plakken bruinen
        in plaats van stomen. 
  \item Haal ze uit de pan en laat uitlekken op keukenpapier.
  \item Verhit een scheut olijfolie in een pan en fruit de knoflook kort op laag vuur.
  \item Voeg de gepelde tomaten toe en maak ze wat kleiner met een lepel. Breng op smaak
        met zout en peper en laat 15--20 minuten zacht koken tot de saus indikt.
        Roer op het eind eventueel basilicum erdoor.
  \item Verwarm de oven voor op 180\,°C (boven- en onderwarmte). 
  \item Schep een laag tomatensaus op de bodem van een ovenschaal. Leg er aubergineplakken
        op, gevolgd door wat mozzarella en een beetje Parmezaan. Herhaal de lagen tot 
        alles op is en eindig met kaas.
  \item Bak 25--30 minuten tot de kaas goudbruin is en de saus pruttelt.
  \item Laat 10 minuten rusten voor het serveren.
\end{steps}
\end{recipe}
