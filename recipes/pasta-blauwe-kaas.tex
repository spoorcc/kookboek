\begin{recipe}{Pasta met blauwe kaas en champignons}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans]{Hoofdgerecht · vegetarisch · Italiaans}
\index[register]{Pasta met blauwe kaas en champignons}
\index[register]{Blauwe kaas}
\index[register]{Champignons}
\index[register]{Pasta}

\meta{25 minuten}

\lettrine{E}{en} romige saus van crème fraîche en blauwe kaas met gebakken champignons
en rode ui. Verse tagliatelle werkt hier goed bij, maar andere verse pasta kan ook.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{200 g}{gorgonzola (of andere blauwe kaas)}\index[register]{Blauwe kaas}
  \ing{250 g}{(kastanje)champignons, in plakken}\index[register]{Champignons}
  \ing{500 g}{verse tagliatelle}\index[register]{Pasta}
  \ing{1}{rode ui, gesnipperd}\index[register]{Rode ui}
  \ing{1}{knoflookteentje, fijngehakt}
  \ing{200 g}{crème fraîche}
  \ingb{olijfolie of boter}
  \ingb{handvol verse peterselie, gehakt}
  \ingb{versgemalen peper en zout}
\end{ingredients}

\blockrule{Bereiding}
\tip{Een handvol verse spinazie erdoor aan het einde geeft wat kleur en groente erbij.
Maak de tagliatelle zelf met het recept voor verse pasta (p.~\pageref{recipe:verse-pasta}),
die romige saus verdient het.}
\begin{steps}
  \item Zet een pan water met zout op voor de pasta. Bak ondertussen de
        champignons in olijfolie op hoog vuur goudbruin: zo bruinen ze in
        plaats van vocht te lossen. Zet ze apart.
  \item Zet het vuur laag. Fruit de rode ui en knoflook 3 minuten in wat boter of
        olijfolie tot de ui glazig is.
  \item Voeg de crème fraîche toe en laat warm worden. Verkruimel de blauwe kaas erdoor
        en roer tot een gladde saus. Breng op smaak met peper en zout.
  \item Kook de verse pasta al dente volgens de verpakking. Bewaar een kop
        kookvocht voor je afgiet.
  \item Roer de champignons door de saus. Giet de pasta af en schep hem met een
        scheut kookvocht door de saus in de pan: crème fraîche en gorgonzola
        worden samen snel dik, het kookvocht houdt de saus zijdeachtig in plaats
        van plakkerig.
  \item Serveer met gehakte peterselie.
\end{steps}
\end{recipe}
