\begin{recipe}{Pasta met Marcella Hazans tomatensaus}
\kicker[Hoofdgerecht,Vegetarisch,Italiaans,Doordeweeks]{Hoofdgerecht · vegetarisch · Italiaans · doordeweeks}
\index[register]{Pasta met Marcella Hazans tomatensaus}
\index[register]{Tomatensaus}
\index[register]{Tomaat}
\index[register]{Boter}

\meta{55 minuten}

\lettrine{D}{it} klassieke recept uit Marcella Hazans boek \textit{Essentials
of Classic Italian Cooking} laat zien hoeveel smaak je met drie ingrediënten
kunt halen. De ui geeft milde zoetheid en de boter maakt de saus zacht en rond.

\blockrule{Ingrediënten}
\begin{ingredients}
  \ing{1 blik (800\,g)}{hele gepelde tomaten (bij voorkeur San Marzano)}\index[register]{Tomaat}
  \ing{1 middelgrote}{ui, gepeld en doormidden gesneden}
  \ing{5 el}{ongezouten boter}\index[register]{Boter}
  \ing{350 g}{pasta (spaghetti, penne of rigatoni)}\index[register]{Pasta}
  \ingb{zout}
  \ingb{geraspte Parmezaanse kaas}
  \ingb{versgemalen zwarte peper}
\end{ingredients}

\blockrule{Bereiding}
\tip{Gooi de gehalveerde ui niet weg. Fijngehakt en door de saus geroerd geeft
hij een zachte, zoete bite. Ook lekker met verse pasta
(p.~\pageref{recipe:verse-pasta}). De saus laat zich goed invriezen.}
\begin{steps}
  \item Doe de tomaten met sap, de boter, de halve ui en een snufje zout in een
        pan met dikke bodem.
  \item Breng aan de kook op middelhoog vuur. Plet de tomaten licht met een
        lepel of spatel en roer af en toe.
  \item Zet het vuur laag en laat de saus ongeveer 45 minuten ongedekt zachtjes
        pruttelen. De saus wordt dikker en er verschijnen vetdruppels aan de
        oppervlakte. Roer af en toe. Haal dan de halve ui eruit: wie wil kan
        hem fijnhakken en door de saus roeren. Proef en voeg zo nodig extra
        zout toe.
  \item Kook ondertussen de pasta al dente in ruim gezouten water. Bewaar een
        kop kookvocht, giet de pasta af en schep hem met een scheut van dat
        kookvocht door de saus in de pan: zo plakt de saus aan de pasta in
        plaats van eronder te lekken.
  \item Serveer met geraspte Parmezaanse kaas en versgemalen peper.
\end{steps}
\end{recipe}
